20  * This is one of the built-in \ref aworx::lib::boxing::Interface "Interface" classes of

27  * For non-array types, this default implementation checks if both boxes share the same type and

67 /** ************************************************************************************************
68  * This is one of the built-in \ref aworx::lib::boxing::Interface "Interface" classes of
69  * <b>ALib %Boxing</b>. This interface is invokable with all boxable types, as a
70  * default implementation is registered (see
71  * \ref aworx::lib::boxing::Boxer::DefineDefaultInterface "Boxer::DefineDefaultInterface").
72  *
73  * The availability of this interface is the basis to implement the concept of
74  * <em>"nullable types"</em>, e.g. pointer types or custom types that (already) support the concept
75  * of being \e nullable.
76  *
77  * This default implementation returns \c true if the raw value of box data equals \c 0, otherwise
78  * \c false. This gives a good default behaviour for all boxable types
79  * stored as pointers to the original object.
80  *
81  * Derived class \ref aworx::lib::boxing::IIsNull_false "IIsNull_false" is provided which returns
82  * constant value \c false. This may be registered with custom types that are supposed to never be
83  * \e nulled.
84  * The default implementation of
85  * \ref alib_namespace_boxing_types_ftypes "boxing of fundamental types" will add this
86  * specialization to integer and floating point types. Hence boxed values of these types will
87  * never be considered \e nulled, even if they represent value \c 0.
88  **************************************************************************************************/

147 /** ************************************************************************************************
148  * This is one of the built-in \ref aworx::lib::boxing::Interface "Interface" classes of
149  * <b>ALib %Boxing</b>. This interface is invokable with all boxable types, as a
150  * default implementation is registered (see
151  * \ref aworx::lib::boxing::Boxer::DefineDefaultInterface "Boxer::DefineDefaultInterface").
152  *
153  * The availability of this interface is the basis to provide the concept of
154  * <em>"emptiness"</em> of boxed objects.
155  * E.g. array and string types are candidates where such definition makes sense.
156  *
157  * For array types (see \ref aworx::lib::boxing::ArrayBoxerT "ArrayBoxerT"), this default
158  * implementation returns \c true if method
159  * \ref aworx::lib::boxing::Box::Length "Box::Length" returns \c 0 for \p box.
160  *
161  * For non-array types this default implementation returns the result of the invocation of
162  * interface #IIsNull on \p box.
163  * This way, by default the concept of <em>"emptiness of a value"</em> is equal
164  * to the concept of <em>"nullable values"</em> for non-array types.
165  * In combination with the fact that the default implementation of
166  * \ref alib_namespace_boxing_types_ftypes "boxing of fundamental types" will add a specialization
167  * of \b %IIsNull that returns constant false to integer and floating point types, such types
168  * will never be considered "empty", even if they represent value \c 0.
169  **************************************************************************************************/

172  public:

175  * Declares a static singleton used for the default interface registration and returns that.
176  * \note We do not inherit template class Singleton here, because each of our descendants
177  * needs to do this as well, which would lead to some confusion.
178  * @return A singleton instance of myself.
179  */
180  static IIsEmpty* GetDefaultSingleton()
181  {
182  static IIsEmpty me;
183  return &me;
184  }
185 
186  /**
187  * Constructor providing our runtime type to the parent
188  */

191  /**
192  * Overridable method that evaluates if the provided box object is \e "empty".
