25 /** ************************************************************************************************
26  * The class is useful to store and pass around enum values of arbitrary C++ scoped enum types in
27  * a type-safe way.
28  * It is implemented to ease the use of C++ 11 scoped enums in situations where programmers
29  * otherwise tend to "fall back" to non-scoped (non type-safe enum types). This is the case, when
30  * enum elements of different types need to be processed.
31  *
32  * In the constructor, enum elements of arbitrary type are accepted.
33  * The enum element is \ref aworx::lib::boxing "boxed" and thus runtime type information is added.
34  * Having the "original" element stored in a box, has the following advantages:
35  * - The box is useful to check the encapsulated enum type at runtime
36  * using \alib{boxing,Box::IsType}.
37  * - The original element can be unboxed using #Get in a type-safe way. (This method is a shortcut
38  * to \alib{boxing,Box::Unbox,Box::Unbox<TEnum>()}
39  * - The underlying integer type value of the enum element can be received easily.
40  *
41  * \note
42  * The implementation of this class using \ref aworx::lib::boxing "ALib Boxing" introduces a
43  * small memory overhead (usually 8 bytes per instance on 64-bit system), in respect to a
44  * possible alternative "direct" implementation. This is due to the fact that boxing allows
45  * one dimensional array types to be boxed as well as scalar types - which is never the case
46  * with this class.
47  *
48  * \note
49  * The advantage of doing so, is that objects of this type (which is derived from class \b %Box)
50  * may be used as if the enum element was just boxed, instead of being emplaced in this
51  * class. This way, custom boxing interfaces that exist for enum types, may be used with instances
52  * of this class.
53  *
54  * \note
55  * For example, this class already specializes \alib{strings,T_Apply} which appends a string
56  * representation of the enum element to an \b %AString. Internally, built-in boxing interface
57  * \alib{strings::boxing,IApply} is used.<br>
58  * This means, that if this interface is set for the boxed C++ 11 enum type already, then
59  * objects of this type can be applied without further (setup) effort of the user code.
60  **************************************************************************************************/
61 struct Enum : public Box
62 {
63  /** ********************************************************************************************
64  * Default constructor.
65  **********************************************************************************************/
66  Enum() :Box(nullptr) {}
67 
68  /** ********************************************************************************************
69  * Implicit constructor which accepts arbitrary elements of scoped or non-scoped enum types.
70  *
71  * @param element The external (user specific) enumeration element.
72  * @tparam TEnum The external (user specific) enumeration type.
73  * @tparam TEnableIf Selects this constructor only for enum types.
74  **********************************************************************************************/
75  template<typename TEnum,
76  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
77  constexpr Enum( TEnum element )
78  : Box( element ) {}
79 
80  /** ********************************************************************************************
81  * This is a shortcut to \alib{boxing,Box::Unbox,Box::Unbox<TEnum>()} to retrieve the
82  * original enum element in a type safe way.
83  *
84  * Prior to invoking this, the boxed type can be checked with #IsType. If the wrong type
85  * is tried to be received, an \alib assertion is raised.
86  *
87  * @tparam TEnum The external (user specific) enumeration type.
88  * @tparam TEnableIf Selects this method only for enum types.
89  * @return The underlying integer value of the encapsulated enum element.
90  **********************************************************************************************/
91  template<typename TEnum,
92  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
93  TEnum Get() const { return Unbox<TEnum>(); }
94 
95  /** ********************************************************************************************
96  * Returns the underlying integer value of the original enum element.
97  *
98  * \see Namespace function \alib{lang,EnumValue}.
99  *
100  * @return The underlying integer value.
101  **********************************************************************************************/
102  inline
103  int Value() const { return static_cast<int>( UnboxRaw() ); }
104 
105  /** ********************************************************************************************
106  * Checks if the this instance has an enum element of type \p TEnum stored.
107  *
108  * @tparam TEnum The external (user specific) enumeration type.
109  * @tparam TEnableIf Selects this method only for enum types.
110  * @return \c true if the encapsulated enum type of type \p TEnum, otherwise \c false.
111  **********************************************************************************************/
112  template<typename TEnum,
113  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
114  inline
115  bool IsType() const { return IsType<TEnum>(); }
116 
117  /** ********************************************************************************************
118  * Checks if the given other object encapsulates the same enumeration type.
119  *
120  * @param lhs The right hand side argument of the comparison.
121  * @return \c true if the encapsulated enum type of this object is the same as the one
122  * encapsulated in \p lhs, otherwise \c false.
123  **********************************************************************************************/
124  inline
125  bool IsSameType(Enum const& lhs) const
126  {
127  return GetTypeInfo<0>() == lhs.GetTypeInfo<0>();
128  }
129 
130  /** ********************************************************************************************
131  * Comparison operator.
132  *
133  * @param lhs The right hand side argument of the comparison.
134  * @return \c true if this object equals \p lhs, \c false otherwise.
135  **********************************************************************************************/
136  inline
137  bool operator==(Enum const& lhs) const { return this->Box::operator==(lhs); }
138 
139  /** ********************************************************************************************
140  * Comparison operator.
141  *
142  * @param lhs The right hand side argument of the comparison.
143  * @return \c true if this object does not equal \p lhs, \c false otherwise.
144  **********************************************************************************************/
145  inline
146  bool operator!=(Enum const& lhs) const { return this->Box::operator!=(lhs); }
147 
148  /** ********************************************************************************************
149  * Comparison operator.
150  *
151  * @param lhs The right hand side argument of the comparison.
152  * @return \c true if this object equals \p lhs, \c false otherwise.
153  **********************************************************************************************/
154  template<typename TEnum,
155  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
156  inline
157  bool operator==(TEnum lhs) const { return Value() == EnumValue( lhs )
158  && GetTypeInfo<0>() == typeid( TEnum ); }
159 
160  /** ********************************************************************************************
161  * Comparison operator.
162  *
163  * @param lhs The right hand side argument of the comparison.
164  * @return \c true if this object does not equal \p lhs, \c false otherwise.
165  **********************************************************************************************/
166  template<typename TEnum,
167  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
168  inline
169  bool operator!=(TEnum lhs) const { return Value() != EnumValue( lhs )
170  || GetTypeInfo<0>() != typeid( TEnum ); }
171 
172 
173  /** ********************************************************************************************
174  * Comparison operator. The sort order is primarily determined by the enum types that
175  * are boxed. If those are the same, then the underlying integer value of the enum elements
176  * is compared.
177  *
178  * This leads to a nested sort order, with the type information being the outer order and
179  * the integer value of the enum being the inner one.
180  *
181  * \note
182  * It is a matter of the compiler how the outer sort of types is performed and thus this
183  * can not be determined by the user code.
184  *
185  *
186  * @param lhs The right hand side argument of the comparison.
187  * @return If the encapsulated type of this instance is the same as that of \p lhs, this
188  * methods \c true if #Value() of this object is smaller than the one of \p lhs and
189  * otherwise \c false. If the types are not the same, than the result is dependent
190  * on the tool chain used for compiling \alib.
191  **********************************************************************************************/
192  inline
193  bool operator< (Enum const& lhs) const
194  {
195  return ( std::type_index( GetTypeInfo<0>() )
196  < std::type_index(lhs.GetTypeInfo<0>() ) )
197  || ( GetTypeInfo<0>() == lhs.GetTypeInfo<0>()
198  && Value() < lhs.Value() );
199  }
