23 /** ************************************************************************************************
24  * The class is useful to store and pass around enum values of arbitrary C++ scoped enum types in
25  * a type-safe way.
26  * It is implemented to ease the use of C++ 11 scoped enums in situations where programmers
27  * otherwise tend to "fall back" to non-scoped (non type-safe enum types). This is the case, when
28  * enum elements of different types need to be processed.
29  *
30  * In the constructor, enum elements of arbitrary type are accepted.
31  * The enum element is \ref aworx::lib::boxing "boxed" and thus runtime type information is added.
32  * Having the "original" element stored in a box, has the following advantages:
33  * - The box is useful to check the encapsulated enum type at runtime
34  * using \alib{boxing,Box::IsType}.
35  * - The original element can be unboxed using #Get in a type-safe way. (This method is a shortcut
36  * to \alib{boxing,Box::Unbox,Box::Unbox<TEnum>()}
37  * - The underlying integer type value of the enum element can be received easily.
38  *
39  * \note
40  * The implementation of this class using \ref aworx::lib::boxing "ALib Boxing" introduces a
41  * small memory overhead (usually 8 bytes per instance on 64-bit system), in respect to a
42  * possible alternative "direct" implementation. This is due to the fact that boxing allows
43  * one dimensional array types to be boxed as well as scalar types - which is never the case
44  * with this class.
45  *
46  * \note
47  * The advantage of doing so, is that the internal box element (which is exposed as a public
48  * field) may be used as if the enum element was just boxed, instead of being emplaced in this
49  * class. This way, custom boxing interfaces that exist for enum types, may be used with instances
50  * of this class.
51  *
52  * \note
53  * For example, this class already specializes \alib{strings,T_Apply} which appends a string
54  * representation of the enum element to an \b %AString. Internally, built-in boxing interface
55  * \alib{strings::boxing,IApply} is used.<br>
56  * This means, that if this interface is set for the boxed C++ 11 enum type already, then
57  * objects of this type can be applied without further (setup) effort of the user code.
58  **************************************************************************************************/
59 struct Enum
60 {
61  /**
62  * The enumeration element as a \ref aworx::lib::boxing "boxed" value. Although the member
63  * name starts with a lower case letter, the value is still public (Bauhaus style).
64  * However, usually the element does not need to be accessed directly.
65  */
66  const Box box;
67 
68  /** ********************************************************************************************
69  * Deleted default constructor.
70  **********************************************************************************************/
71  Enum()= delete;
72 
73  /** ********************************************************************************************
74  * Implicit constructor which accepts arbitrary elements of scoped or non-scoped enum types.
75  *
76  * @param element The external (user specific) enumeration element.
77  * @tparam TEnum The external (user specific) enumeration type.
78  * @tparam TEnableIf Selects this constructor only for enum types.
79  **********************************************************************************************/
80  template<typename TEnum,
81  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
82  constexpr Enum( TEnum element )
83  : box( element ) {}
84 
85  /** ********************************************************************************************
86  * This is a shortcut to \alib{boxing,Box::Unbox,Box::Unbox<TEnum>()} to retrieve the
87  * original enum element in a type safe way.
88  *
89  * Prior to invoking this, the boxed type can be checked with #IsType. If the wrong type
90  * is tried to be received, an \alib assertion is raised.
91  *
92  * @tparam TEnum The external (user specific) enumeration type.
93  * @tparam TEnableIf Selects this method only for enum types.
94  * @return The underlying integer value of the encapsulated enum element.
95  **********************************************************************************************/
96  template<typename TEnum,
97  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
98  TEnum Get() const { return box.Unbox<TEnum>(); }
99 
100  /** ********************************************************************************************
101  * Returns the underlying integer value of the original enum element.
102  *
103  * \see Namespace function \alib{lang,EnumValue}.
104  *
105  * @return The underlying integer value.
106  **********************************************************************************************/
107  inline
108  int Value() const { return static_cast<int>( box.UnboxRaw() ); }
109 
110  /** ********************************************************************************************
111  * Checks if the this instance has an enum element of type \p TEnum stored.
112  *
113  * @tparam TEnum The external (user specific) enumeration type.
114  * @tparam TEnableIf Selects this method only for enum types.
115  * @return \c true if the encapsulated enum type of type \p TEnum, otherwise \c false.
116  **********************************************************************************************/
117  template<typename TEnum,
118  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
119  inline
120  bool IsType() const { return box.IsType<TEnum>(); }
121 
122  /** ********************************************************************************************
123  * Checks if the given other object encapsulates the same enumeration type.
124  *
125  * @param lhs The right hand side argument of the comparison.
126  * @return \c true if the encapsulated enum type of this object is the same as the one
127  * encapsulated in \p lhs, otherwise \c false.
128  **********************************************************************************************/
129  inline
130  bool IsSameType(Enum const& lhs) const
131  {
132  return box.GetTypeInfo<0>() == lhs.box.GetTypeInfo<0>();
133  }
134 
135  /** ********************************************************************************************
136  * Comparison operator.
137  *
138  * @param lhs The right hand side argument of the comparison.
139  * @return \c true if this object equals \p lhs, \c false otherwise.
140  **********************************************************************************************/
141  inline
142  bool operator==(Enum const& lhs) const { return box == lhs.box; }
143 
144  /** ********************************************************************************************
145  * Comparison operator.
146  *
147  * @param lhs The right hand side argument of the comparison.
148  * @return \c true if this object does not equal \p lhs, \c false otherwise.
149  **********************************************************************************************/
150  inline
151  bool operator!=(Enum const& lhs) const { return box != lhs.box; }
152 
153  /** ********************************************************************************************
154  * Comparison operator.
155  *
156  * @param lhs The right hand side argument of the comparison.
157  * @return \c true if this object equals \p lhs, \c false otherwise.
158  **********************************************************************************************/
159  template<typename TEnum,
160  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
161  inline
162  bool operator==(TEnum lhs) const { return Value() == EnumValue( lhs )
163  && box.GetTypeInfo<0>() == typeid( TEnum ); }
164 
165  /** ********************************************************************************************
166  * Comparison operator.
167  *
168  * @param lhs The right hand side argument of the comparison.
169  * @return \c true if this object does not equal \p lhs, \c false otherwise.
170  **********************************************************************************************/
171  template<typename TEnum,
172  typename TEnableIf = typename std::enable_if<std::is_enum<TEnum>::value>::type >
173  inline
174  bool operator!=(TEnum lhs) const { return Value() != EnumValue( lhs )
175  || box.GetTypeInfo<0>() != typeid( TEnum ); }
176 
177 
178  /** ********************************************************************************************
179  * Comparison operator. The sort order is primarily determined by the enum types that
180  * are boxed. If those are the same, then the underlying integer value of the enum elements
181  * is compared.
182  *
183  * This leads to a nested sort order, with the type information being the outer order and
184  * the integer value of the enum being the inner one.
185  *
186  * \note
187  * It is a matter of the compiler how the outer sort of types is performed and thus this
188  * can not be determined by the user code.
189  *
190  *
191  * @param lhs The right hand side argument of the comparison.
192  * @return If the encapsulated type of this instance is the same as that of \p lhs, this
193  * methods \c true if #Value() of this object is smaller than the one of \p lhs and
194  * otherwise \c false. If the types are not the same, than the result is dependent
195  * on the tool chain used for compiling \alib.
196  **********************************************************************************************/
197  inline
198  bool operator< (Enum const& lhs) const
199  {
200  return ( std::type_index( box.GetTypeInfo<0>() )
201  < std::type_index(lhs.box.GetTypeInfo<0>() ) )
202  || ( box.GetTypeInfo<0>() == lhs.box.GetTypeInfo<0>()
203  && Value() < lhs.Value() );
204  }

206 };

231 template<> struct T_Apply<Enum> : public std::true_type

233  /**
234  * If the enum type encapsulated in \p src has boxing interface \alib{strings::boxing,IApply}
235  * defined, then this interface is invoked.
236  *
237  * If the interface is not available, in debug compilations, the output is
238  *
239  * typename(<int value>)
240  *
241  * where \e typename is the name of the type that was encapsulated on construction of \p source.
242  *
243  * In release compilations it is:
244  *
245  * Enum(<int value>)
246  * because printable run-time type information is not available then.
247  *
248  * @param target The object that \b Apply was invoked on.
249  * @param src The value to write.
250  * @return Constant \c 1.
251  */
252  static integer Apply( AString& target, const aworx::lib::lang::Enum& src )
253  {
254  if( src.box.HasInterface<aworx::lib::strings::boxing::IApply>() )
255  src.box.Invoke<aworx::lib::strings::boxing::IApply>( target );
256  else
257  target << ALIB_REL_DBG( "Enum", src.box.GetTypeInfo() )
258  << '(' << src.Value() << ')';
259 
260  return 1;
261  }
262 };
