50 /** ************************************************************************************************
51  * This struct is used as the default implementation of a helper type of class
52  * \ref aworx::lib::util::StringTree "StringTree".
53  * Class \b %StringTree uses this class to allocate memory to store character sequences for child
54  * node names when child nodes are created.
55  *
56  * This default implementation simply allocates the memory using \c new and copies the data.
57  * The interface of methods #allocate and #free provide the value of the node. There might
58  * be implementations where the value has already a copy of the child name allocated (or wishes
59  * to do). In this case, that string can be returned.
60  *
61  * See also
62  * \ref aworx::lib::util::StringTreeConstKeyAllocator "StringTreeConstKeyAllocator".
63  *
64  * @tparam T The value type of elements stored in the nodes of the tree.
65  **************************************************************************************************/
66 template<typename T> struct StringTreeDefaultKeyAllocator
67 {
68  /**
69  * Invoked when child nodes are created. Needs to return a 'safe' copy of the child name
70  * string, ready to be stored as permanent key value.
71  * @param childName The name of child, hence the key string.
72  * @param value The child value that is about to be inserted into the \b StringTree.
73  * @return A version of \p childName that is safely allocated.
74  */
75  static String allocate( const String& childName, T& value )
76  {
77  (void) value;
78  char* buffer= new char[ childName.Length() ];
79  memcpy( buffer, childName.Buffer(), static_cast<size_t>( childName.Length() ) );
80  return String( buffer, childName.Length() );
81  }
82 
83  /**
84  * Invoked when child nodes are deleted. Needs to free memory that had been allocated in a
85  * corresponding call to #allocate.
86  * @param childName The key string used to previously store the node of \p value.
87  * @param value The child value that is about to be deleted from \b StringTree.
88  */
89  static void free( String& childName, T& value )
90  {
91  (void) value;
92  delete[] childName.Buffer();
93  }
94 };

131 /** ************************************************************************************************
132  * This class manages an object of internal type
133  * \ref aworx::lib::util::StringTree::Node "Node", which consists of
134  * - A value of custom (template) type \p T,
135  * - A pointer to a parent \b %Node and
136  * - A hash table which recursively stores pointers to \b Node objects, referred to by
137  * a string-type key.
138  *
139  * This recursive data structure is what comprises a graph with named edges. As this object does
140  * not allow insertion of nodes with circular dependencies, the graph is of tree type.
141  * The way from the root node to the leafs usually is called "path" and the class incorporates
142  * functionality to work with string representations of such paths where names of edges are
143  * concatenated and separated by a special separation character.
144  * Now, this explains why this class is called \b %StringTree.
145  *
146  * The search and creation of tree nodes by using aforementioned path strings, is very similar to
147  * what is well known from addressing files and folders in file systems.
148  * The difference is however, that this class does not differentiate between 'folders' and 'files',
149  * hence between 'nodes' and 'leafs'. Every node has the same value type \p T and may or may not
150  * have child nodes.
151  * If such differentiation - or other semantics - is wanted, this may be modeled in custom
152  * attributes provided in template type \p T.
153  *
154  * The internal node structure is not exposed publically and the class has no direct interface to
155  * manipulate nodes. Instead, the interface is defined by three public inner types.
156  * Those are:
157  * - \ref aworx::lib::util::StringTree::Cursor "Cursor",
158  * - \ref aworx::lib::util::StringTree::StdIterator "StdIterator" (C++ version only) and
159  * - \ref aworx::lib::util::StringTree::Walker "Walker".
160  *
161  * and are explained in the following paragraphs.
162  *
163  * \anchor cpp_alib_containers_stringtree_cursor
164  * <b>Inner class %Cursor: Inserting, Retrieving And Deleting Nodes</b><br>
165  * The class has no direct interface to manipulate nodes. The main interface into
166  * objects of this type is defined by public, inner type
167  * \ref aworx::lib::util::StringTree::Cursor "Cursor". The only way to create an
168  * initial cursor is with method #Root, which creates a cursor object referring to the
169  * root node of the tree. With this, string names and complete paths can be used to
170  * move the cursor along existing nodes of the tree or to create new child nodes or a whole
171  * path of children at once.
172  * Class cursor is quite lightweight as it contains just two pointers (to the \b %StringTree and the current node).
173  * Hence, cursors can be cloned, assigned and passed around very efficiently.
174  *
175  * Once a cursor is created it can traverse over the tree nodes using methods
176  * - \ref aworx::lib::util::StringTree::Cursor::MoveToChild "Cursor::MoveToChild",
177  * - \ref aworx::lib::util::StringTree::Cursor::MoveToParent "Cursor::MoveToParent",
178  * - \ref aworx::lib::util::StringTree::Cursor::MoveTo "Cursor::MoveTo" and
179  * - \ref aworx::lib::util::StringTree::Cursor::MoveToExistingPart "Cursor::MoveToExistingPart".
180  *
181  * For the creation of new child nodes or a complete path of such, methods
182  * - \ref aworx::lib::util::StringTree::Cursor::MoveToChildCreate "Cursor::MoveToChildCreate" and
183  * - \ref aworx::lib::util::StringTree::Cursor::MoveToAndCreateNonExistingPart "Cursor::MoveToAndCreateNonExistingPart"
184  *
185  * are provided. Then, two methods for erasing nodes exist:
186  * - \ref aworx::lib::util::StringTree::Cursor::DeleteChild "Cursor::DeleteChild" and
187  * - \ref aworx::lib::util::StringTree::Cursor::DeleteChildren "Cursor::DeleteChildren"
188  *
189  * Finally, class cursor can also be used like an iterator. But it explicitly and for good reason
190  * does \b not follow the concepts provided with <c>C++ std::iterator</c> types. The iterating
191  * feature comes with methods
192  * - \ref aworx::lib::util::StringTree::Cursor::MoveToParentUnchecked "Cursor::MoveToParentUnchecked" and
193  * - \ref aworx::lib::util::StringTree::Cursor::IsValid "Cursor::IsValid".
194  *
195  * The first transitions the cursor to refer to the parent node. Using this method is (almost)
196  * the only way how a cursor can get invalidated. Hence a simple loop can be written that visits
197  * all nodes from a starting node, up to the root of the tree until \b %Cursor::IsValid evaluates
198  * to \c false.
199  *
200  * <b>Inner class iterator: A std::iterator implementation</b><br>
201  * A second inner class is provided with
202  * \ref aworx::lib::util::StringTree::StdIterator "StdIterator". It implements the concept
203  * [C++ InputIterator](http://en.cppreference.com/w/cpp/concept/InputIterator) and due to the
204  * implementation of methods
205  * - \ref aworx::lib::util::StringTree::begin "StringTree::begin" and
206  * - \ref aworx::lib::util::StringTree::end "StringTree::end",
207  *
208  * this type can be used together with <em>C++ 11 ranged-for
209  * loops</em> as well as \c std::for_each. And these loops indicate already the typical use of
210  * class \b %StdIterator: The implementation of algorithms that need to visit all nodes of a tree or a
211  * branch in an depth-first order, with unsorted child order.
212  *
213  * With methods
214  * - \ref aworx::lib::util::StringTree::Cursor::begin "Cursor::begin" and
215  * - \ref aworx::lib::util::StringTree::Cursor::end "Cursor::end",
216  *
217  * StdIterator objects may be initialized using a different start node than the root, which
218  * is chosen with the direct \b %StringTree interface.
219  * Existing \b %StdIterator object can also be reused: They can be freshly initialized with methods
220  *
221  * - \ref aworx::lib::util::StringTree::Cursor::SetIterator "Cursor::SetIterator" and
222  * - \ref aworx::lib::util::StringTree::Walker::SetIterator "Walker::SetIterator".
223  *
224  * Please also note the paragraph about recursive iteration below.
225  *
226  * Class \ref aworx::lib::util::StringTree::StdIterator "StdIterator" is quite lightweight.
227  * Nevertheless, if frequently used, it might be advisable
228  * to reuse an exiting \b %StdIterator, because internal buffers are allocated then only once. Of-course,
229  * re-using an instance is a showstopper for convenient <em>C++ 11 ranged-for loops</em>.
230  *
231  * <b>Inner class %Walker: Sorted Iterations</b><br>
232  * Finally, with
233  * \ref aworx::lib::util::StringTree::Walker "Walker", a third \b %StdIterator class is exposed.
234  * The rational for having this class is to provide a configurable and controlled way of iterating
235  * a tree or a branch. Some features of the class are:
236  * - Can be created from (or reset to) the \b %StringTree itself,
237  * \ref aworx::lib::util::StringTree::Cursor "Cursor" objects and from
238  * \ref aworx::lib::util::StringTree::StdIterator "StdIterator" objects.
239  * - Can work recursively (with depth limitation) or just on the current nodes' direct children.
240  * - Has several sort options:
241  * - built-in by path name, ascending/descending, case sensitive or ignoring
242  * - user defined by path name
243  * - user defined by template type
244  *
245  * The class is rather 'heavy' and recursive iteration needs to allocate memory for a sorted vector
246  * of child nodes.
247  * Therefore it is recommended to reuse instances of the class with subsequent, similar iterations.
248  * In addition this explains why this class does not follow the concept of <c>std::iterator</c>,
249  * same as class
250  * \ref aworx::lib::util::StringTree::Cursor "Cursor" does not - but then unlike class
251  * \ref aworx::lib::util::StringTree::StdIterator "StdIterator" which does!
252  *
253  * \anchor cpp_alib_containers_stringtree_iterator
254  * <b>Recursive Iteration</b><br>
255  *
256  * Both inner types that provide depth-iteration, namely class
257  * \ref aworx::lib::util::StringTree::StdIterator "StdIterator" and class
258  * \ref aworx::lib::util::StringTree::Walker "Walker", allow recursive iteration. There are
259  * two specialities to note on this topic which need explanation:
260  *
261  * 1. <b>Skipping the Initial %Node:</b><br>
262  * When the recursive iteration is limited, then the very first node (the root node of the
263  * iteration) is skipped. With unlimited recursion, the root node is included (as the first
264  * result). In the case, that the root node should not be included in an unlimited, recursive
265  * iteration, it has to be skipped 'manually'.
266  * 2. <b>Default Recursion Depth:</b><br>
267  * There are different default values for the recursion limit, depending on the way a \b %StdIterator
268  * is initialized.
269  * For example, method \b %StringTree::begin() sets up an recursive \b %StdIterator, while
270  * \b %Cursor::begin limits the recursion to \c 1, hence to iterate only the direct child
271  * nodes. This default behavior was identified to be reasonable for many situations.<br>
272  * If different recursiveness is wanted, a parameter needs to be passed to \b %begin, which of
273  * course is then not applicable to nice and short <em>C++ 11 ranged-for loops</em>.
274  *
275  * <b>Iterators And Changes Of The Underlying Container</b><br>
276  * As with container classes in namespace \c std, objects of iterator-like types of this class (i.e.
277  * \ref aworx::lib::util::StringTree::Cursor "Cursor",
278  * \ref aworx::lib::util::StringTree::StdIterator "StdIterator" and
279  * \ref aworx::lib::util::StringTree::Walker "Walker") may become invalid when
280  * the underlying \b %StringTree object changes and have to be re-initialized after such change.
281  * I.e, there is no guarantee given that changes in the tree are not invalidating iterators that
282  * represent 'higher' nodes or or 'sibling' branches. Such invalid state can not be detected. As
283  * a result, all iterator like objects have to be reset or disposed on changes of the tree.
284  *
285  * <b>Key String Allocations:</b><br>
286  * The C++ version of this class implements user-defined allocation of the characters sequences
287  * stored as keys in the child hash tables of each node. For this, template parameter
288  * \p StringTreeDefaultKeyAllocator is given which defaults to struct
289  * \ref aworx::lib::util::StringTreeDefaultKeyAllocator "StringTreeDefaultKeyAllocator", which
290  * just uses new and delete to allocate and free the space.
291  * This is done, because chances are good that
292  * - either the key values provided in the child creation methods of class \b %Cursor are
293  * permanently valid or
294  * - the templated value element \p T contains a "copy" of the key.
295  *
296  * If the first case applies, type
297  * \ref aworx::lib::util::StringTreeConstKeyAllocator "StringTreeConstKeyAllocator" may be used,
298  * which does not allocate anything and does not copy the key even.<br>
299  * If the second case applies, a custom type may be provided which stores the key in \p T and
300  * returns this copy to be used. Note that for this reason, class \b StringTree asserts that
301  * node elements and with them custom objects of type \p T are deleted always \b after they have
302  * been removed from the tree, so that the key memory can safely deallocated.
303  *
304  * @tparam T The value type of elements stored in the nodes of the tree.
305  * @tparam TKeyAllocator Type with static allocation methods as defined in default struct
306  * \ref aworx::lib::util::StringTreeDefaultKeyAllocator "StringTreeDefaultKeyAllocator".
307  * @tparam TSeparator The separation character used with cursor paths.
308  **************************************************************************************************/
309 template<typename T,
310  typename TKeyAllocator= StringTreeDefaultKeyAllocator<T>,
311  char TSeparator = '/'

386  /**
387  * Returns a child node identified by parameter \p childName.
388  * If the child is not found or is illegal (empty, <c>"."</c> or <c>".."</c>
389  * or contain a separator character), \c nullptr is returned.
390  * In debug compilations, an \ref ALIB_WARNING is reported on illegal child names.
391  *

393  * @return The child node found, \c nullptr on failure.
394  */

396  {
397  if( childName.IsEmpty()
398  || childName.Equals( "." )
399  || childName.Equals( ".." )
400  || childName.IndexOf( TSeparator) >=0 )
401  {
402  ALIB_WARNING( "Illegal child name {!Q}.", childName );
403  return nullptr;
404  }

406  return it != children.end() ? it->second
407  : nullptr;
408  }

410  /**
411  * Non-checking version of #createChild. Must be used only if it is assured that
412  * no child with that name exists, yet.
413  *
414  * @param childName The name of the child to be created and inserted
415  * @return A pointer to the newly created child node.
416  */
417  Node* createChildNoCheck( const String& childName )
418  {
419  Node* child= new Node( this ); // create child first, as we are using its name object as key.
420 
421  ALIB_DBG( auto result= )
422  children.emplace( TKeyAllocator::allocate( childName, child->value ), child );
423  ALIB_ASSERT_ERROR( result.second, "Internal error. Replacing child node." )
424  return child;
425  }
426 
427  /**
428  * Creates and returns a child node. If a node already exists, nothing is done and
429  * \c nullptr is returned as this is considered an error.
430  *
431  * @param childName The name of the child
432  * @return Pointer to the newly created child in case of success. \c nullptr if a child
433  * with the given name already existed.
434  */
435  Node* createChild( const String& childName )
436  {
437  ALIB_ASSERT_ERROR( childName.Length() > 0 , "Can note create unnamed node");
438 
439  if( getChild( childName ) != nullptr )
440  return nullptr;
441 
442  return createChildNoCheck( childName );
443  }
444 
445  /**
446  * Searches and - if no existent - creates a child node.<br>
447  * If a child needs to be created, \p childName is first checked to not equal
448  * to <c>"."</c> or <c>".."</c> and that it does not contain the separation character.
449  * If it does, still \c true is returned and with \c nullptr for the new child.
450  * Furthermore, an \ref ALIB_WARNING is reported.
451  *
452  * @param childName The name of the child
453  * @return A pair of the child node found or created and a boolean which is \c true
454  * when the child was created and \c false when it was already existing.
455  */
456  std::pair<Node*, bool> getOrCreateChild( const String& childName )
457  {
458  Node* child= getChild( childName );
459  bool wasCreated= child==nullptr;
460  if( wasCreated )
461  {
462  if( childName.Equals( "." )
463  || childName.Equals( ".." )
464  || childName.IndexOf( TSeparator) >=0 )
465  {
466  ALIB_WARNING( "Illegal child name {!Q}.", childName );
467  }
468  else
469  child= createChildNoCheck( childName );
470  }
471  return std::make_pair( child, wasCreated );
472  }
473 
474  /**
475  * Deletes named child node.
476  * Key strings are deallocated after the child is removed from the map using template
477  * type TKeyAllocator.
478  * @param childName The name of the child
479  * @return \c true if the child was found and deleted, \c false otherwise.
480  */
481  bool deleteChild( const String& childName )
482  {
483  auto it= children.find( childName );
484  if( it == children.end() )
485  return false;
486 
487  // remove child and then delete
488  String key= (*it).first;
489  Node* child= (*it).second;
490  children.erase( it );
491  TKeyAllocator::free( key, child->value );
492  delete child;
493  return true;
494  }
495 
496  /**
497  * Deletes all child nodes. It is assured that all children are removed
498  * from the map before the node structs and the key strings are delete.
499  * Key strings are deallocated using template type TKeyAllocator
500  */
501  void deleteChildren()
502  {
503  Node* childObjectsLocalStore[1024];
504  String childNamesLocalStore [1024];
505  Node** childObjects;
506  String* childNames;
507 
508  size_t qtyChildren= children.size();
509  if( qtyChildren > 1024 )
510  {
511  childNames= new String[qtyChildren];
512  childObjects= new Node* [qtyChildren];
513  }
514  else
515  {
516  childObjects= childObjectsLocalStore;
517  childNames= childNamesLocalStore;
518  }
519 
520 
521  size_t i= 0;
522  for (auto it : children )
523  {
524  childNames [i]= it.first;
525  childObjects[i]= it.second;
526  i++;
527  }
528 
529  children.clear();
530 
531  for( i= 0; i < qtyChildren ; ++i )
532  {
533  TKeyAllocator::free( childNames[i], childObjects[i]->value );
534  delete childObjects[i];
535  }
536 
537  if( qtyChildren > 1024 )
538  {
539  delete[] childObjects;
540  delete[] childNames;
541  }
542  }

893  /**
894  * Finds a child node along the \p path given, but does not create new nodes.
895  * Incomplete results may occur if a child along the path was not found.
896  * In this case, parameter \p path contains
897  * the remaining path, excluding a leading separator.
898  *
899  * A leading slash (aka \p TSeparator) allows absolute path addressing, which means
900  * the root of \p node is searched if a leading separator is found.
901  *
902  * Besides normal child names, this method accepts
903  * - multiple separator characters (ignored)
904  * - child name "." (ignored)
905  * - child name ".." for parent node
906  *
907  * If, while processing the path string, the root node is found an the next
908  * path element is "..", this element is ignored and processing continues.
909  * As a sample, the paths:
910  *
911  * /a/../b
912  * and
913  *
914  * /a/../../b
915  * both evaluate to
916  *
917  * /b
918  * assumed that \e /a and \e /b exist.
919  *
920  * @param node The start node.
921  * @param[in,out] path Creation path. Provided as reference and consumed as far
922  * as the path exits.
923  * @return The node found
924  */
925  static
926  Node* followPath( Node* node, Substring& path )
927  {
928  // check for "root addressing"
929  if( path.CharAtStart() == TSeparator )
930  {
931  path.ConsumeChars( 1 );
932  while( node->parent != nullptr )
933  node= node->parent;
934  }
935 
936  // loop over node names in path
937  for(;;)
938  {
939  // multiple separators are ignored
940  while(path.ConsumeChar( TSeparator ) )
941  ;
942 
943  if( path.IsEmpty() )
944  return node;
945 
946 
947  String name= String( path, 0, path.IndexOfOrLength( TSeparator ) );
948 
949 
950  if( name.Equals( ".." ) )
951  {
952  // we move up only if possible. If not, the ".." is just ignored (consumed)
953  // (This behavior was just more helpful in the use cases so far)
954  if( node->parent != nullptr )
955  node= node->parent;
956  }
957 
958  else if( !name.Equals( "." ) )
959  {
960  auto it= node->children.find( name );
961  if( it == node->children.end() )
962  return node;
963  node= it->second;
964  }
965 
966  path.ConsumeChars( name.Length() );
967  }
968  }
969 
970  /**
971  * Creates child nodes corresponding to the given \p path.
972  * If the first node already exists, nothing is done and \c nullptr is returned as
973  * this is considered an error.
974  * If the given path is empty the given node is returned as this is \b not
975  * considered an error.
976  *
977  * Child names <c>"."</c> and <c>".."</c> are ignored, but considered an error.
978  * In debug builds, an \ref ALIB_WARNING is reported.
979  *
980  * The final leaf node is returned.
981  *
982  * @param node The start node.
983  * @param[in,out] path Creation path. Provided as reference and consumed if not
984  * errorneous.
985  * @return The leaf node of all created nodes. \c nullptr in the case that the
986  * first child given in \p path already exists.
987  */
988  static
989  Node* createPathAndReturnLeafx( Node* node, Substring& path )
990  {
991  bool isFirst= true;
992  for(;;)
993  {
994  // consume and check
995  while(path.ConsumeChar( TSeparator ) )
996  ;
997  if( path.IsEmpty() )
998  return node;
999 
1000  // get next name
1001  String childName= String( path, 0, path.IndexOfOrLength( TSeparator ) );
1002  if( childName.Equals( "." ) || childName.Equals("..") )
1003  {
1004  //ALIB_WARNING( "Illegal child name {!Q} in creation path", childName );
1005  return node;
1006  }
1007 
1008  Node* child;
1009  if( isFirst )
1010  {
1011  child= node->createChild( childName );
1012  if( child == nullptr )
1013  return nullptr;
1014  }
1015  else
1016  {
1017  child= node->createChildNoCheck( childName );
1018  }
1019  isFirst= false;
1020 
1021  // done?
1022  path.ConsumeChars( childName.Length() + 1);
1023  node= child;
1024  }
1025  }

1550  * iteration node relative to the start node can be activated. See #PathGeneration

1967  * @return The result of #IsValid after the 'walk'. */
1968  bool Next()
1969  {
1970  ALIB_ASSERT_ERROR( IsValid(), "Invalid iterator")
1971  while( stackSize > 0 )
1972  {
1973  RecursionData* nc= &stack[ stackSize -1 ];
1974 
1975  if( nc->actChild == nc->sortedChildren->end() )
1976  nc->actChild= nc->sortedChildren->begin();
1977  else
1978  nc->actChild++;
1979 
1980  // not end of child ?
1981  if( nc->actChild != nc->sortedChildren->end() )
1982  {
1983  // set path string (if activated)
1984  if(static_cast<int>( stackSize ) < recursionDepth )
1985  {
1986  // set path string for #GetPath
1987  if( stackSize > 1 && actPath.IsNotNull() )
1988  {
1989  if( actPath.Length() )
1990  actPath << TSeparator;
1991  actPath << nc->nodeName;
1992  }
1993 
1994  if( stack.size() == stackSize )
1995  {
1996  stack.emplace_back();
1997  nc= &stack[ stackSize -1 ]; // might have changed
1998  }
1999 
2000  stack[ stackSize ].reset( nc->actChild->second, nc->actChild->first, sorter );
2001  stackSize++;
2002  }
2003  break;
2004  }
2005 
2006  // climb down
2007  stackSize--;
