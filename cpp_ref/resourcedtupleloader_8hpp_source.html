113  /**
114  * Specialization of TMP function that recursively compiles. In each compilation recursion
115  * step, one of the overloaded #readElement functions is inserted into the code.
116  *
117  * @param tuple The tuple to read.
118  * @param tknzr The tokenizer to read the elements from.
119  * @tparam I The actual (compile time) tuple element.
120  * @tparam TTuple The tuple to read.
121  * @return Ignored.
122  */
123  template<std::size_t I = 0,
124  typename TTuple >
125  inline
126  typename std::enable_if<I != std::tuple_size<TTuple>::value>::type
127  static readEachElement( TTuple& tuple, Tokenizer& tknzr)
128  {
129  readElement( std::get<I>(tuple), tknzr.Next(Whitespaces::Keep) );
130  readEachElement<I + 1, TTuple>(tuple, tknzr); // compiler recursion
131  }

139  /** ****************************************************************************************
140  * Loads a vector of \c std::tuple objects from the resources.
141  *
142  * The entries are either all read from resource variable named \p name, or, if this is
143  * not found, this methods searches for all resource variables named \p nameN with \p N
144  * starting at \c 0 and increasing until resource string \p nameN is not found.
145  *
146  * The types of elements supported are:
147  * - \c aworx::String:<br>
148  * Read 'as is' from the resource string.
149  * - \c char:<br>
150  * Read 'as is' from the resource string.
151  * - Integer types:<br>
152  * Read using singleton \alib{strings,NumberFormat::Computational} which
153  * supports decimal, binary, octal and hexadecimal formats.
154  *
155  * \note
156  * While all other types are simply ignored and default-constructed, still a separation
157  * character has to exist in the resource string.
158  *
159  * @param tuples The vector of tuples to fill
160  * @param library The library object to read from.
161  * @param name The name of the resource string in \p resources.
162  * @param delim The delimiter of the values.
163  * Defaults to \b ','.
164  * @tparam TTuple The tuple type. Deducted by the compiler from parameter \p tuples.
165  ******************************************************************************************/
166  template<typename TTuple>
167  static void LoadTable( std::vector<TTuple>& tuples,
168  Library & library,
169  const String& name,
170  char delim= ',' )
171  {
172  // get category table and search entry
173  String tupleRes= library.GetTry( name );
174  if( tupleRes.IsNotNull() )
175  {
176  Tokenizer tknzr( tupleRes, delim );
177  while( tknzr.HasNext() )
178  {
179  TTuple tuple;
180  readEachElement( tuple, tknzr );
181  tuples.emplace_back( tuple );
182  }
183  }
184  // entry not found? Try to load numbered entries.
185  else
186  {
187  String64 nameNumbered(name);
188  int nr= 0;
189  for(;;)
190  {
191  tupleRes= library.GetTry( nameNumbered._(nr) );
192  if( tupleRes.IsNull() )
193  break;
194  Tokenizer tknzr( tupleRes, delim );
195  TTuple tuple;
196  readEachElement( tuple, tknzr );
197  tuples.emplace_back( tuple );
198  nameNumbered.SetLength( name.Length() );
199  nr++;
200  }
201 
202  }
203  #if ALIB_MODULE_ALL //Report
204  ALIB_ASSERT_ERROR( tuples.size() > 0,
205  "No tuple resource \"" , name,
206  "\" found in category: ", library.ResourceCategory );
207  #else
208  #if ALIB_DEBUG
209  if( tuples.size() == 0 )
210  {
211  String64 tName ( name );
212  String64 tCategory( library.ResourceCategory );
213  ALIB_ERROR( "No tuple resource \"" , tName.ToCString(),
214  "\" found in category: ", tCategory.ToCString() );
215  }
216  #endif
217  #endif
218  }
