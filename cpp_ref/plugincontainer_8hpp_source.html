45 /** ************************************************************************************************
46  * This class usually is used derived from. Within \alib itself, for example class
47  * \alib{config,Configuration} inherits this class publically.
48  *
49  * Plug-ins are organized with a prioritization. This means, that plug-ins which are inserted with
50  * a higher priority are 'asked' first, and those with a lower value become 'asked' only if higher
51  * prioritized plug-ins did not answer.<br>
52  * However, a derived class can deviate from this behaviour. Using the typical \alib class design,
53  * all internal fields are \c protected, hence freely accessible by derived classes.
54  *
55  * @tparam pTPlugin The type of the plug-ins managed. This type is publicly exposed as #TPlugin.
56  * @tparam pTPriorities The enumeration type providing the default "plug-in slots" and priorities
57  * of the plug-ins managed. This type is publicly exposed as #TPriorities.
58  **************************************************************************************************/
59 template<typename pTPlugin, typename pTPriorities>
60 class PluginContainer : public threads::ThreadLock
61 {

114  /** ****************************************************************************************
115  * Adds the given plug-in to the list of plug-ins..
116  * Higher numbers for parameter priority, prioritize this plug-in against those with
117  * lower values.
118  *
119  * Default values may be provided in case template type #TPriorities is an
120  * \alib{lang,T_EnumIsArithmetical,arithmetical enumeration}.
121  *
122  * @param plugin The plug-in to insert.
123  * @param priority The priority of the plug-in.
124  * @param responsibility If \c Responsibility::Transfer, the given plugin will be deleted
125  * with destruction of this object.
126  * Defaults to \c Responsibility::KeepWithSender which denotes that
127  * the life-cycle of the given external buffer is managed elsewhere.
128  ******************************************************************************************/
129  void InsertPlugin( TPlugin* plugin, TPriorities priority,
130  lang::Responsibility responsibility
131  = lang::Responsibility::KeepWithSender )
132  {
133  LOCK_HERE
134  ALIB_ASSERT_ERROR( plugin != nullptr, "Nullptr provided for plugin." );
135 
136  ALIB_WARNINGS_ALLOW_TEMPLATE_META_PROGRAMMING
137  ALIB_ASSERT_ERROR( std::has_virtual_destructor<TPlugin>::value
138  || responsibility == lang::Responsibility::KeepWithSender,
139  "Can't take responsibility for plug-in destruction. TPlugin has no "
140  "virtual destructor." )
141  ALIB_WARNINGS_RESTORE
142  if( responsibility == Responsibility::Transfer )
143  pluginsOwned.emplace_back( plugin );
144 
145  plugins.insert( std::find_if( plugins.begin(), plugins.end(),
146  [priority]( Slot& ppp)
147  {
148  ALIB_ASSERT_ERROR( ppp.priority != priority,
149  "PluginContainer::InsertPlugin(): Plug-in with same priority exists" );
150 
151  return ppp.priority < priority;
152  } ),
153 
154  Slot {priority, plugin} );
155 
156  }
157 
158 
159 
160  /** ****************************************************************************************
161  * Remove the given plug-in from the list of plug-ins.
162  *
163  * Responsibility for deletion of removed plug-ins is passed to the remover in case the
164  * plug-in was inserted with parameter \p responsibility set to
165  * \alib{lang,Responsibility::Transfer}.
166  *
167  * @param plugin The plug-in to be removed.
168  * @return \c true if the plug-in was removed, else \c false, which indicates that the
169  * given plug-in was not found.
170  ******************************************************************************************/
171  bool RemovePlugin( TPlugin* plugin )
172  {
173  LOCK_HERE
174 
175  size_t qty= plugins.size();
176  plugins.erase( std::remove_if( plugins.begin(), plugins.end(), [plugin](Slot& pair)
177  {
178  return pair.plugin == plugin;
179  } ),
180  plugins.end() );
181 
182  ALIB_ASSERT_WARNING( plugins.size() != qty, "PluginContainer::RemovePlugin(): No Plug-in was removed " )
183 
184  // remove from list of managed plug-ins (if exists)
185  for( auto it= pluginsOwned.begin() ; it != pluginsOwned.end() ; ++it )
186  if( *it == plugin )
187  {
188  pluginsOwned.erase( it );
189  break;
190  }
191 
192  return plugins.size() < qty;
193  }
194

196  /** ****************************************************************************************
197  * Remove the plug-in with the given priority.
198  *
199  * Responsibility for deletion of removed plug-ins is passed to the remover in case the
200  * plug-in was inserted with parameter \p responsibility set to
201  * \alib{lang,Responsibility::Transfer}.
202  *
203  * @param priority The priority of the plug-in to remove.
204  * @return \c true if the plug-in was removed, else \c false, which indicates that no
205  * plug-in with the given priority was found.
206  ******************************************************************************************/
207  TPlugin* RemovePlugin( TPriorities priority )
208  {
209  LOCK_HERE
210 
211  TPlugin* plugin= nullptr;
212  plugins.erase( std::remove_if( plugins.begin(), plugins.end(),
213  [priority, &plugin](Slot& entry)
214  {
215  if( entry.priority == priority)
216  {
217  plugin= entry.plugin;
218  return true;
219  }
220  return false;
221  } ),
222  plugins.end() );
223 
224  ALIB_ASSERT_WARNING( plugin, "PluginContainer::RemovePlugin(): No Plug-in was removed " )
225 
226  return plugin;
227  }

274  /** ****************************************************************************************
275  * Returns the plug-in with the given priority. If the plug-in does not exist, \c nullptr
276  * is returned.
277  *
278  * \note
279  * The method is not thread-safe. At least, the returned plugin may become invalid
280  * after the method returned.
281  * For thread-safe operations, this object needs to be
282  * \alib{threads,ThreadLock::Acquire,acquired} and
283  * \alib{threads,ThreadLock::Release,released} explicitly.
284  *
285  * @param priority The priority of the plug-in to return.
286  *
287  * @return The plug-in requested or \c nullptr if not available.
288  ******************************************************************************************/
289  TPlugin* GetPlugin( TPriorities priority )
290  {
291  LOCK_HERE
292 
293  auto it = std::find_if( plugins.begin(), plugins.end(),
294  [priority](Slot& pair)
295  {
296  return pair.priority == priority;
297  } );
298  if( it != plugins.end() )
299  return it->plugin;
300  return nullptr;
301  }

303  /** ****************************************************************************************
304  * Same as #GetPlugin, but converts the plug-in found to the template type, which has
305  * to be explicitly provided with the invocation of this method.
306  *
307  * A type-check is performed using standard C++ \c dynamic_cast mechanics.
308  * If the plugin has a different type, \c nullptr is returned.
309  *
310  * @param priority The priority of the plug-in to return.
311  * @tparam TPluginType The type of the plugin to search.
312  * @return The plug-in of requested type and priority. \c nullptr if not available.
313  ******************************************************************************************/
314  template<typename TPluginType>
315  inline
316  TPluginType* GetPluginTypeSafe( TPriorities priority )
317  {
318  return dynamic_cast<TPluginType*>( GetPlugin( priority ) );
319  }
320 
321  /** ****************************************************************************************
322  * Searches the list of plug-ins for the first found with type \p TPluginType.
323  *
324  * @tparam TPluginType The type of the plugin to search.
325  * @return The plug-in of requested type. \c nullptr if not available.
326  ******************************************************************************************/
327  template<typename TPluginType>
328  TPluginType* GetPluginTypeSafe()
329  {
330  TPluginType* casted= nullptr;
331  for( auto& ppp : plugins )
332  if( (casted= dynamic_cast<TPluginType*>( ppp.plugin )) != nullptr )
333  break;
334 
335  return casted;
336  }
