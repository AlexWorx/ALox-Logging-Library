128  /** ****************************************************************************************
129  * Returns the time span between the current system time and the internal start value.
130  * In addition this value is added to the sum of sample times and the sample counter is
131  * increased by one. Lastly the internal reference value is set to now. Therefore, a
132  * subsequent call to this function would measure the time span from this call to this
133  * subsequent call (if the internal start time value was not set differently meanwhile).
134  *
135  * @return The time difference between the current system time and the internal
136  * reference value.
137  ******************************************************************************************/
138  inline
139  const TimeStamp Sample ()
140  {
141  cntSamples++;
142  TimeLib::TRaw st= startTime.Raw();
143  startTime.Set();
144  TimeLib::TRaw diff= startTime.Raw() - st;
145 
146  TimeStamp sample( diff );
147  sum+= diff;
148 
149  return sample;
150  }
