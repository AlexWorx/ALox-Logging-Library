38  /**
39  * The logging Buffer. This can be accessed publicly and hence used as preferred.

41  * \ref aworx::lib::strings::AString::Clear "AString::Clear".
42  * In multi-threaded environments, \c Lox interfaces' mutex should be acquired
43  * before accessing this buffer. The initial size of the buffer is 8kb.
44  */
45  AString MemoryLog;

47  /**
