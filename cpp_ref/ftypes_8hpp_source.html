212 /** Implements 'Equals' - interface for boxed \c double values. */
213 class IEquals_Tdouble : public IEquals, public Singleton<IEquals_Tdouble>
214 {
215  /**
216  * Checks if \p lhs with boxed type \c double equals \p rhs. This is done by first comparing
217  * the exact values. If they differ, the difference is calculated and compared to
218  * <c> 2.0f * std::numeric_limits<float>::epsilon()</c>. If smaller, \c true is returned.
219  *
220  * @param lhs The box that the interface was invoked on.
221  * @param rhs The boxed value to compare us to.
222  * @return \c true if \p comp equals \p lhs, \c false otherwise.
223  */
224  ALIB_API
225  virtual bool Invoke( const Box& lhs, const Box& rhs );
226 };
