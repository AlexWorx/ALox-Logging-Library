30 /** ************************************************************************************************
31  * This class is used for defining a region (sub-string) on an existing character array.
32  * Objects of this class will not manipulate the underlying data.
33  * In particular, there is no terminating zero written.
34  *
35  * Unlike in the \e C# and \e Java versions of ALib, the represented region is defined
36  * by the protected fields inherited from class #String.
37  * The difference to base class \ref aworx::lib::strings::String "String" is, that class \b %Substring
38  * allows to change the buffer's start and its length. Otherwise, this class has the
39  * very same lightweight nature and performance as its base.
40  * In general, while working with a \b %Substring, the size of it should shrink, e.g. by trimming,
41  * but should not grow.
42  *
43  * Like base class
44  * \ref aworx::lib::strings::String "String", the class can not, and therefore does not, verify that the
45  * underlying buffer is still allocated or contains termination characters within the region
46  * (which would be is against the rules).
47  * It is up to the user of this class to make sure the buffer stays intact until any referencing
48  * object of this type is destructed (or just is not used any more).
49  *
50  * Objects of this class can be reused by freshly initializing them using of method
51  * #Set(const String&) and #Set(const String&,integer,integer).
52  *
53  *<p>
54  * \note To generate \b %Substring objects which are separated by a delimiter character within a
55  * character array, use class
56  * \ref aworx::lib::strings::util::Tokenizer "Tokenizer".
57  **************************************************************************************************/

87  * Sets this object to represent the given region of an
88  * \ref aworx::lib::strings::String "String" object. The region is
89  * checked (and adjusted) to be between 0 and \p as length.
90  *
91  * \note When using the non-checking version, parameter \p regionLength must be set
92  * explicitly to the correct value (instead of using the default value).
93  *
94  * @tparam TCheck Chooses checking or non-checking implementation. Defaults to true.
95  * @param src The source string to set the region from (allowed to be same).
96  * @param regionStart The start of the region within the given \b %String.
97  * @param regionLength The length of the region within the given \b %String.
98  * Defaults to CString::MaxLen.
99  * @return \c *this to allow concatenated calls.
100  ******************************************************************************************/
101  template <bool TCheck= true>
102  inline
103  Substring& Set( const String& src, integer regionStart, integer regionLength= CString::MaxLen )
104  {
105  if (TCheck)
106  src.AdjustRegion( regionStart, regionLength );
107  else
108  {
109  //---- non-checking version ----
110  ALIB_ASSERT_ERROR( regionStart >= 0 && regionLength >= 0
111  && regionLength != CString::MaxLen // has to be separately checked!
112  && regionStart + regionLength <= src.Length(),
113  "NC: Invalid region given" )
114  }
115 
116  buffer= src.Buffer() + regionStart;
117  length= regionLength;
118  return *this;
119  }

377  * Cuts the given number of characters from the beginning of the Substring and optionally
378  * places the portion that was cut in parameter \p target (if provided).<br>
379  * Parameter \p regionLength is checked to be between 0 and length. If negative, nothing
380  * is cut and \p target is set empty. If \p regionLength is greater than this
381  * objects' length, all contents is 'moved' to \p target.
382  *
383  * @tparam TCheck Defaults to \c true which is the normal invocation mode.
384  * If \c <false> is added to the method name, no parameter check is
385  * performed.
386  *
387  * @param regionLength The length of the region at the start to delete.
388  * @param target An optional target \b %Substring that receives the portion that
389  * is cut from this object. Defaults to nullptr.
390  *
391  * @return The new length of the substring.
392  ******************************************************************************************/

395  integer ConsumeChars( integer regionLength, Substring* target= nullptr )

398  {
399  if ( regionLength < 0 )
400  {

406  regionLength= length;
407  }
408  else
409  {

411  "NC: regionLength out of bounds" )

416 
417  buffer+= regionLength;

845  /** ****************************************************************************************
846  * Splits this substring into two parts. What remains in this object is the region
847  * from 0 to \p position.
848  * \p target receives the rest. If \p separatorWidth is given, this is subtracted from
849  * the front of \p target.
850  *
851  * @tparam TCheck Defaults to \c true which is the normal invocation mode.
852  * If \c <false> is added to the method name, no parameter check is
853  * performed.
854  *
855  * @param position The index where this object is split.
856  * @param target The target substring to receive the right part of the string.
857  * @param separatorWidth This does not change what remains in this object, but defines
858  * the number of characters that are cut from the front of the
859  * \p target. Defaults to 0.
860  * @param trim If \c true, both substrings will be trimmed.
861  *
862  * @return \c *this to allow concatenated calls.
863  ******************************************************************************************/
864  template <bool TCheck= true>
865  inline
866  Substring& Split( integer position, Substring& target, integer separatorWidth =0, bool trim= false )
867 
868  {
869  if ( TCheck )
870  {
871  AdjustRegion( position, separatorWidth );
872  }
873  else
874  {
875  ALIB_ASSERT_ERROR( position >=0 && position <= length,
876  "NC: position out of bounds" )
877  ALIB_ASSERT_ERROR( position + separatorWidth <= length,
878  "NC: position + separator width out of bounds" )
879  }
880 
881  target.Set( *this, position + separatorWidth, length - position - separatorWidth );
882  length= position;
883  if( trim )
884  {
885  target.Trim();
886  this ->Trim();
887  }
888  return *this;
889  }
