330 /**
331  * This template struct is used to define customized behaviour for boxing values of type
332  * \p TBoxable.
333  *
334  * If this class is \b not specialized for template type \p TBoxable, then default
335  * boxing applies. By default values and references of a type are boxed to pointers of the type,
336  * while pointer types remain such pointer types.
337  *
338  * To specialize the class for a custom type, the following changes have to be made
339  * - The specialization has to be derived from \c std::true_type.
340  * - Method #IsArray has to be implemented to return \c true if the type should be boxed
341  * as an array type
342  * - %Type definition #Type needs to use the boxed type. In case of array boxing, the
343  * array element type.
344  * - Method #IsUnboxable has to return \c false if the type should not be unboxable.
345  * - Methods #HasCustomBoxing and HasCustomUnboxing have to be implemented to return either
346  * \c true or false.
347  * - Methods #Boxing and #Unboxing have to be declared. The also need to be implemented
348  * if #HasCustomBoxing, respectively #HasCustomUnboxing return \c true.
349  *
350  * \see Macro \ref ALIB_BOXING_SPECIALIZE which provides a simple method
351  * to defined specializations of this class.
352  *
353  * @tparam TBoxable The type to customize boxing for.
354  */
355 template<typename TBoxable> struct T_Boxing : std::false_type
356 {
357  /**
358  * If \c true is returned, \p TBoxable is boxed as an array type. In this case alias type
359  * #Type needs to be set to denote the element type of the array.
360  * @return \c true if boxed type is an array type, \c false otherwise.
361  */
362  static constexpr bool IsArray() { return false; }
363 
364  /**
365  * This flag is asserted with keyword \c static_assert when invoking methods
366  * \ref aworx::lib::boxing::Box::IsType "Box::IsType" and
367  * \ref aworx::lib::boxing::Box::Unbox "Box::Unbox". This way, types that are not meant to
368  * be unboxed produce an error.
369  *
370  * @return \c true if boxed type can be unboxed, \c false otherwise.
371  */
372  static constexpr bool IsUnboxable() { return true; }
373 
374 
375  /**
376  * If returns \c true, the specialized method #Boxing will be invoked for boxing a value of
377  * type \p TBoxable. Otherwise, default function
378  * \ref aworx::lib::boxing::DefaultBoxing will be used and method #Boxing does not need
379  * to be implemented.
380  *
381  * @return \c true if custom boxing implementation is to be used.
382  */
383  static constexpr bool HasCustomBoxing() { return false; }
384 
385  /**
386  * If returns \c true, the specialized method #Unboxing will be invoked for boxing a value of
387  * type \p TBoxable. Otherwise, default function
388  * \ref aworx::lib::boxing::DefaultUnboxing will be used and method #Unboxing does not need
389  * to be implemented.
390  *
391  * @return \c true if custom implementation for unboxing is to be used.
392  */
393  static constexpr bool HasCustomUnboxing(){ return false; }
394 
395  /**
396  * Optional custom implementation of boxing a value.
397  * Fields
398  * \ref aworx::lib::boxing::BoxData::Value "BoxData::Value" and
399  * \ref aworx::lib::boxing::BoxData::Length "BoxData::Length" have to be set, because for
400  * efficiency reasons, both are not initialized otherwise.
401  *
402  * If this method is not specialized, template function
403  * \ref aworx::lib::boxing::DefaultBoxing "DefaultBoxing" us used.
404  *
405  * @param box The destination box
406  * @param value The value to box
407  */
408  static void Boxing(Box& box, TBoxable const & value)
409  {
410  (void) box;
411  (void) value;
412  assert(0);
413  }
414 
415  /**
416  * Optional custom implementation of unboxing a value.
417  *
418  * \note
419  * Specializations of this class usually will return type \c TBoxable instead of
420  * \c TBoxable&. The reason why this method here is declared to return a reference, is of
421  * pure technical (TMP) nature.
422  *
423  * If method #HasCustomUnboxing is specialized to return \c false, only the declaration
424  * of this method is needed, because namespace function
425  * \ref aworx::lib::boxing::DefaultUnboxing "DefaultUnboxing" is used in this case.
426  * However, while the declaration is still necessary, the implementation is not.
427  *
428  * @param box The box object to unbox the data from
429  * @return \c The unboxed object.
430  */
431  static TBoxable& Unboxing(const Box& box)
432  {
433  (void) box;
434  assert(0);
435  return *((TBoxable*) 1);
436  }
437 
438  /**
439  * Needs to be using the mapped of template parameter \p TBoxable.
440  */
441  using Type= TBoxable;
442 };

444 /**
445  * This function is invoked by the constructor of class
446  * \ref aworx::lib::boxing::Box "Box" to determine the length of an array.
447  *
448  * Prior to the invocation, the length of the array provided, is deducted from the type using TMP
449  * and passed as a parameter to this method.
450  *
451  * Normally, this method should not be specialized. The whole rational to do this exercise is to
452  * prepare the right handling of literal string arrays. Those are including a terminating
453  * \c 0 character (added by the compiler) and hence their boxing implementation might be deducted
454  * by \c 1. As a sample, the standard implementation of boxing fundamental types, provided with
455  * namespace #aworx::lib::boxing::ftypes (optionally replaced by a custom implementation of
456  * fundamental types boxing) uses this feature. This way, other string types (e.g. of various widely used
457  * libraries) can be boxed as a character array using their right string length.
458  *
459  * @tparam TPlain The value type of the array (e.g. \c int, \c char or \c MyType) excluding
460  * const/volatile and array extends.
461  * @param arrayExtent The array extent as deducted using TMP in constructor of class
462  * \ref aworx::lib::boxing::Box "Box".
463  * @returns The length as given by \p arrayExtent. Specializations of this method may return
464  * something different.
465  */
466 template<typename TPlain>

497 /**
498  * This template function provides the default implementation used to store values in struct
499  * \ref aworx::lib::boxing::BoxData "BoxData" when they are placed in a
500  * \ref aworx::lib::boxing::Box "Box".
501  *
502  * To customize boxing for a type, specialize struct
503  * \ref aworx::lib::boxing::T_Boxing (with the help of macro \ref ALIB_BOXING_SPECIALIZE) with
504  * implementing methods
505  * \ref aworx::lib::boxing::T_Boxing::HasCustomBoxing "T_Boxing::HasCustomBoxing" to return
506  * \c true and having
507  * \ref aworx::lib::boxing::T_Boxing::Boxing "T_Boxing::Boxing" to do the boxing.
508  *
509  * Fields
510  * \ref aworx::lib::boxing::BoxData::Value "BoxData::Value" and
511  * \ref aworx::lib::boxing::BoxData::Length "BoxData::Length" have to be set, because for
512  * efficiency reasons, both are not initialized otherwise.
513  *
514  * However, specializing this method for custom boxable types is seldom necessary, because
515  * complex custom types (structs and classes) usually get announced to the \b %ALib boxing system as
516  * pointer types.
517  * The default implementation of this function, as well as corresponding function
518  * \ref aworx::lib::boxing::DefaultUnboxing "DefaultUnboxing", are using \c memcpy for reading
519  * and writing the value. This is necessary to avoid de-referencing type-punned pointers which
520  * breaks the strict-aliasing rule when compiling the code with higher optimization levels.
521  *
522  * Typically, a custom version of boxing is needed for fundamental types (value types), when a type
523  * is stored as a different type in a box. For example, custom string types might be stored as
524  * pointer to the array of characters and the string length.
525  * The advantage is, that this way all custom string classes are stored in the same way, and code
526  * that is processing a \b %Box does not need to check for more than one type.
527  *
528  * See \ref aworx::lib::boxing "documentation of \b %ALib %Boxing" for more information.
529  *
530  * @param box The \b %Box to set the data vor
531  * @param value The 'native' data value
532  */
533 template<typename TBoxable> inline void DefaultBoxing(Box& box, const TBoxable& value)
534 
535 {
536  if( sizeof(TBoxable) < sizeof(boxvalue) ) // clear value if the data is smaller
537  box.data.Value= 0;
538  ALIB_ASSERT_ERROR(sizeof(value) <= sizeof(box.data.Value),

707  * The object that is passed as parameter is the result of static invocation
708  * of a method named \b "GetSingleton()" on the third template type \p TInterface.

713  * of the interface for type \p TBoxed.

720  : (Boxer*) BoxerT<TBoxed>::GetSingleton() )
721 
722  ->DefineInterface( TInterface::GetSingleton() );
723 }
