124  /** ********************************************************************************************
125  * Copies or appends a region of a given \ref aworx::lib::strings::String "String" into
126  * a given object of type \c std::vector<char>.
127  * The region is adjusted to the length of the %String.
128  *
129  * @param src The \ref aworx::lib::strings::String "String" to convert into a std::vector<char>.
130  * @param target A result \c std::vector<char> to copy the specified region into.
131  * @param regionStart The start index of the region to be copied.
132  * @param regionLength The maximum length of the region to be copied.
133  * Defaults to CString::MaxLen.
134  * @param targetData If \c CurrentData::Keep, the target \c std::vector<char> is not cleared before
135  * the result is written. Defaults to \c CurrentData::Clear.
136  *
137  * @return The (modified) \c std::vector<char>& result that was provided (for concatenation of calls).
138  **********************************************************************************************/
139  inline
140  std::vector<char>& ToStdVector( const String& src,
141  std::vector<char>& target,
142  integer regionStart,
143  integer regionLength = CString::MaxLen,
144  lang::CurrentData targetData = lang::CurrentData::Clear )
145  {
146 
147  if ( targetData == lang::CurrentData::Clear )
148  target.clear();
149 
150  // adjust range, if empty return empty vector<char>
151  if ( src.AdjustRegion( regionStart, regionLength ) )
152  return target;
153 
154  // copy our buffer into target
155  target.reserve( static_cast<size_t>( regionLength ) );
156  const char* ps= src.Buffer() + regionStart;
157  const char* pe= ps + regionLength;
158  while(ps < pe )
159  target.push_back( *ps++ );
160  return target;
161  }
