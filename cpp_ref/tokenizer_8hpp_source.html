26 /** ************************************************************************************************
27  * This class operates on strings which contains data separated by a delimiter character.
28  * It identifies the substrings between the delimiters as \e tokens of type
29  * \ref aworx::lib::strings::Substring "Substring". After an instance of this class is constructed,
30  * three methods are available:
31  * - #HasNext: Indicates if there are further tokens available.
32  * - #Next: Sets field #Actual (which is of type Substring) to reference the next token and
33  * returns it.<br>
34  * With each call to %Next, a different delimiter can be provided, which then serves as the
35  * delimiter for this and subsequent tokens.<br>
36  * The returned token by default will be trimmed according to the current trimable characters.
37  * - #Rest:
38  * Like #Next, however returns the complete remaining region without searching for
39  * further delimiters (and tokens).<br>
40  * After this method was invoked, #HasNext() will return \c false.
41  *
42  * After a token was retrieved, it might be modified using the interface of class
43  * \ref aworx::lib::strings::Substring "Substring" as the tokenizer does not rely on the bounds of
44  * the current token when receiving the next. Furthermore, even field #Rest is allowed
45  * to be changed using the interface of \b %Substring if it seems appropriate. The effect is the
46  * same as if method #Set was invoked to apply a different source string.
47  *
48  * Objects of this class can be reused by freshly initializing them using method #Set.
49  *
50  * <b>Sample code</b>:<br>
51  * The following code sample shows how to tokenize a string:
52  *
53  * \snippet "DOX_ALIB_TOKENIZER.cpp" DOX_ALIB_TOKENIZER
54  *
55  * The output will be:
56  *
57  * \snippet "DOX_ALIB_TOKENIZER.txt" OUTPUT
58  *
59  **************************************************************************************************/
60 class Tokenizer

71  */

145  /** ****************************************************************************************
146  * Returns the next token, which is afterwards also available through field #Actual.
147  * If no further token was available, the returned
148  * \ref aworx::lib::strings::Substring "Substring" will be \e nulled.
149  * (see \ref aworx::lib::strings::String::IsNull "String::IsNull").

151  * checked using method #HasNext().

157  * Defaults to \b Whitespaces.Trim.

159  * current delimiter intact.

161  * @return \c true if a next token was available, \c false if not.
162  ******************************************************************************************/

166  /** ****************************************************************************************
167  * Returns the currently remaining string (without searching for further delimiter
168  * characters).
169  * After this call #HasNext will return \c false and #Next will return a \e nulled Substring.
170  * @param trimming Determines if the token is trimmed in respect to the white space
171  * characters defined in field #Whitespaces.
172  * Defaults to \b Whitespaces.Trim.
173  * @return The rest of the original source string, which was not returned by #Next(), yet.
174  ******************************************************************************************/
175  Substring& GetRest( lang::Whitespaces trimming= lang::Whitespaces::Trim )
176  {
177  // set start, end and end of tokenizer
178  Actual= Rest;
179  Rest.SetNull();
180  if ( trimming == lang::Whitespaces::Trim )
181  Actual.Trim( Whitespaces );
182  return Actual;
183  }
184 
185  /** ****************************************************************************************
186  * If this returns \c true, a call to #Next will be successful and will return a
187  * Substring which is not \e nulled.
188  * @return \c true if a next token is available.
189  ******************************************************************************************/
190  bool HasNext() { return Rest.IsNotNull() && ( !skipEmpty || Rest.IsNotEmpty() ); }
