168  /** ****************************************************************************************
169  * Stores a new value for the given scope.
170  * @param value The value to set.
171  * @return Returns the previous value stored.
172  ******************************************************************************************/
173  StoreT Store ( StoreT value )
174  {
175  ALIB_ASSERT( value != ScopeStoreType<StoreT>::NullValue() );
176  return access( 0, value );
177  }
178 
179 
180  /** ****************************************************************************************
181  * Removes a value for the given scope.
182  * @param value The value to remove (must only be given for thread-related \e Scopes).
183  * @return Returns the previous value stored.
184  ******************************************************************************************/
185  StoreT Remove ( StoreT value )
186  {
187  return access( 1, value );
188  }
189 
190  /** ****************************************************************************************

192  * If Thread::NullThreadId is given, the ID from \p scopeInfo is used.
193  * @return Returns the current value stored.
194  ******************************************************************************************/
195  StoreT Get ()
196  {
197  return access( 2, ScopeStoreType<StoreT>::NullValue() );
198  }

205  * @param localObject The 'local object' returned by a walk after Scope::ThreadInner

207  ******************************************************************************************/
208  ALOX_API
209  void InitWalk( Scope startScope, const StoreT localObject );
