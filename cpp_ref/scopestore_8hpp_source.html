172  * @return Returns the previous value stored.
173  ******************************************************************************************/
174  StoreT Store ( StoreT value )
175  {
176  ALIB_ASSERT( value != ScopeStoreType<StoreT>::NullValue() );
177  return access( 0, value );

181  /** ****************************************************************************************

184  * @return Returns the previous value stored.
