162 /** ********************************************************************************************
163  * This is a specializable struct which defaults the question whether a given type
164  * can be used to construct an \ref aworx::lib::strings::String "String" to \c false.
165  *
166  * Specializations of this struct inherit from \c std::true_type, to define a type to be a usable
167  * constructor parameter of \b %String.
168  *
169  * Hence, this template struct supports converting 'external' <em>user defined</em> to
170  * \ref aworx::lib::strings::String "String". It has a central role for using ALib in a
171  * \ref aworx::lib "non intrusive" way, by allowing to pass external string types just as
172  * the are when invoking ALib functions and methods that expect parameters of type
173  * <c>const %String&</c>. Due to the implicit constructor invocation of C++
174  * and the template meta programming in constructor of class \b %String, custom types
175  * (that specialize this struct) may be passed "as is" to such methods.
176  *
177  * Methods #Buffer and #Length of partially specialized versions of this struct are invoked
178  * by constructor \ref aworx::lib::strings::String::String(const TStringLike&) "String(const TStringLike&)"
179  * to receive a pointer to the character buffer and the length of the external string type.
180  *
181  *
182  * The following sample demonstrates the use of this technique:
183  *
184  * \snippet "DOX_ALIB_TOAS.cpp" DOX_ALIB_TOAS
185  * The output will be:
186  * \snippet "DOX_ALIB_TOAS.txt" OUTPUT
187  *
188  * \note
189  * As shown in the sample, if a specialized version of this struct for a custom type exists, such
190  * type becomes automatically also an "applicable" type to class #AString, because template meta
191  * programming of method \b %AString::Apply checks for the existence of the specialization as well
192  * as the constructor of class #String.
193  *
194  * A compile time assertion will be raised when a String is constructed with a type that has
195  * no specialization of this struct.
196  * **********************************************************************************************/
197 template<typename TStringLike>
198 struct T_String : public std::false_type
199 {
200  /** ********************************************************************************************
201  * In specialized versions, this method has to return a pointer to the buffer of the given
202  * object of type \p TStringLike.
203  *
204  * This default implementation makes some static assertions to detect usage of an unsupported
205  * type (a type that is not allowed to construct class \b %String with).
206  *
207  * @returns Has to return a pointer to the buffer of object of type T, respectively its length.
208  **********************************************************************************************/
209  static inline constexpr
210  const char* Buffer( const TStringLike& )
211  {
212  // should never happen, because of static assert in string constructor
213  return nullptr;
214  }
215 
216  /** ********************************************************************************************
217  * In specialized versions, this method has to return the length of the buffer of the given
218  * object of type \p TStringLike.
219  *
220  * @returns Has to return the length of the string.
221  **********************************************************************************************/
222  static inline
223  integer Length( const TStringLike& )
224  {
225  return (integer) 0;
226  }
227 };

350 /** ************************************************************************************************
351  * This class is the base class of all ALib string classes. It represents a character string
352  * whose data is allocated outside the scope of objects of this class.
353  *
354  * Once constructed, objects
355  * of this class are immutable. This means, there is no interface to change their buffer or length
356  * during their lifetime, which normally is rather volatile.
357  *
358  * The buffer represented might be zero-terminated or not. This is dependent on the data it was
359  * constructed from. The buffer must not be changed and might even reside in read-only memory.
360  *
361  * <b>Templated Construction</b><br>
362  * What makes this class very flexible, is the constructor variant
363  * \ref aworx::lib::strings::String::String(const TStringLike&) "String(const TStringLike&)".
364  * This template method internally uses so called
365  * <em>template meta programming</em> to detect known types and, and
366  * to convert them to constant references of those. They are then passed to template methods
367  * \ref aworx::lib::strings::T_String "T_String" that simply return a pointer to the external types' buffer
368  * and length. This way, objects of this class can be implicitly constructed from just anything that
369  * 'smells' like a string.<br>
370  * For more information on how to make %String support to implicitly construct from user defined types,
371  * see namespace template function
372  * \ref aworx::lib::strings::T_String "T_String".
373  *
374  * This class provides compiler defined copy and move constructors and assignment operators.
375  * Once and object is constructed, methods #Buffer and #Length allow read access to the contents.
376  *
377  * <b>Null-State</b><br> \anchor CPP_STRINGS_AS_NULLSTATE
378  * Objects of this class can be \e nulled which means that it is a difference whether they are
379  * representing an empty string or a \c nullptr. As objects are immutable, this is decided
380  * on construction: once a %String is nulled, it will remain nulled.
381  * In other words, it makes a difference if an %String is constructed using
382  * - <em>%String()</em>, which creates a \e nulled object, method #IsNull will give \c true, or
383  * - <em>%String("")</em>, which creates an empty object. Method #IsNull will give \c false while
384  * method #IsEmpty will give \c true
385  *
386  * <b>Non-checking Method Variants</b><br> \anchor CPP_STRINGS_AS_NC
387  * Some of the provided methods are templated with boolean parameter <em>TCheck</em> which allow
388  * to invoke a faster and shorter version of the same method. For more information see
389  * \ref alib_namespace_strings_nonchecking in the namespace documentation.
390  *
391  * \note Almost all methods of this class are declared <em>inline</em>.
392  *
393  * <p>
394  * \note The immutable nature of %String is lifted by derived types. While class
395  * \ref aworx::lib::strings::Substring "Substring" allows to change the start and
396  * length of the string represented, class
397  * \ref aworx::lib::strings::AString "AString" holds a copy of the data and allows to
398  * modify the contents. Field #buffer of this class is declared
399  * as an anonymous union of a <em>const char*</em> and a <em>char*</em> but the latter is
400  * not exposed. It might be exposed by derived classes (and is by class \p %AString).
401  *
402  * <p>
403  * \note For an introduction into the ALib string classes see
404  * \ref aworx::lib::strings "namespace strings"
405  **************************************************************************************************/
