31  * Finds a tuple in a vector of those by performing a comparison of a given string with
32  * string data provided in one of the tuple elements.
33  *
34  * The first template parameter T_StringColumn denotes
35  * the index of the tuple elements that provides the string to search parameter \p needle in.
36  *
37  * The other template types are fetched with the second, variadic template parameter
38  * \p T_TupleTypes and and denote the types of the tuples. These template parameters do not
39  * need to be specified with invocations of the function, as they are inferred by the compiler
40  * from parameter \p haystack.
41  *
42  *
43  * @param haystack Vector of tuples to search in.
44  * @param needle The string to search.
45  * @tparam sensitivity Character case sensitivity of the comparison.
46  * Defaults to \b Case::Sensitive.
47  * @tparam T_StringColumn The tuple element number of the comparison string.
48  * @tparam T_TupleTypes The template type of the tuple. (Inferred by the compiler.)
49  *
50  * @return A pointer to the tuple found. \b nullptr on failure.
51  */
52 template< Case sensitivity,
53  int T_StringColumn,
54  typename... T_TupleTypes >
55 std::tuple<T_TupleTypes...> const * FindStringInTupleVector( const std::vector<std::tuple<T_TupleTypes...>>& haystack,
56  const String& needle )

67 /**
68  * Finds a tuple in a vector of those by performing a comparison of a given string with
69  * string data provided in one of the tuple elements.
70  *
71  * The function has two integer template parameters. The first, \p T_StringColumn denotes
72  * the index of the tuple elements that provides the string to search method parameter \p needle in.
73  *
74  * The second, \p T_MinCharIndex,give the index of an element of the tuple that
75  * holds an \c int value. This value is used to determine possible
76  * "abbreviations" in the search. For example, if this template parameter is set to \c 1, then
77  * the second column of the table is interpreted as this minimum value. If then a value of
78  * for example \c 3 is found in an entry, then a match is returned if just the first three
79  * characters of the string are provided with method parameter \p needle.
80  * This allows to search for identifiers in a table which are abbreviated up to a mimimum length.
81  *
82  * The other template types are fetched with the second, variadic template parameter
83  * \p T_TupleTypes and and denote the types of the tuples. These template parameters do not
84  * need to be specified with invocations of the function, as they are inferred by the compiler
85  * from parameter \p haystack.
86  *
87  * @param haystack Vector of tuples to search in.
88  * @param needle The string to search.
89  *
90  * @tparam sensitivity Character case sensitivity of the comparison.
91  * Defaults to \b Case::Sensitive.
92  * @tparam T_StringColumn The tuple element number of the comparison string.
93  * @tparam T_MinCharColumn The element number of the column providing a minimum length
94  * of the given search string \p needle.
95  * @tparam T_TupleTypes The template type of the tuple. (Inferred by the compiler.)
96  *
97  * @return A pointer to the tuple found. \b nullptr on failure.
98  */
99 template< Case sensitivity,
100  int T_StringColumn,
101  size_t T_MinCharColumn,
102  typename... T_TupleTypes >
103 std::tuple<T_TupleTypes...> const * FindStringStartInTupleVector( const std::vector<std::tuple<T_TupleTypes...>>& haystack,
104  const String& needle
105  )
106 {
107  for( auto& entry : haystack )
108  {
109  String identifier= std::get<T_StringColumn>(entry);
110  int minChars= std::get<T_MinCharColumn>(entry);
111 
112  if ( Substring(needle).ConsumePartOf<sensitivity>( identifier, minChars ) == needle.Length() )
113  return &entry;
114  }
115  return nullptr;
116 }

118 /** ************************************************************************************************
119  * Variadic Template class implementing a vector of tuples whose first element is of type
120  * \c ref aworx::lib::strings::String "String" and the further elements are
121  * the variadic types \p TAssociatedTypes.
122  *
123  * When new tuples are added with method #Add, then memory for copying the provided string is
124  * allocated using an internal field of type
125  * \c ref aworx::lib::lang::MemoryBlock "MemoryBlock".
126  * This allows to add single strings, which are allocated in bigger memory chunks.
127  *
128  * Note that standard vector operations, including insertions and deletions are still allowed!
129  * In other words, it is allowed to store string objects which are pointing to string data that
130  * is \b not allocated in the internal chunks.
131  *
132  * The associated values are especially useful when the vector is sorted (e.g. using
133  * \c std::sort), because such associations will be kept intact and allow referencing back
134  * to whatever the strings represents.
135  *
136  * \note
137  * This class is new with \b %ALib. It is not considered finished, optimized and stable in design.
138  **************************************************************************************************/
139 template< typename... TAssociatedTypes >
140 class StringTable : public std::vector<std::tuple<String, TAssociatedTypes...>>
141 {
142  // #############################################################################################
143  // Type definitions
144  // #############################################################################################
145  public:
146  /**
147  * This is a shortcut giving the type of the concrete
148  * standard <c>std::tuple<String, ...></c> that instantiations of this template class
149  * store.
150  */
151  using ElementType= std::tuple<String, TAssociatedTypes...>;
152 
153  /**
154  * This is a shortcut giving the type of the vector that this class implements.
155  * Used internally to access inherited methods.
156  */
157  using VectorType = std::vector<ElementType>;
158 
159 
160  // #############################################################################################
161  // protected fields
162  // #############################################################################################
163  protected:
164  /** The list of allocated memory blocks. */
165  MemoryBlocks blocks;
166 
167  // #############################################################################################
168  // Constructor / Destructor
169  // #############################################################################################
170  public:
171  /** ****************************************************************************************
172  * Constructor.
173  * Accepts a value \c stdBlockSize to manipulate the standard size of allocated memory
174  * chunks. (This value, is forwarded to the constructor of class
175  * \alib{util,MemoryBlocks}.)
176  *
177  * @param stdBlockSize The standard size of memory blocks allocated.
178  ******************************************************************************************/
179  StringTable( int stdBlockSize = 8 * 1024 )
180  : blocks( stdBlockSize )
181  {
182  }
183 
184  // #############################################################################################
185  // Interface
186  // #############################################################################################
187  public:
188  /** ****************************************************************************************
189  * Adds a tuple. The contents of the string of the tuple (its first element) is copied
190  * from the given string \p src. The memory for this string is allocated using
191  * field #blocks.<br>
192  * The other members of the tuple added are forwarded from variadic parameter block \p args.
193  *
194  * \note
195  * With some compilers (as the type of writing this with \b clang), a compilation error
196  * might occur. To avoid this, the tuple type used needs to be fixed with using macro
197  * \ref ALIB_STRING_CONSTRUCTOR_FIX "ALIB_STRING_CONSTRUCTOR_FIX( std::tuple<String ALIB_COMMA ...<yourtypes>... )"
198 
199  * @param src The string to copy into the first member of the tuple.
200  * @param args Variadic arguments to fill the rest of the inserted tuple.
201  *
202  * @return A reference to the string that was added.
203  ******************************************************************************************/
204  String& Add( const String& src, TAssociatedTypes... args )
205  {
206  char* mem= blocks.GetMemory(src.Length());
207  // on errors, see note above!
208  VectorType::emplace_back( String( mem, src.CopyTo( mem ) ), args... );
209  return std::get<0>(VectorType::back());
210  }
211 
212 
213  /** ****************************************************************************************
214  * Clears this vector of tuples and frees the allocated block memory, either for reuse
215  * or completely. (Parameter \c deallocate is forwarded to
216  * \ref aworx::lib::lang::MemoryBlocks::Clear "MemoryBlocks::Clear".)
217  *
218  * @param deallocate If \b %CurrentData::Clear , all allocated memory chunks are freed.
219  * If \b %CurrentData::Keep, which is the default, they
220  * will be reused for future new strings that are added.
221  ******************************************************************************************/
222  void Clear( CurrentData deallocate= CurrentData::Keep )
223  {
224  VectorType::clear();
225  blocks.Clear( deallocate );
226  }
227 };
