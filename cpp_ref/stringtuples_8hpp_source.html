28 /** ************************************************************************************************
29  * Variadic Template class implementing a vector of tuples whose first element is of type
30  * \c ref aworx::lib::strings::String "String" and the further elements are
31  * the variadic types \p TAssociatedTypes.
32  *
33  * When new tuples are added with method #Add, then memory for copying the provided string is
34  * allocated using an internal field of type
35  * \c ref aworx::lib::lang::MemoryBlock "MemoryBlock".
36  * This allows to add single strings, which are allocated in bigger memory chunks.
37  *
38  * Note that standard vector operations, including insertions and deletions are still allowed!
39  * In other words, it is allowed to store string objects which are pointing to string data that
40  * is \b not allocated in the internal chunks.
41  *
42  * As with other types of \b %ALib, a (templated) alias name #aworx::StringTuples exists.
43  * In addition, non-templated alias name #aworx::Strings is defined which has an empty
44  * variadic type list. Therefore, the latter can be used to create simple string vectors without
45  * associated values.
46  *
47  * The associated values are especially useful when the vector is sorted (e.g. using
48  * \c std::sort), because such associations will be kept intact and allow referencing back
49  * to whatever the strings represent.
50  *
51  * \note
52  * This class is new with ALib. It is not considered finished, optimized and stable in design.
53  **************************************************************************************************/
54 template< typename... TAssociatedTypes >
55 class StringTuples : public std::vector<std::tuple<String, TAssociatedTypes...>>
56 {
57  // #############################################################################################
58  // Type definitions
59  // #############################################################################################
60  public:
61  /**
62  * This is a shortcut giving the type of the concrete
63  * standard <c>std::tuple<String, ...></c> that instantiations of this template class
64  * store.
65  */
66  using ElementType= std::tuple<String, TAssociatedTypes...>;
67 
68  /**
69  * This is a shortcut giving the type of the vector that this class implements.
70  * Used internally to access inherited methods.
71  */
72  using VectorType = std::vector<ElementType>;
73 
74

76  // protected fields

78  protected:

80  MemoryBlocks blocks;
81 
82  // #############################################################################################
83  // Constructor / Destructor
84  // #############################################################################################
85  public:
86  /** ****************************************************************************************
87  * Constructor.
88  * Accepts a value \c stdBlockSize to manipulate the standard size of allocated memory
89  * chunks. (This value, is forwarded to the constructor of class
90  * \ref aworx::lib::lang::MemoryBlocks "MemoryBlocks".)
91  *
92  * @param stdBlockSize The standard size of memory blocks allocated.
93  ******************************************************************************************/
94  StringTuples( int stdBlockSize = 8 * 1024 )
95  : blocks( stdBlockSize )
96  {
97  }
98 
99  // #############################################################################################
100  // Interface
101  // #############################################################################################
102  public:
103  /** ****************************************************************************************
104  * Adds a tuple. The contents of the string of the tuple (its first element) is copied
105  * from the given string \p src. The memory for this string is allocated using
106  * field #blocks.<br>
107  * The other members of the tuple added are forwarded from variadic parameter block \p args.
108  *
109  * @param src The string to copy into the first member of the tuple.
110  * @param args Variadic arguments to fill the rest of the inserted tuple.
111  *
112  * @return A reference to the string that was added.
113  ******************************************************************************************/
114  String& Add( const String& src, TAssociatedTypes... args )
115  {
116  char* mem= blocks.GetMemory(src.Length());
117  VectorType::emplace_back( String( mem, src.CopyTo( mem ) ), args... );
118  return std::get<0>(VectorType::back());
119  }
120 
121 
122  /** ****************************************************************************************
123  * Clears this vector of tuples and frees the allocated block memory, either for reuse
124  * or completely. (Parameter \c deallocate is forwarded to
125  * \ref aworx::lib::lang::MemoryBlocks::Clear "MemoryBlocks::Clear".)
126  *
127  * @param deallocate If \c true, all allocated memory chunks are freed. Otherwise, they
128  * will be reused for future new strings that are added.
129  ******************************************************************************************/
130  void Clear( bool deallocate= false )
131  {
132  VectorType::clear();
133  blocks.Clear( deallocate );
134  }
135 };
