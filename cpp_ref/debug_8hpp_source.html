60  /**
61  * Writes information about the template type \p TBoxable in respect to how boxing of
62  * objects (values) of this type are performed.
63  *
64  * \note
65  * Does not work for type \c void*, which is boxed and unboxed as identical type \c void*.
66  *
67  * @return A string containing the requested information.
68  */
69  template<typename TBoxable>
70  AString& GetBoxingInfo()
71  {
72  using TGiven= typename std::remove_const <
73  typename std::remove_reference<TBoxable>::type>::type;
74 
75  target._() << "Boxing information for type: " << typeid(TGiven) << NewLine;
76 
77  if( std::is_same<TGiven, decltype(nullptr)>::value )
78  return target << " 'decltype(nullptr)' not customizable" << NewLine;
79 
80  if( std::is_array<TGiven>::value )
81  {
82  using TAElem= typename std::remove_extent<TGiven>::type;
83  return target << " Array Type (not customizable, boxed as array of "
84  << typeid( TAElem ) << ')' << NewLine;
85  }
86 
87  if( !lib::boxing::T_Boxing<TGiven>::value )
88  {
89  if( std::is_pointer<TGiven>::value )
90  {
91  using TNP= typename std::remove_pointer<TGiven>::type;
92  if( !lib::boxing::T_Boxing<TNP>::value )
93  {
94  target << " Customized: No" << NewLine
95  << " Target Type: " << "Ident (" << typeid(TGiven) << ')' << NewLine
96  << " Unboxable: Yes";
97  }
98  else
99  {
100  target << " Customized: Yes" << NewLine
101  << " Target Type: " << typeid(typename lib::boxing::T_Boxing<TNP>::Type);
102  if( lib::boxing::T_Boxing<TNP>::IsArray() )
103  target << "[]";
104  target << NewLine
105  << " Custom Boxing: " << (lib::boxing::T_Boxing<TNP>::HasCustomBoxing() ? "Yes" : "No" ) << NewLine
106  << " Unboxable: No";
107  }
108 
109  }
110  else
111  {
112  using TP= TGiven*;
113  if( !lib::boxing::T_Boxing<TP>::value )
114  {
115  target << " Customized: No" << NewLine
116  << " Target Type: " << "As pointer type (" << typeid(TP) << ')' << NewLine
117  << " Unboxable: No (pointer type only)";
118  }
119  else
120  {
121  target << " Customized: Yes" << NewLine
122  << " Target Type: " << typeid(typename lib::boxing::T_Boxing<TP>::Type);
123  if( lib::boxing::T_Boxing<TP>::IsArray() )
124  target << "[]";
125 
126  target << NewLine
127  << " Custom Boxing: " << (lib::boxing::T_Boxing<TP>::HasCustomBoxing() ? "Yes" : "No" )
128  << NewLine
129  << " Unboxable: No";
130  }
131  }
132  }
133  else
134  {
135  target << " Customized: Yes" << NewLine;
136  target << " Target Type: " << typeid(typename lib::boxing::T_Boxing<TGiven>::Type);
137  if( lib::boxing::T_Boxing<TGiven>::IsArray() )
138  target << "[]";
139 
140  target << NewLine
141  << " Custom Boxing: " << (lib::boxing::T_Boxing<TGiven>::HasCustomBoxing() ? "Yes" : "No" )
142  << NewLine
143  << " Unboxable: ";
144 
145  if( lib::boxing::T_Boxing<TGiven>::IsUnboxable() )
146  {
147  target << "Yes" << NewLine
148  << " Custom Unboxing: " << (lib::boxing::T_Boxing<TGiven>::HasCustomUnboxing() ? "Yes" : "No" );
149  }
150  else
151  target << "No";
152  }
153 
154  return target << NewLine;
155  }
