27 /** ************************************************************************************************
28  * Returns the value of an environment variable.
29  *
30  * While under posix compatible operating systems, environment variable names are
31  * case sensitive, under Microsoft windows they are not. This behavior is reflected by the
32  * default value of parameter \p sensitivity. Its options are as follows:
33  * - <b>GNU/Linux</b><br>
34  * Defaults to \b Cases::Sensitive. Can be changed by passing \b Case::Ignore.
35  * - <b>Windows OS</b><br>
36  * Defaults to \b Cases::Ignore. A run-time assertion is thrown when \b Case::Sensitive
37  * is passed, because this is not supported. In release compilations, the parameter is ignored.
38  * - <b>Mac OS</b><br>
39  * Defaults to \b Cases::Sensitive. An compile time assertion is given when \b Case::Ignore
40  * is passed, because this is not supported. In release compilations, the parameter is ignored.
41  *
42  * @param varName The name of the variable to be retrieved.
43  * @param target An AString to write the result to.
44  * @param targetData If \c CurrentData::Keep, the parameter \p target is not cleared
45  * before the result is written. Defaults to \c CurrentData::Clear.
46  * @param sensitivity Use case sensitive mode for variable names. Not supported under
47  * Microsoft Windows OS. Defaults to
48  * \ref aworx::lib::lang::Case::Ignore "Case::Ignore".
49  * @return \c true if variable was found.
50  **************************************************************************************************/

59  #elif defined(_WIN32)

61  #else
