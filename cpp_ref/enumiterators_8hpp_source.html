41 /**
42  * # Introduction #
43  * Standard iterator functionality \c std::iterator and C++ 11 keyword <c>for( : )</c> are not
44  * supported with C++ enumerations - and this is obvious because enumerations are types and not
45  * containers or other iteratable objects.<br>
46  *
47  * Nevertheless it would still be nice if they were!
48  *
49  * \~Comment #################################################################################### \~
50  * # %ALib %Enum Iteration #
51  *
52  * To have an easy mechanism for iterating over enum types, this TMP struct may be specialized.
53  * The restriction for doing this for template type \p TEnum needs to fulfill the following
54  * constraint:<br>
55  *
56  * <b><em>
57  * TEnum is "sparsely" defined, which means the difference of the
58  * integer value which is underlying each of its elements, is always <c>+1</c>, starting from
59  * the first to the last element.<br>
60  * (While the first element does not necessarily need to start with integer value \c 0.)
61  * </em></b>
62  *
63  * \attention \alib is not able to check if this requirement is met for a given type.
64  * It is the users responsibility do ensure this and specialize this TMP struct only for
65  * such types.
66  *
67  * As a sample, let us look at the simple enum type introduced in documentation of
68  * \alib{lang,T_EnumMetaDataDecl}:
69  *
70  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_CUSTOM_REPEAT
71  *
72  * It is obvious that this enum fulfills the requirements.
73  * In that same sample, a code iterating over all enumerations was presented:
74  *
75  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_CUSTOM_SAMPLE
76  *
77  * Here, the enums are stuffed in a \c std::vector to be iteratable. This is inefficient and
78  * duplicate code in respect to naming all enums in two places.
79  *
80  * So let us specialize this struct for \c MyNamespace::Pets. We use helper macro
81  * \ref ALIB_LANG_ENUM_IS_ITERATABLE to do so:
82  *
83  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_CUSTOM_MAKE_ITERATABLE
84  *
85  * Now, we do not need the vector any more. Instead we use template class \alib{lang,EnumIterator},
86  * which is provided exactly for this reason, in the \c for statement. The code can be rewritten as follows:
87  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_CUSTOM_SAMPLE_WITH_ITERATOR
88  *
89  * \~Comment #################################################################################### \~
90  * # Details #
91  *
92  * ### %Enum Types Use "Underlying" Integer Types ###
93  * Above we used
94  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_CUSTOM_MAKE_ITERATABLE
95  * to announce enum \c Pets to \alib.
96  * Besides the enum type, the macro expects the "value of the enum element behind the last one".
97  * Well, if you find this phrasing irritating or even wrong, then look at the following
98  * correct C++ code:
99  *
100  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_IRRITATING
101  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_IRRITATING_2
102  *
103  * Well, that enough to say: It is not our fault, it no ones fault, C++ is just an effective language.
104  *
105  * ### %Add and Subtract Operators ###
106  * We look at the same code once more:
107  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_CUSTOM_MAKE_ITERATABLE
108  * You might have noticed that the term <c>MyNamespace::Pets::Snake + 1</c> usually is not valid
109  * C++ code, as we are adding an integer value to a C++ 11 scoped enum type.
110  *
111  * The reason why this still compiles is that \alib provides operators
112  * - \ref operator+<TEnum, int>
113  * - \ref operator-<TEnum, int>
114  * which accepts an \p TEnum at the left hand side and an \c int on the right.
115  *
116  * Similar to various operator functions provided with \alib that are only applicable to types that
117  * have a specialization of TMP struct \alib{lang,T_EnumIsBitwise} in place, these operators
118  * are only applicable to types that have a specialization of this TMP struct,
119  * \b %T_EnumIsIteratable in place, what is just happening within that code line!
120  *
121  *
122  * ### Stop Enums ###
123  * In the sample discussed, <c>Pets::Snake + 1</c> was used as the "end value" of an iteration.
124  * This is error prone in the respect that if the enumeration type gets extended, our macro
125  * invocation might be changes, as <c>Pets::Snake</c> then is not the last in the list.
126  *
127  * A way out, is to add a "stopper" element to the enumeration and name it special, e.g. in
128  * upper case <c>"END_OF_ENUM"</c>. It is then rather unlikely, that some programmer would
129  * put a new element behind this one. The macro invocation would never be needed to changed:
130  *
131  * ALIB_LANG_ENUM_IS_ITERATABLE(MyEnum, MyEnum::END_OF_ENUM )
132  *
133  * A next advantage is that within the enum declaration itself it becomes obvious that this is
134  * an iteratable enum type and somewhere the specialization for \alib{lang,T_EnumIsIteratable}
135  * is found.<br>
136  * Of-course, the drawback is that an enum element is presented to the C++ compiler that is not
137  * an element like the other ones...
138  *
139  * In consideration to \alib \alib{lang,T_EnumMetaDataDecl,enum meta data feature}, adding such a
140  * stopper is no problem. At least for the built-in mechanisms. When custom meta data types
141  * are used, then the programmer may judge which way to go.
142  *
143  * ### Start Enums ###
144  * So far, all our samples used macro
145  * - \ref ALIB_LANG_ENUM_IS_ITERATABLE
146  *
147  * to specialize this struct. In fact, this macro is just a shortcut to macro
148  * - \ref ALIB_LANG_ENUM_IS_ITERATABLE_BEGIN_END
149  *
150  * passing <c>TEnum(0)</c> as a start value.
151  *
152  * This lifts the restriction of having integer \c 0 underlying the first enum element.
153  * To do might be useful when different types scoped enum values passed around
154  * as values (which can be done in a type safe way using \alib \alib{lang,Enum}). Of-course,
155  * this is never recommended, but there are situations where it is just a must, i.e. if result
156  * codes have to be passed to libraries of different programming languages. It is still better
157  * to use scope enums internally in the C++ code and "export" their values as needed than
158  * using non-scoped enums internally.
159  *
160  * ### Iterator Type ###
161  * The \c std::iterator returned with methods \alib{lang,EnumIterator::begin} and
162  * \alib{lang,EnumIterator::begin} implements the standard library concept of "RandomAccessIterator".
163  * Hence, it has various operators, including subscript \c operator[].
164  *
165  * \~Comment #################################################################################### \~
166  * ## Bitwise %Enums And Iteration ##
167  * Iteration works well, if an enumeration is declared \alib{lang,T_EnumIsBitwise,bitwise}.
168  * The restriction described in the introduction above, that the enum must not be "sparsely"
169  * defined, in this case means that, every next enum element has the next bit set,
170  * hence its internal value is doubled.
171  *
172  * If this struct is specialized for a bitwise enum type using shortcut macro
173  * \ref ALIB_LANG_ENUM_IS_ITERATABLE, the start element with integer value \c 1 is chosen.
174  * If \ref ALIB_LANG_ENUM_IS_ITERATABLE_BEGIN_END is used, iteration might start on higher values.
175  *
176  * In short: everything said above, equally applies to bitwise enumerations.
177  *
178  *
179  * \~Comment #################################################################################### \~
180  * ## Performance Considerations ##
181  * Class \alib{lang,EnumIterator} is empty in respect to fields. Created on the stack there
182  * is no performance penalty. The same is true for the internal iterator type, which is
183  * returned with class \alib{lang,EnumIterator::begin} and class \alib{lang,EnumIterator::end}.
184  * This iterator class uses an \p TEnum element as its only field member.
185  * While the code with operators, casting and conversion seems quite complex, at least with
186  * compiler optimizations turned on (release builds), the loop will perform the same as an integer
187  * while loop:
188  *
189  * int i= 0; int stop= 5;
190  * while( i++ < stop) { ... }
191  *
192  * This is a matter of the magic of C++ templates!
193  *
194  * \~Comment #################################################################################### \~
195  * # Reference Documentation #
196  *
197  * \note
198  * The methods of the base version of this TMP struct are \b not defined, but rather
199  * "tricked" into this documentation.
200  * Only if specialized with corresponding macros
201  * \ref ALIB_LANG_ENUM_IS_ITERATABLE and
202  * \ref ALIB_LANG_ENUM_IS_ITERATABLE_BEGIN_END they will be implemented.
203  *
204  * @tparam TEnum The enum type to provide information for.
205  * @tparam TCheckEnum This parameter has a default expressions and <b>must not</b> be provided
206  * with specializations of this struct.
207  * It is used internally to ensure that only enum types are to be used with
208  * the first template parameter \p TEnum.
209  */
210 template<typename TEnum,
211  typename TCheckEnum= typename std::enable_if<std::is_enum<TEnum>::value>::type>
212 struct T_EnumIsIteratable : public std::false_type
213 {
214  #if defined(DOX_PARSER)
215 
216  /**
217  * Specializations have to return the first enum element of the iteration.
218  *
219  * @return The first enumeration element.
220  */
221  static constexpr TEnum Begin;
222  #else
223  static constexpr TEnum Begin= TEnum(0);
224  #endif
225 
226  #if defined(DOX_PARSER)
227  /**
228  * Specializations have to return the element value after the last enum element of the
229  * iteration.
230  *
231  * @return The element after the last enumeration element.
232  */
233  static constexpr TEnum End;
234  #else
235  static constexpr TEnum End = TEnum(0);
236  #endif
237 
238 };

329 int getHighestBit(uint64_t value)
330 {
331  int digits;
332  #if defined(__GNUC__) || defined(_WIN64)
333 
334  #if defined(__GNUC__)
335  digits= 64 - __builtin_clzll( value );
336  #else
337  unsigned long Index;
338  _BitScanReverse64( &Index, value );
339  digits= Index + 1;
340  #endif
341 
342  #else
343  // rough
344  digits = value < (((uint64_t)1) << 8) ? 8
345  : value < (((uint64_t)1) << 16) ? 16
346  : value < (((uint64_t)1) << 32) ? 32
347  : value < (((uint64_t)1) << 48) ? 48 : 64;
348 
349  // fine with loop
350  uint64_t actSize= ((uint64_t)1) << (digits - 1);
351  while( value < actSize )
352  {
353  digits--;
354  actSize >>= 1;
355  }
356  #endif
357 
358  return digits;
359 }

364 /** ************************************************************************************************
365  * Implements a \c std::iterator class for scoped and non-scoped enum types.
366  * The documentation is found with TMP struct \alib{lang,T_EnumIsIteratable}, which needs to be
367  * specialized for template type \p TEnum. For other types, this class is not constructible.
368  *
369  * @tparam TEnum The enum type to iterate over.
370  * @tparam TEnableIf This parameter has a default expressions and <b>must not</b> be provided
371  * with specializations of this struct.
372  * It is used internally to ensure that only if a specialization of
373  * \alib{lang,T_EnumIsIteratable,T_EnumIsIteratable<TEnum>} exists, this class is
374  * constructible.
375  **************************************************************************************************/
376 template<typename TEnum,
377  typename TEnableIf= typename std::enable_if<aworx::lib::lang::T_EnumIsIteratable<TEnum>::value>::type >
378 struct EnumIterator
379 {
380 
381  /** ********************************************************************************************
382  * Default constructor.
383  **********************************************************************************************/
384  EnumIterator()= default;
385 
386  /** ****************************************************************************************
387  * Implementation of \c std::iterator for enum type \p TEnum. This class exposes
388  * #ConstIterator which uses <c>const TEnum*</c> and <c>const TEnum&</c> as
389  * pointer and reference types.
390  *
391  * As the name of the class indicates, this iterator satisfies the std library concept
392  * [RandomAccessIterator](http://en.cppreference.com/w/cpp/concept/RandomAccessIterator).
393  ******************************************************************************************/
394  template<typename TPointer, typename TReference>
395  class RandomAccessIteratorBase
396  : public std::iterator< std::random_access_iterator_tag, // iterator_category
397  TEnum, // value_type
398  integer, // distance
399  TPointer, // pointer
400  TReference // reference
401  >
402  {
403  protected:
404  /** The actual enum element. */
405  TEnum p;
406 
407  /** The underlying integer type. */
408  using TEnumAsInt= typename std::underlying_type<TEnum>::type;
409 
410 
411  public:
412  /** Constructor.
413  * @param _p Our initial value */
414  constexpr
415  explicit RandomAccessIteratorBase( TEnum _p = TEnum(0) ) : p(_p)
416  {
417  }
418 
419  //###################### To satisfy concept of InputIterator ######################
420 
421  /** Prefix increment operator.
422  * @return A reference to ourselves. */
423  RandomAccessIteratorBase& operator++()
424  {
425  if( !T_EnumIsBitwise<TEnum>::value )
426  p= p + 1;
427  else
428  p= TEnum( EnumValue( p ) << 1 );
429 
430  return *this;
431  }
432 
433  /** Postfix increment operator.
434  * @return A reference to ourselves. */
435  RandomAccessIteratorBase operator++(int)
436  {
437  if( !T_EnumIsBitwise<TEnum>::value )
438  return RandomAccessIteratorBase(p= p + 1);
439  else
440  return RandomAccessIteratorBase(p= TEnum( EnumValue( p ) << 1) );
441  }
442 
443  /** Comparison operator.
444  * @param other The iterator to compare ourselves to.
445  * @return \c true if this and given iterator are equal, \c false otherwise. */
446  constexpr
447  bool operator==(RandomAccessIteratorBase other) const
448  {
449  return p == other.p;
450  }
451 
452  /** Comparison operator.
453  * @param other The iterator to compare ourselves to.
454  * @return \c true if this and given iterator are not equal, \c false otherwise. */
455  constexpr
456  bool operator!=(RandomAccessIteratorBase other) const
457  {
458  return !(*this == other);
459  }
460 
461  /** Retrieves the enum element that this iterator references.
462  * @return The enum element. */
463  constexpr
464  TEnum operator*() const
465  {
466  return p;
467  }
468 
469 
470  //################## To satisfy concept of BidirectionalIterator ##################
471 
472  /** Prefix decrement operator.
473  * @return A reference to ourselves. */
474  RandomAccessIteratorBase& operator--()
475  {
476  if( !T_EnumIsBitwise<TEnum>::value )
477  p= p + 1;
478  else
479  p= TEnum( );
480  return *this;
481  }
482 
483 
484  /** Postfix decrement operator.
485  * @return An iterator that with the old value. */
486  RandomAccessIteratorBase operator--(int)
487  {
488  if( !T_EnumIsBitwise<TEnum>::value )
489  return RandomAccessIteratorBase(p= p - 1);
490  else
491  return RandomAccessIteratorBase(p= TEnum( EnumValue( p ) >> 1) );
492  }
493 
494 
495  //################## To satisfy concept of RandomAccessIterator ###################
496 
497  /** Addition assignment.
498  * @param n The value to subtract.
499  * @return A reference to ourselves. */
500  RandomAccessIteratorBase& operator+=(TEnumAsInt n)
501  {
502  if( !T_EnumIsBitwise<TEnum>::value )
503  p= p + n;
504  else
505  p= TEnum( EnumValue( p ) << n );
506  return *this;
507  }
508 
509  /** Subtraction assignment.
510  * @param n The value to subtract.
511  * @return A reference to ourselves. */
512  RandomAccessIteratorBase& operator-=(TEnumAsInt n)
513  {
514  if( !T_EnumIsBitwise<TEnum>::value )
515  p= p - n;
516  else
517  p= TEnum( EnumValue( p ) >> n );
518  }
519 
520  /** Addition.
521  * @param n The value to subtract.
522  * @return A reference to the new iterator. */
523  RandomAccessIteratorBase operator+(TEnumAsInt n) const
524  {
525  if( !T_EnumIsBitwise<TEnum>::value )
526  return RandomAccessIteratorBase( p + n );
527  else
528  return RandomAccessIteratorBase( TEnum( EnumValue( p ) << n ) );
529  }
530 
531  /** Subtraction.
532  * @param n The value to subtract.
533  * @return A reference to the new iterator. */
534  RandomAccessIteratorBase operator-(TEnumAsInt n) const
535  {
536  if( !T_EnumIsBitwise<TEnum>::value )
537  return RandomAccessIteratorBase( p - n );
538  else
539  return RandomAccessIteratorBase( TEnum( EnumValue( p ) >> n ) );
540  }
541 
542  /** Difference (distance) from this iterator to the given one.
543  * @param other The iterator to subtract
544  * @return The iterator to subtract. */
545  integer operator-(RandomAccessIteratorBase other) const
546  {
547  if( !T_EnumIsBitwise<TEnum>::value )
548  return static_cast<integer>(EnumValue(p) - EnumValue(other.p));
549  else
550  return static_cast<integer>(getHighestBit(static_cast<uint64_t>( EnumValue( p ) )))
551  - static_cast<integer>(getHighestBit(static_cast<uint64_t>( EnumValue( other.p ) )));
552  }
553 
554  /** Subscript operator.
555  * @param n The iterator to subtract
556  * @return <c>*( (*this) + n )</c>. */
557  TEnum operator[]( integer n ) const
558  {
559  if( !T_EnumIsBitwise<TEnum>::value )
560  return ( p + static_cast<TEnumAsInt>( n ) );
561  else
562  return TEnum( EnumValue( p ) << n );
563  }
564 
565  //#### Comparison operators (also needed to satisfy concept of RandomAccessIterator) ###
566 
567  /** Compares this iterator with the given one.
568  * @param other The iterator to compare
569  * @return \c true if this iterator is \e smaller than \p other,
570  * \c false otherwise. */
571  bool operator<(RandomAccessIteratorBase other) const
572  {
573  return p < other.p;
574  }
575 
576  /** Compares this iterator with the given one.
577  * @param other The iterator to compare
578  * @return \c true if this iterator is \e smaller than or equal to \p other,
579  * \c false otherwise. */
580  bool operator<=(RandomAccessIteratorBase other) const
581  {
582  return p <= other.p;
583  }
584 
585 
586  /** Compares this iterator with the given one.
587  * @param other The iterator to compare
588  * @return \c true if this iterator is \e greater than \p other,
589  * \c false otherwise. */
590  bool operator>(RandomAccessIteratorBase other) const
591  {
592  return p > other.p;
593  }
594 
595  /** Compares this iterator with the given one.
596  * @param other The iterator to compare
597  * @return \c true if this iterator is \e greater than or equal to \p other,
598  * \c false otherwise. */
599  bool operator>=(RandomAccessIteratorBase other) const
600  {
601  return p >= other.p;
602  }
603  };
