17 /** ************************************************************************************************
18  * # Introduction # {#alib_namespace_strings_propertyformatter_map_overview}
19  *
20  * This template class provides a map of \alib{strings::format,PropertyFormatter} objects that use
21  * format strings which are defined by variables of a \alib{config,Configuration}.
22  *
23  * The use case is about having different versions of how an object is formatted and
24  * to have these versions configurable to end users through the \alib configuration system.
25  *
26 \~Comment ####################################################################################### \~
27  * # Sample # {#alib_namespace_strings_propertyformatter_map_sample}
28  * The concept is best explained by a code sample. From the documentation of class
29  * \alib{strings::format,PropertyFormatter}, we are using the same setup as introduced in
30  * documentation of class \alib{strings::format,PropertyFormatter}:
31  *
32  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_1
33  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_2
34  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_3
35  *
36  * We want to be able to have three different output formats: \e Short, \e Default and \e All.
37  * We define an enumeration for this that we declare to be representing configuration variables
38  * using macro \ref ALIB_CONFIG_VARIABLES "ALIB_CONFIG_VARIABLES":
39  *
40  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_1
41  *
42  * The variables need to be resourced. For this sample, we do this right in our sample source.
43  * Of-course, this code usually is placed in method \alib{lang,Library::init}. Also, note that this
44  * sample simply uses the \alib library object. Usually a custom application would provide its own
45  * library:
46  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_2
47  *
48  * Note that the default value and variable comment string would be saved when we write
49  * back the variables to a configuration file. This hints users to the fact that the output format
50  * is changeable, i.e. when they open an INI-file.
51  * See \alib{config,Configuration::FetchFromDefault} for information about how to write built-in
52  * default variables like this to an external configuration file.
53  *
54  * With this in place, the \b %PropertyFormatters can be created and its method
55  * #Format can be used:
56  *
57  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_31
58  *
59  * But we're not done yet: The next goal is to have a lightweight object
60  * that is \"\alib{strings,T_Apply,applicable}\" to objects of class \alib{strings,AString}.
61  * Such object is given with template type \alib{strings::format,PropertyFormatterMapApplicable}.
62  * It is a simple helper struct that holds a reference to the the formatter map, the format
63  * enumeration value and the object to format.<br>
64  * We define a shortcut to the type with the right template arguments in place:
65  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_32
66  *
67  * Now, we are set and can start using the formatter:
68  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_4
69  *
70  * Running the code above produces the following output:
71  * \snippet "DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP.txt" OUTPUT
72  *
73  * To make "application calls" to \b %AString even shorter, macros may be used:
74  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_5
75  *
76  * The output is:
77  * \snippet "DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_2.txt" OUTPUT
78  *
79 \~Comment ####################################################################################### \~
80  * # Using The Applicable With Formatters # {#alib_namespace_strings_propertyformatter_map_usingwithfmts}
81  * The sample above showcased how to \"\alib{strings,T_Apply,apply}\" a lightwight object of helper
82  * class \alib{strings::format,PropertyFormatterMapApplicable} to objects of class \alib{strings,AString}.
83  * A final step is to define the \ref aworx::lib::boxing "boxing" interface
84  * \alib{strings::boxing,IApply} to make boxed objects of our helper type applicable. As the type
85  * itself is already applicable, we can use templated helper
86  * \alib{strings::boxing,IApply_TApplicable} for that:
87  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_6
88  * As described in the namespace documentation of \alib boxing, such interface definition should
89  * be done only once and be placed in the bootstrap section of a software.
90  *
91  * With this, using formatters in combination with our helper struct works fine:
92  * \snippet "DOX_ALIB_STRINGS.cpp" DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_7
93  *
94  * This code produces the following output.
95  * \snippet "DOX_ALIB_STRINGS_PROPERTY_FORMATTER_MAP_3.txt" OUTPUT
96  *
97 \~Comment ####################################################################################### \~
98  * # Reference Documentation #
99  * @throws aworx::lib::strings::format::Exceptions::MissingConfigurationVariable
100  *
101  * @tparam TFormattable The type that is formatted
102  * @tparam TVariable The enumeration of variables which define the format strings.
103  **************************************************************************************************/
104 template<typename TFormattable, typename TVariable >
105 class PropertyFormatters
106 {
107  protected:
108  /** The map of formatters */
109  std::map< TVariable, PropertyFormatter<TFormattable>* > formatters;
110 
111  /** The callback table for the property formatters. (As given in the constructor.) */
112  typename PropertyFormatter<TFormattable>::TCallbackTable & callbacks;
113 
114  /** The base formatter to use. (As given in the constructor.) */
115  Formatter* baseFormatter;
116 
117  protected:
118 
119  public:
120  /**
121  * Constructor storing arguments in fields.
122  * All default values of the variables are stored in the \ref aworx::lib::config "configuration"
123  * and thus - if the configuration is write enabled - can be noted by the end-user to be
124  * configurable.
125  *
126  * @param callbackTable The callback table as specified with
127  * \alib{strings::format,PropertyFormatter::TCallbackTable}.
128  * @param formatter The underlying formatter. Used for construction of
129  * \alib{strings::format,PropertyFormatter}s.<br>
130  * Defaults to \c nullptr which denotes to use method
131  * \alib{strings::format,Formatter::AcquireDefault} to get the
132  * default formatter.
133  */
134  PropertyFormatters( typename PropertyFormatter<TFormattable>::TCallbackTable& callbackTable,
135  Formatter* formatter= nullptr )
136  : callbacks ( callbackTable )
137  , baseFormatter ( formatter )
138  {
139  auto varMetaData= EnumMetaData<TVariable>::GetSingleton();
140  varMetaData->CheckLoad();
141  Variable var;
142  for( auto& optionRecord : varMetaData->Table )
143  lib::lang::T_Resourced<TVariable>::Lib().Config->StoreDefault( var.Declare( TVariable( std::get<0>( optionRecord ) ) ) );
144  }
145 
146  /**
147  * Destructor. Deletes the \alib{strings::format,PropertyFormatter}s which got created.
148  */
149  ~PropertyFormatters()
150  {
151  for( auto& formatter : formatters )
152  delete formatter.second;
153  }
154 
155  /**
156  * Chooses or - if not yet available - creates the right \alib{strings::format,PropertyFormatter}
157  * and invokes its method \alib{strings::format,PropertyFormatter::Format,Format}.
158  *
159  * @param target The target string to write into.
160  * @param option The user defined formatting option.
161  * @param formattable The custom object which is passed to the callback methods to collect
162  * the formatter arguments.
163  */
164  void Format( AString& target, TVariable option, TFormattable& formattable )
165  {
166  auto formatter= formatters.find( option );
167  if( formatter == formatters.end() )
168  {
169  // find option in the table
170  Variable var;
171  auto varMetaData= EnumMetaData<TVariable>::GetSingleton();
172  for( auto& varEntry : varMetaData->Table )
173  {
174  TVariable varEnum= TVariable( std::get<0>(varEntry) );
175  if( varEnum == option )
176  {
177  if( lib::lang::T_Resourced<TVariable>::Lib().Config->Load( var.Declare( varEnum ) ) == Priorities::NONE
178  || var.GetString()->IsEmpty() )
179  throw Exception(ALIB_SRCPOS_REL_NULLED, Exceptions::MissingConfigurationVariable,
180  var.Fullname );
181 
182  formatter= formatters.insert( std::make_pair(option, new PropertyFormatter<TFormattable>( var.GetString(), callbacks, baseFormatter) ) ).first;
183  break;
184  }
185  }
186 
187  ALIB_ASSERT_ERROR( formatter != formatters.end(),
188  "No entry for option <{}> found in option table of PropertyFormatters.",
189  option );
190  }
191 
192  formatter->second->Format( target, formattable );
193  }
194

196 }; // class PropertyFormatters

198 /**

200  * Objects of this type are \alib{strings,T_Apply,applicable} to \b %AStrings.
201  *
202  * A custom using statement should be used to access this type in a very short fashion,
203  * to make "applications" to \b %AStrings a short call. Alternatively one ore more preprocessor
204  * macros might even lead to shorter invocations. (See sample in class description).
205  */

240 /**
241  * Specialization of \alib{strings,T_Apply} for helper struct \b %PropertyFormatterMapApplicable
242  * used to format objects with \alib{strings::format,PropertyFormatters}.
243  * @tparam TFormattable The formattable object type.
244  * @tparam TOptionEnum The format option enumeration type.
245  */
246 template<typename TFormattable, typename TOptionEnum >
247 struct T_Apply<typename format::PropertyFormatterMapApplicable<TFormattable,TOptionEnum> > : public std::true_type
248 {
249  /**
250  * Invokes the \alib{strings::format,PropertyFormatters::Format} of the formatter map
251  * encapsulated in \p src.
252  * @param target The target string.
253  * @param src The helper struct object.
254  * @return Constant \p 1.
255  */
256  static integer Apply( AString& target, typename format::PropertyFormatterMapApplicable<TFormattable,TOptionEnum>& src )
257  {
258  src.FormatterMap.Format( target, src.Option, src.Formattable );
259  return 1;
260  }
261 };
