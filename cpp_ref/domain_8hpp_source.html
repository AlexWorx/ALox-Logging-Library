143  /** ****************************************************************************************
144  * Adds a new entry in field #Data and recursively demands the same from its sub-domains.
145  * Checks if a logger with the same name exists.

147  * @param logger The logger to add.

150  inline
151  int AddLogger( core::Logger* logger)
152  {
153  // let our root do this
154  if ( Parent != nullptr )
155  return Parent->AddLogger( logger );

252  /** ****************************************************************************************
253  * Sets the verbosity for a logger of this domain of all its sub domains to the specified
254  * value. If given priority is lower than those actually stored, nothing is set and
255  * recursion is stopped.
256  *
257  * @param loggerNo The number of the \e Logger to set the \e Verbosity for.
258  * @param verbosity The verbosity value to set.
259  * @param priority The priority of the setting.
260  * @return The new \e Verbosity.
261  ******************************************************************************************/
262  inline
263  Verbosity SetVerbosity( int loggerNo, Verbosity verbosity, int priority )
264  {
265  LoggerData& ld= Data[static_cast<size_t>(loggerNo)];
266  if( priority >= ld.Priority )
267  {
268  ld.Priority= priority;
269  ld.LoggerVerbosity= verbosity;
270 
271  for( Domain* subDomain : SubDomains )
272  subDomain->SetVerbosity( loggerNo, verbosity, priority );
273  }
274  return ld.LoggerVerbosity;
275  }

311  /** ****************************************************************************************
312  * Determines if the domain is active in respect to the given Verbosity.
313  *
314  * @param loggerNo The number of the \e Logger whose \e Verbosity is to be evaluated against
315  * \p statement.
316  * @param statement The \e Verbosity to check.
317  * @return \c true if domain is active (log should be performed)
318  ******************************************************************************************/
319  inline
320  bool IsActive( int loggerNo, Verbosity statement )
321  {
322  Verbosity domain= GetVerbosity( loggerNo );
323 
324  // domain ^ / stmnt > | Off Error Warning Info Verbose
325  // ---------------------------------------------------------------------
326  // Off | - - - - -
327  // Errors | - Y - - -
328  // Warning | - Y Y - -
329  // Info | - Y Y Y -
330  // Verbose | - Y Y Y Y
331 
332  if( statement != Verbosity::Off
333  && ( ( domain == Verbosity::Error && statement == Verbosity::Error )
334  || ( domain == Verbosity::Warning && ( statement == Verbosity::Warning || statement == Verbosity::Error ) )
335  || ( domain == Verbosity::Info && statement != Verbosity::Verbose )
336  || domain == Verbosity::Verbose )
337  )
338  {
339  Data[static_cast<size_t>(loggerNo)].LogCallsPerDomain++;
340  return true;
341  }
342 
343  return false;
344  }
