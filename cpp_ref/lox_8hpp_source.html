184  {
186  public: enum class Type
187  {

203  DomainSubstitutionRule( const TString& s, const TString& r )
204  {
205  ALIB_WARN_ONCE_PER_INSTANCE_DISABLE( Search, ReplaceExternalBuffer );
206  ALIB_WARN_ONCE_PER_INSTANCE_DISABLE( Replacement, ReplaceExternalBuffer );
207 
208  // get type and adjust given search parameter
209  int startPos= 0;
210  int length= s.Length();
211  if ( s.CharAtStart() == '*' )
212  {
213  startPos++;
214  length--;
215  if ( s.CharAtEnd() == '*' )
216  {
217  type= Type::Substring;
218  length--;
219  }
220  else
221  type= Type::EndsWith;
222  }
223  else
224  {
225  if ( s.CharAtEnd() == '*' )
226  {
227  type= Type::StartsWith;
228  length--;
229  }
230  else
231  type= Type::Exact;
232  }
233  Search._( s, startPos, length );
234 
235  // minimum rule check
236  if ( ( ( type == Type::Exact
237  || type == Type::StartsWith )
238  && !Search.StartsWith( "/" )
239  )
240  || ( type == Type::EndsWith
241  && Search.EndsWith( "/" )
242  )
243  )
244  Search._(); // illegal rule
245 
246 
247  Replacement= r;
248  }
249  };
