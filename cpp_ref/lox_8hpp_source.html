236  /** Constructor.
237  * @param s The path to search.
238  * @param r The replacment.
239  */
240  DomainSubstitutionRule( const String& s, const String& r )
241  {
242  ALIB_WARN_ONCE_PER_INSTANCE_DISABLE( Search, ReplaceExternalBuffer );
243  ALIB_WARN_ONCE_PER_INSTANCE_DISABLE( Replacement, ReplaceExternalBuffer );
244 
245  // get type and adjust given search parameter
246  integer startPos= 0;
247  integer length= s.Length();
248  if ( s.CharAtStart() == '*' )
249  {
250  startPos++;
251  length--;
252  if ( s.CharAtEnd() == '*' )
253  {
254  type= Type::Substring;
255  length--;
256  }
257  else
258  type= Type::EndsWith;
259  }
260  else
261  {
262  if ( s.CharAtEnd() == '*' )
263  {
264  type= Type::StartsWith;
265  length--;
266  }
267  else
268  type= Type::Exact;
269  }
270  Search._( s, startPos, length );
271 
272  // minimum rule check
273  if ( ( ( type == Type::Exact
274  || type == Type::StartsWith )
275  && !Search.StartsWith( "/" )
276  )
277  || ( type == Type::EndsWith
278  && Search.EndsWith( "/" )
279  )
280  )
281  Search._(); // illegal rule
282 
283 
284  Replacement= r;
285  }

409  /** ****************************************************************************************
410  * Adds \p path to an internal list of substrings that are used to trim the path of
411  * a source file name. Trimmed paths are used for \e Scope mechanisms and can be
412  * logged (e.g. with meta information of class \b TextLogger.
413  *
414  * By default such setting affects all instances of class \b Lox, not only
415  * this instance. This can be altered using parameter \p global.
416  * one other The given trim information can either
417  *
418  * If given \p path starts with character <c> '*'</c>, the rest of the string is searched
419  * within source paths. Otherwise, it is checked if a source path starts with the given
420  * path.
421  *
422  * Parameter \p trimReplacement optionally provides a replacement string for the trimmed
423  * path. This can be used for example to provide the right absolute path for an IDE
424  * to find source files of a library.
425  *
426  * Parameter \p includeString determines if the searched substring should be included in the
427  * resulting source path or not. In addition, parameter \p trimOffset, which can be negative
428  * or positive, is added to the position of trimming. This can be used to increase the
429  * length of the search path, and then cut only a portion of what was searched for.
430  *
431  * Finally, parameter \p sensitivity determines whether the match is performed case
432  * sensitive or not. It defaults to non-sensitive, for convenience and for the fact that
433  * for example Microsoft C++ compilers' preprocessor passes lower case path-strings!
434  *
435  * \note
436  * If the platform (compiler) specific path separator is <c>'/'</c>, then characters
437  * <c>'\'</c> found in parameters \p path and \p trimReplacement are replaced by <c>'\'</c>
438  * and vice versa. This allows to specify paths and substrings thereof in a platform
439  * independent way.
440  *
441  * \attention
442  * Setting global rules (when parameter \p global equals \c Inclusion::Include) is not
443  * protected by a \c mutex against concurrent access. Therefore, global rules have
444  * to be either at bootstrap of a process, before threads are created, or such creation
445  * has to 'manually' be protected by locking all existing instances of this class!
446  *
447  * @param path The path to search for. If not starting with <c> '*'</c>,
448  * a prefix is searched.
449  * @param includeString Determines if \p path should be included in the trimmed path or not.
450  * Optional and defaults to \b %Inclusion::Exclude.
451  * @param trimOffset Adjusts the portion of \p path that is trimmed.
452  * Optional and defaults to \c 0.
453  * @param sensitivity Determines if the comparison of \p path with a source files' path
454  * is performed case sensitive or not.
455  * Optional and defaults to \b Case::Ignore.
456  * @param trimReplacement Replacement string for trimmed portion of the path.
457  * Optional and defaults to \b %NullString.
458  * @param reach Denotes whether the rule is applied locally (to this \b %Lox only)
459  * or applies to all instances of class \b %Lox.
460  * Defaults to \b %Reach::Global.
461  * @param priority The priority of the setting. Defaults to
462  * \ref aworx::lib::config::Configuration::PrioDefault "Configuration::PrioDefault".

558  /** ****************************************************************************************
559  * Sets the \e %Verbosity of the <em>Log Domain</em> which is evaluated from parameter \p domain and
560  * applicable <em>Scope Domains</em>. The \p verbosity given, is set recursively for all sub-domains.
561  *
562  * With the first invocation of this method for a distinct \p logger, this \e %Logger
563  * is registered with this \e %Lox. In this case, prior to setting the given \e Verbosity
564  * for the evaluated sub-domain, the \e Verbosity for all domains is set to
565  * \b %Verbosity::Off.
566  *
567  * To deregister a \e Logger with a \b Lox, use method #RemoveLogger.
568  * To 'disable' a \e Logger, invoke this method with parameters \p verbosity equaling to
569  * \b %Verbosity::Off and \p domain to \c "/".
570  *
571  * Optional parameter \p priority defaults to
572  * \ref aworx::lib::config::Configuration::PrioDefault "Configuration::PrioDefault",
573  * which is a lower priority than those of the standard plug-ins of external configuration
574  * data. Therefore, external configuration by default 'overwrite' settings made from
575  * 'within the source code', which simply means by invoking this method.<br>
576  * The parameter can be provided for two main reasons:
577  * - To 'lock' a verbosity setting against external manipulation.
578  * - to 'break' the standard mechanism that an invocation of this method sets all
579  * sub-domains recursively. If a sub-domain was set with a higher priority
580  * (e.g. <c>%Configuration::PrioDefault + 1</c>, then this sub-domain will not be affected
581  * by future invocations of this method with standard-priority given.
582  *
583  * For more information on how to use external configuration variables with priority and
584  * on protecting verbosity settings, consult the [ALox user manual](../manual.html).
585  *
586  * \attention
587  * The same as with most interface methods of this class, the given \p domain parameter is
588  * combined with <em>%Scope Domains</em> set for the callers' \e %Scope. In standard use
589  * cases of %ALox, the \e %Verbosity of a \e Domain is set using absolute domain path
590  * addressing. Therefore, it is recommended to have any domain path passed to this method
591  * starting with <c> '/'</c>, which suppresses the concatenation of <em>%Scope Domains</em>.<br>
592  * This is why this parameter with this method defaults to <c> '/'</c>, while with other
593  * methods of this class, it defaults to an empty string.
594  * <p>
595  * \attention
596  * Even when using an absolute domain path, <em>%Scope Domains</em> of
597  * \e %Scope.ThreadInner, will still apply. This means that from within a thread that
598  * has such <em>%Scope Domain</em> set, this method is (almost) not usable!
599  * This all aligns with the concept (advice), that \e Loggers and their \e %Verbosity
600  * are generally set outside of such scopes, hence in configuration sections of a
601  * process.<p>
602  * \attention
603  * Consequently, this method may be (mis-) used to modify the 'actual' (default) scope
604  * when explicitly giving an empty string with parameter \p domain. This is useful, to
605  * temporarily adjust a scope. But remember: %ALox was designed to avoid temporary
606  * code lines...
607  *
608  * @param logger The logger to be to be affected (case insensitive).
609  * @param verbosity The 'level of verboseness' to be set.
610  * @param domain The parent (start) domain to be set. The use of absolute paths
611  * starting with <c> '/'</c> are recommended.
612  * Defaults to root domain \"/\".
613  * @param priority The priority of the setting. Defaults to
614  * \ref aworx::lib::config::Configuration::PrioDefault "Configuration::PrioDefault".
615  ******************************************************************************************/
616  ALOX_API

701  /** ****************************************************************************************
702  * Adds a <em>Domain Substitution Rule</em>.
703  * <em>Domain Substitution</em> is performed as a last step when evaluating the domain path of a <em>Log Statement</em>,
704  * taking <em>Scope Domains</em> and the optional parameter \p domain of the statement into
705  * account.<br>
706  *
707  * <b>Wildcards</b><br>
708  * Parameter \p domainPath supports \b 'wildcard' character <c> '*'</c> at its beginning
709  * and at its end (or both). This allows to have four types of rules:
710  * - Exact match
711  * - Prefix match (\c * at the end of \p domainPath)
712  * - Suffix match (\c * at the start of \p domainPath)
713  * - Substring match (\c * at both, start and the end of \p domainPath)
714  *
715  * Only minimal checks are performed, e.g. if an exact match is requested, but \p domainPath
716  * does not start with character <c> '/'</c>. In this and some other cases, the rule is not
717  * stored and an internal warning is logged. Further checks, for example for illegal
718  * domain path characters are not performed (those will be eliminated when the resulting
719  * domain path is to be created internally).
720  *
721  * <b>Circular Dependencies</b><br>
722  * If the given rules have circular dependencies, only a limited number (ten) replacements
723  * are performed. If this number of replacements for one <em>Log Statement</em> is exceeded, an internal
724  * warning message is logged. This is done only \e once over the life-time of a \e Logger.
725  *
726  * <b>Application of Rules</b><br>
727  * Rules are applied in the order of their definition. After all rules have been applied
728  * this is repeated as long as at least one rule matched (up to ten times).
729  *
730  * <b>Deletion of Rules</b>
731  * To delete a rule, invoke the method with same parameter \p domainPath and a 'nulled'
732  * or empty string for parameter \p replacement.
733  * To delete all rules, invoke the method with parameter \p domainPath 'nulled'
734  * or empty.
735  *
736  * <b>Final remarks</b>
737  * Domain substitution is useful to permanently change ('redirect') domain paths of
738  * 3rd party code (e.g. libraries using ALox) or log statements that must not be changed
739  * for other reasons. It is advised to not 'overuse' this feature, as side effects
740  * are inherent to the concept of <em>Domain Substitution</em>. For example, an unwanted side effect might be
741  * that <em>Prefix Logables</em> are not applicable to the substituted domain, while other <em>Prefix Logables</em> are
742  * bound to the resulting domain.
743  *
744  * For \b %Lox objects that should be protected of external manipulation, it is advisable,
745  * to remove all <em>Domain Substitution Rules</em> right after the \b %Lox was created by invoking this method with
746  * a nulled value for parameter \p domainPath. The reason is, that otherwise, through
747  * configuration files or command line parameters, domains of the \b %Lox can be substituted
748  * and then the resulting domains \e Verbosities be \e overwritten using further configuration
749  * variables. Any prioritized \e 'internal' setting of \e Verbosities this way could be
750  * circumvented!
751  *
752  * For more information consult the [ALox User Manual](../man_domain_substitution.html).
753  *
754  * @param domainPath The path to search. Has to start with either <c> '/'</c> or <c> '*'</c>.
755  * @param replacement The replacement path.
756  ******************************************************************************************/
757  ALOX_API
758  void SetDomainSubstitutionRule( const String& domainPath, const String& replacement );

1357  /** ****************************************************************************************
1358  * Logs given \e logables once, up to \p quantity times, or every n-th time.
1359  * In its simplest overloaded version, the counter is bound to the source code line, hence,
1360  * only the first execution of this exact <em>Log Statement</em> is executed.
1361  *
1362  * With parameter \p group, a set of <em>Log Statements</em> that share the same group key,
1363  * can be grouped and of such set, only the one which is first executed actually logs.<br>
1364  * Alternatively, when \p key is omitted (or nullptr or empty), but a
1365  * \ref aworx::lox::Scope "Scope" is given with parameter \p scope, then the
1366  * counter is associated with the scope.<br>
1367  * Finally, parameters \p key and \p scope can also be used in combination. The key is
1368  * then unique in respect to the \ref aworx::lox::Scope "Scope" provided.
1369  *
1370  * Using, none, one or both of the parameters \p group and \p scope, among others, the
1371  * following use cases can be achieved.
1372  * - %Log a specific statement up to n-times.
1373  * - %Log only the first n of a group of statements.
1374  * - %Log only the first n statements within a method.
1375  * - %Log only the first n statements belonging to the same group and method .
1376  * - %Log only the first n statements within any method of
1377  * - a source file
1378  * - a directory of source files
1379  * - a parent directory of source files and all sources recursively
1380  * - %Log only the first n statements which belong to the same group and are placed within
1381  * any method of
1382  * - a source file
1383  * - a directory of source files
1384  * - a parent directory of source files and all sources recursively
1385  * - %Log a <em>Log Statement</em> n-times per new thread.
1386  * - %Log only the first n statements of a group of statements executed by a specific thread.
1387  *
1388  * When parameter \p quantity is a negative value, the log statement is executed every n-th time
1389  * instead n-times. E.g, if \p quantity is \c -5, the first statement is executed and afterwards
1390  * every fifth invocation.
1391  *
1392  * \note
1393  * Unlike other methods of this class which accept an arbitrary amount of logables, this
1394  * method and its overloaded variants accept only one object. To supply several objects
1395  * at once, a container of type \ref aworx::lib::boxing::Boxes "Boxes" may be passed with
1396  * parameter \p logables, like in the following sample:
1397  * \snippet "ut_alox_lox.cpp" DOX_ALOX_LOX_ONCE
1398  *
1399  *
1400  * @param domain Optional <em>Log Domain</em> which is combined with <em>%Scope Domains</em>
1401  * set for the \e %Scope of invocation.
1402  * @param verbosity The verbosity of the <em>Log Statement</em> (if performed).
1403  * @param logables The objects to log (Multiple objects may be provided within
1404  * container class Boxes.)
1405  * @param group The optional name of the statement group . If used, all statements that
1406  * share the same group name are working on the same counter (according
1407  * to the \p scope.)
1408  * If omitted (or empty or nullptr), the counter is is bound to the \e %Scope
1409  * provided. If omitted and \p scope is Scope::Global, then the
1410  * counter is associated exclusively with the single <em>Log Statement</em> itself.
1411  * @param scope The \e %Scope that the group or counter is bound to.
1412  * @param pathLevel Used only if parameter \p scope equals
1413  * \ref aworx::lox::Scope::Path "Scope::Path"
1414  * to reference parent directories. Optional and defaults to \c 0.
1415  * @param quantity The number of logs to be performed. As the name of the method indicates,
1416  * this defaults to \c 1.
1417  * If negative, the first and every "-quantity-th" statement is executed.
1418  ******************************************************************************************/
