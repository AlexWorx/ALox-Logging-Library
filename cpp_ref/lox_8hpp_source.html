200  DomainSubstitutionRule( const TString& s, const TString& r )
201  {
202  ALIB_WARN_ONCE_PER_INSTANCE_DISABLE( Search, ReplaceExternalBuffer );
203  ALIB_WARN_ONCE_PER_INSTANCE_DISABLE( Replacement, ReplaceExternalBuffer );
204 
205  // get type and adjust given search parameter
206  int startPos= 0;
207  int length= s.Length();
208  if ( s.CharAtStart() == '*' )
209  {
210  startPos++;
211  length--;
212  if ( s.CharAtEnd() == '*' )
213  {
214  type= Type::Substring;
215  length--;
216  }
217  else
218  type= Type::EndsWith;
219  }
220  else
221  {
222  if ( s.CharAtEnd() == '*' )
223  {
224  type= Type::StartsWith;
225  length--;
226  }
227  else
228  type= Type::Exact;
229  }
230  Search._( s, startPos, length );
231 
232  // minimum rule check
233  if ( ( ( type == Type::Exact
234  || type == Type::StartsWith )
235  && !Search.StartsWith( "/" )
236  )
237  || ( type == Type::EndsWith
238  && Search.EndsWith( "/" )
239  )
240  )
241  Search._(); // illegal rule
242 
243 
244  Replacement= r;
245  }
246  };
