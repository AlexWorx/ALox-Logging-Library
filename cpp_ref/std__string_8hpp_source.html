124  /** ********************************************************************************************
125  * Copies or appends a region of a given \ref aworx::lib::strings::String "String" into
126  * an object of type \c std::string.
127  * The given region is adjusted to match the length of the %String.
128  *
129  * @param src The \ref aworx::lib::strings::String "String" to convert into a std::string.
130  * @param target A result std::string to copy the specified region into.
131  * @param regionStart The start index of the region to be copied.
132  * @param regionLength The maximum length of the region to be copied.
133  * Defaults to CString::MaxLen.
134  * @param targetData If \c CurrentData::Keep, the target \c std::string is not cleared before
135  * the result is written. Defaults to \c CurrentData::Clear.
136  *
137  * @return The (modified) std::string& result that was provided (for concatenation of calls).
138  **********************************************************************************************/
139  inline
140  std::string& ToStdString( const String& src,
141  std::string& target,
142  integer regionStart,
143  integer regionLength = CString::MaxLen,
144  lang::CurrentData targetData = lang::CurrentData::Clear )
145  {
146 
147  if ( targetData == lang::CurrentData::Clear )
148  target.clear();
149 
150  // adjust range, if empty return empty string
151  if ( src.AdjustRegion( regionStart, regionLength ) )
152  return target;
153 
154  // copy our buffer into target
155  target.append( src.Buffer() , static_cast<size_t>(regionStart), static_cast<size_t>(regionLength ) );
156  return target;
157  }
158 
159  /** ********************************************************************************************
160  * Copies a region of a given \ref aworx::lib::strings::String "String" into an object of type
161  * \c std::string, which is created.
162  * The given region is adjusted to match the length of the %String.
163  *
164  * @param src The \ref aworx::lib::strings::String "String" to convert into a std::string.
165  * @param regionStart The start index of the region in this to create the std::string from.
166  * @param regionLength The maximum length of the region to be copied.
167  * Defaults to CString::MaxLen.
168  *
169  * @return A \c std::string that represents the specified sub region of this object.
170  **********************************************************************************************/
171  inline
172  std::string ToStdString( const String& src,
173  integer regionStart,
174  integer regionLength= CString::MaxLen )
175  {
176  std::string result;
177  ToStdString( src, result, regionStart, regionLength, lang::CurrentData::Keep );
178  return result;
179  }
