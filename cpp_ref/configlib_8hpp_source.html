181 /** ************************************************************************************************
182  * A data record used to declare an
183  * \ref aworx::lib::config::Variable "ALib Configuration Variable".
184  * While variables can also be declared by setting their attributes individually, it is recommended
185  * to declare all external configuration variables in a central place, using statically or
186  * dynamically allocated objects of this type.<br>
187  * Objects of class \b %Variable can be constructed and existing objects can be reused by invoking
188  * \ref aworx::lib::config::Variable::Declare "Variable::Declare". Both, construction and
189  * re-declaration of variables, use objects of this type.
190  *
191  * All fields (except #Delim) support placeholders \c "%1", \c "%2" ... \c "%N", which are
192  * replaced with the constructor of class \b %Variable and method
193  * \ref aworx::lib::config::Variable::Declare "Variable::Declare".
194  * This allows to define a series of variables whose category, name, description and value is
195  * dependent on runtime information.
196  *
197  * Besides the use for passing parameters into objects of type Variable, this struct does not
198  * provide further logic.
199  *
200  * In this C++ implementation of \alib, this class can be created from elements of enum types.
201  * For this, the concept of \alib{lang,T_EnumMetaDataDecl,enum meta data} is used. The meta data
202  * of applicable enumeration types has to follow a certain tuple type, specified with #TTuple.
203  * Macro \ref ALIB_CONFIG_VARIABLES implements this association and is to be used to make a C++ enum
204  * type compatible with the constructor of this struct. The macro in addition enables the enum
205  * meta data to be loaded from the \alib{lang,Resources} of a \alib{lang,Library}.
206  **************************************************************************************************/
207 struct VariableDecl
208 {
209  /**
210  * The type of the tuple of \alib{lang,T_EnumMetaDataDecl,enum meta data} used with
211  * enumerations that denote configuration variables.
212  *
213  * The elements of the tuple have the following meaning:
214  *
215  * <center>Index</center> | <center>Description</center>
216  * - - - - - - - - - - - -| - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
217  * 0 | The originating enum value of the element of the custom C++ enumeration type.
218  * 1 | Value for field \alib{config,Variable::Category}.
219  * 2 | Value for field \alib{config,Variable::Name}.
220  * 3 | Value for field \alib{config,Variable::DefaultValue}.
221  * 4 | Value for field \alib{config,Variable::Delim}.
222  * 5 | Value for field \alib{config,Variable::FormatAttrAlignment}.
223  * 6 | Value for field \alib{config,Variable::FmtHints}.
224  * 7 | Value for field \alib{config,Variable::Comments}.
225  *
226  * \see Macro \ref ALIB_CONFIG_VARIABLES.
227  */
228  using TTuple= lang::T_EnumMetaDataSpecification<VariableDecl>::TTuple;
229 
230

232  * The variable declaration data.
233  */
234  TTuple Tuple;

238  * Constructor usually used with static variable declarations (declarations that are not
239  * using enumeration types associated with \ref ALIB_CONFIG_VARIABLES.
240  *
241  * @param category Value for field \alib{config,Variable::Category}.
242  * @param name Value for field \alib{config,Variable::Name}.
243  * @param defaultValue Value for field \alib{config,Variable::DefaultValue}.
244  * @param delim Value for field \alib{config,Variable::Delim}.
245  * @param formatAttrAlignment Value for field \alib{config,Variable::FormatAttrAlignment}.
246  * @param formatHints Value for field \alib{config,Variable::FmtHints}.
247  * @param comments Value for field \alib{config,Variable::Comments}.
248  **********************************************************************************************/
249  inline

251  String name,
252  String defaultValue,
253  char delim,
254  String formatAttrAlignment,
255  FormatHints formatHints,

259  {}

261 
262  /** ********************************************************************************************
263  * Constructor that accepts an element of a C++ enum type equipped with meta information
264  * that contains the declaration data.
265  *
266  * First, method \alib{lang,EnumMetaData::CheckLoad} is invoked on the meta data singleton
267  * of the enum type \p TEnum. Then, meta data table entry corresponding to parameter
268  * \p declaration is assigned to field #Tuple.
269  *
270  * After that, the following steps are performed:
271  * - If tuple element \c 3 ("default value") is an empty string, the string is nulled. Otherwise
272  * the string is used as a resource name and the resulting default value is loaded from
273  * the resources.
274  * - If tuple element \c 7 ("comment") is an empty string, the string is nulled. Otherwise
275  * the string is used as a resource name and the resulting comment string is loaded from
276  * the resources
277  *
278  * For both load operations, the the same library and resource category as for the enum meta
279  * data itself is used.
280  *
281  * \see Description of this class and macro \ref ALIB_CONFIG_VARIABLES.
282  *
283  * @param declaration Element of an enum class that is representing configuration
284  * variables.
285  * @tparam TEnum The type of parameter \p declaration
286  * @tparam TEnableIf Not to be specified. Used by the compiler to select this constructor
287  * only for associated custom C++ enum types.
288  **********************************************************************************************/
289  template<typename TEnum,
290  typename TEnableIf= typename std::enable_if<
291  std::is_same< VariableDecl::TTuple, typename lang::T_EnumMetaDataDecl<TEnum>::TTuple >::value
292  >::type>
293  inline
294  VariableDecl( TEnum declaration )
295  {
296  // copy our data tuple from the enum meta data table entry
297  auto enumMetaData= EnumMetaData<TEnum>::GetSingleton();
298  enumMetaData->CheckLoad();
299  Tuple= *enumMetaData->Get( declaration );
300 
301  // if element 3 (default value) or 7 (comment) is empty, set to nullptr, else use as
302  // resource name and load
303  for(int i= 0; i<2 ; ++i )
304  {
305  String& value= i == 0 ? std::get<3>(Tuple) : std::get<7>(Tuple);
306  if( value.IsEmpty() )
307  value= NullString;
308  else
309  value= lang::T_Resourced<TEnum>::Lib().Get( value );
310  }
311  }
312
