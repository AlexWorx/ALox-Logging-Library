27 /** ************************************************************************************************
28  * This utility class implements so called <em>wildcard string matching</em>. Wildcard characters
29  * are
30  * - <c>'*'</c>: Matches zero or more characters.
31  * - <c>'?'</c>: Matches exactly one character.
32  *
33  * Method #Compile accepts the pattern string and translates it to an internal (simple) list of
34  * "matching commands". This way, the class is optimized for performance, because after compilation,
35  * subsequent invocations of #Match do not need to parse the pattern string.
36  **************************************************************************************************/
37 class WildcardMatcher
38 {
39  /**
40  * The result list of commands created with #Compile and executed with #Match.
41  */
42  std::vector<std::pair<int,String>> commands;
43 
44  public:
45  /** ****************************************************************************************
46  * Constructs a WildcardMatcher to work on a given string. Passes the optional parameters
47  * to method #Compile.
48  *
49  * @param pattern The string pattern to match.
50  * Defaults to \b NullString to allow parameterless construction,
51  * with later invocation of #Compile.
52  ******************************************************************************************/
53  inline
54  WildcardMatcher( const String& pattern= NullString )
55  {
56  Compile( pattern );
57  }
58 
59  public:
60  /** ****************************************************************************************
61  * Resets this object to use the given pattern.
62  *
63  * @param pattern The string pattern to match.
64  ******************************************************************************************/
65  ALIB_API
66  void Compile( const String& pattern );
67 
68  /** ****************************************************************************************
69  * Tests if given \p haystack matches the actual pattern.
70  * If #Compile was not invoked or an empty pattern string was given, \c true is returned.
71  *
72  * @param haystack The string to test.
73  * @param sensitivity Denotes whether the matching is performed case sensitive.
74  * Defaults to \c %Case::Sensitive.
75  * @return \c true if given \p haystack matches the actual pattern, \c false otherwise.
76  ******************************************************************************************/
77  ALIB_API
78  bool Match( const String& haystack, Case sensitivity = Case::Sensitive );
79 
80 }; // class WildcardMatcher
