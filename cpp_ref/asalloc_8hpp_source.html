683  void GrowBufferAtLeastBy( int minimumGrowth )
684  {
685  int capacity= Capacity();
686 
687  // big enough?
688  ALIB_ASSERT_WARNING ( capacity < length + minimumGrowth,
689  "Unnecessary invocation of Grow()" );
