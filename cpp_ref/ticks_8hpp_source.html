674 
679  void SetStartTime( const Ticks startTime ) { this->startTime.Set( startTime ); }
680 
681 
686  void Reset() { sum.Set( 0L ); cntSamples= 0; Start(); }
687 
688 
689  // #############################################################################################
690  // Interface to Measurement
691  // #############################################################################################
692 
703  const Ticks Sample ()
704  {
705  cntSamples++;
706  int_fast64_t st= startTime.Raw();
707  startTime.Set();
708  int_fast64_t diff= startTime.Raw() - st;
709 
710  Ticks sample( diff );
711  sum.Add( diff );
712 
713  return sample;
714  }
715 
720  int GetSampleCnt() { return cntSamples; }
721 
729  Ticks GetCumulated() { return sum; }
730 
737  Ticks GetAverage()
738  {
739  return Ticks ( cntSamples== 0 ? 0L : ( sum.Raw() / cntSamples) );
740  }
