32 /** ************************************************************************************************
33  * This utility class wraps [boost::regex library](http://www.boost.org) and interfaces
34  * \alibmod_strings with it.
35  *
36  * The availability of the class is dependent on code selection symbol \ref ALIB_FEAT_BOOST_REGEX,
37  * which is controlled by using compiler symbols #ALIB_FEAT_BOOST_REGEX_ON and
38  * #ALIB_FEAT_BOOST_REGEX_OFF (the default).
39  *
40  * Method #Compile accepts the pattern string and compiles it to \b boost::regex.
41  * Subsequent invocations of #Match will then use the compiled regular expression for testing
42  * a given string.
43  *
44  * The syntax of the regular expressions is compatible to
45  * [Perl Regular Expressions](http://perldoc.perl.org/perlretut.html).
46  *
47  * \note
48  * This a most very basic wrapper that supports a just the <b>bare minimum of the features</b>
49  * available in the original \b boost::regex library.
50  * I.e, the expression syntax is fixed to be \e Perl-compatible and no string replacement
51  * features or even match positioning are available.
52  **************************************************************************************************/
53 class RegexMatcher
54 {
55  protected:
56  /**
57  * This is the internal regex matcher. Yes, nasty \c reinterpret_cast is performed in the
58  * compilation unit. This is for avoiding the inclusion of external headers with \alib
59  * headers at the expense of an otherwise unnecessary heap allocation.
60  */
61  void* boostRegex = nullptr;
