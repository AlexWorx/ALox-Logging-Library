112  /** ****************************************************************************************
113  * Returns the time span between the current system time and the internal start value.
114  * In addition this value is added to the sum of sample times and the sample counter is
115  * increased by one. Lastly the internal reference value is set to now. Therefore, a
116  * subsequent call to this function would measure the time span from this call to this
117  * subsequent call (if the internal start time value was not set differently meanwhile).
118  *

120  * reference value.
121  ******************************************************************************************/
122  inline
123  const Ticks Sample ()

126  rawtime st= startTime.Raw();
127  startTime.Set();
128  rawtime diff= startTime.Raw() - st;
129 
130  Ticks sample( diff );

132
