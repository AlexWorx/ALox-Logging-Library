39 /** ************************************************************************************************
40  * This virtual class is used by classes derived from
41  * \ref aworx::lib::config::ConfigurationPlugin "ConfigurationPlugin"
42  * to convert external string to internal values and vice versa, and for parsing lists of values
43  * from an external string.
44  *
45  * Class \b %ConfigurationPlugin owns a default object with field
46  * \ref aworx::lib::config::ConfigurationPlugin::defaultStringConverter "ConfigurationPlugin::defaultStringConverter",
47  * to which field
48  * \ref aworx::lib::config::ConfigurationPlugin::StringConverter "ConfigurationPlugin::StringConverter"
49  * by default points to.
50  *
51  * \note
52  * Replacing the converters is deemed to be an advanced usage of \b %ALib. Consult the source code
53  * for detailed information about how to implement an application specific converter.
54  *
55  * This default implementation proceeds as follows:
56  * - "Externalizing" a value:
57  * - Value is surrounded by quotes if it starts or ends with spaces or if it includes
58  * the delimiter token.
59  * - A few characters are escaped using \c '\\'. Those are
60  * \c \\n, \c \\r, \c \\t , \c \\a, \c \\b, \c \\v, \c \\f, \c \\e and also
61  * the double quotation marks \c \\" and the backslash itself (\c \\\\).
62  *
63  * - "Internalizing" a value:
64  * - If (non-escaped) quote \c " characters are found, those are removed and whitespaces
65  * within such quotes are kept.
66  * - Escaped characters are converted to their original value.
67  *
68  * - Loading variables from external strings:
69  * - If provided variable has a valid delimiter set, this character is used to tokenize
70  * the external string.
71  * - Delimiters found within a pair of quotes \c " are ignored.
72  * - Each value found is internalized separately
73  **************************************************************************************************/
74 class XTernalizer
75 {
76  public:
77 
78  /** ****************************************************************************************
79  * Virtual destructor
80  *****************************************************************************************/
81  virtual ~XTernalizer() {}
82 
83  /** ********************************************************************************************
84  * Parses values found in string \p src and adds them to \p variable.
85  * @param variable The destination variable.
86  * @param src The source string
87  **********************************************************************************************/
88  ALIB_API
89  virtual
90  void LoadFromString( Variable& variable, const String& src );
91 
92  /** ********************************************************************************************
93  * Internalizes a variable value that is provided in externalized format.
94  * Specifically first the value is trimmed, then non-escaped quotes are removed, including
95  * any whitespaces found between quoted strings. For example, the following sequence:
96  *
97  * "one" - two" - "three"- four"
98  *
99  * would be read as:
100  *
101  * one- two - three- four
102  *
103  * Furthermore, escaped characters are converted into their original value.
104  *
105  * @param src The source string
106  * @param dest The destination string
107  **********************************************************************************************/
108  ALIB_API
109  virtual
110  void InternalizeValue( Substring& src, AString& dest );
111 
112  /** ********************************************************************************************
113  * Converts the given \p src string to an external representation.
114  *
115  * @param src The source string
116  * @param dest The destination string
117  * @param delim If this character is found in the string, the value is quoted
118  **********************************************************************************************/
119  ALIB_API
120  virtual
121  void ExternalizeValue( Substring& src, AString& dest, char delim );
122 };

308 /** ************************************************************************************************
309  * Specialization of abstract interface class #ConfigurationPlugin, which takes all command line
310  * parameters in the constructor and reads variable values from those parameters on request.
311  * Its priority value usually is \alib{config,Priorities::CLI}, which is higher
312  * than all other default plug-ins provided.
313  *
314  * Variable categories are used as a prefix together with an underscore \c '_'.
315  * This means, if variable <em>LOCALE</em> in category <em>ALIB</em> is accessed, the command line
316  * parameter <em>--ALIB_LOCALE=xyz</em> is read.
317  *
318  * Category and Variable names are insensitive in respect to character case.
319  *
320  * Command line variables may be passed with either one hyphen ('-') or two ('--').
321  * Both are accepted.
322  *
323  * An application can specify one or more "default categories" by adding their string names to
324  * public field #DefaultCategories. Variables of these categories are recognized by the plug-in
325  * also when given without the prefix of category name and underscore \c '_'.
326  *
327  * Furthermore, an application may set public field #AllowedMinimumShortCut to a value greater than
328  * \c 0. In this case, the plug-in recognizes variables in CLI arguments already when at least
329  * this amount of characters is provided. In other words, when reading an argument as many
330  * characters of the variable name as possible are 'consumed' and if a minimum number is given with
331  * #AllowedMinimumShortCut, such minimum is sufficient. If the remaining part of the argument
332  * string is either empty or continues with an equal sign \c '=', then the variable is recognized
333  * (with an empty value or the value after the equal sign).<br>
334  * Fields #AllowedMinimumShortCut and #DefaultCategories may also be used in combination.
335  **************************************************************************************************/
336 class CLIArgs : public ConfigurationPlugin
337 {
338  /** Friend detail function used for implementation of iterator. */

341  protected:
342  size_t argCount = 0; ///< The number of command line arguments

344  bool wArgs = false; ///< Determines if argv is of type '<em>wchar_t **</em>'
345  ///< or '<em>char **</em>'
346 
347  /**
348  * This vector may be set with method #SetArgs(std::vector<String>*). If set, it is used
349  * as the source of options instead of using fields #argCount and #argVector.
350  */
351  std::vector<String>* args = nullptr;
352 
353  public:
354

356  * An application can specify one or more "default categories" by adding the category names

358  * also when given without the prefix of <c>category_</c>.
359  */
360  std::vector<String> DefaultCategories;

381  /** ****************************************************************************************

383  * Should not be invoked directly. Rather use
384  * \ref aworx::lib::config::Configuration::SetCommandLineArgs "Configuration::SetCommandLineArgs".
385  *
386  *\note
