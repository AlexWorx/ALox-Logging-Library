40 /** ************************************************************************************************
41  * This virtual class is used by classes derived from
42  * \ref aworx::lib::config::ConfigurationPlugin "ConfigurationPlugin"
43  * to convert external string to internal values and vice versa, and for parsing lists of values
44  * from an external string.
45  *
46  * Class \b %ConfigurationPlugin owns a default object with field
47  * \ref aworx::lib::config::ConfigurationPlugin::defaultStringConverter "ConfigurationPlugin::defaultStringConverter",
48  * to which field
49  * \ref aworx::lib::config::ConfigurationPlugin::StringConverter "ConfigurationPlugin::StringConverter"
50  * by default points to.
51  *
52  * \note
53  * Replacing the converters is deemed to be an advanced usage of \b %ALib. Consult the source code
54  * for detailed information about how to implement an application specific converter.
55  *
56  * This default implementation proceeds as follows:
57  * - "Externalizing" a value:
58  * - Value is surrounded by quotes if it starts or ends with spaces or if it includes
59  * the delimiter token.
60  * - A few characters are escaped using \c '\\'. Those are
61  * \c \\n, \c \\r, \c \\t , \c \\a, \c \\b, \c \\v, \c \\f, \c \\e and also
62  * the double quotation marks \c \\" and the backslash itself (\c \\\\).
63  *
64  * - "Internalizing" a value:
65  * - If (non-escaped) quote \c " characters are found, those are removed and whitespaces
66  * within such quotes are kept.
67  * - Escaped characters are converted to their original value.
68  *
69  * - Loading variables from external strings:
70  * - If provided variable has a valid delimiter set, this character is used to tokenize
71  * the external string.
72  * - Delimiters found within a pair of quotes \c " are ignored.
73  * - Each value found is internalized separately
74  **************************************************************************************************/
75 class XTernalizer
76 {
77  public:
78 
79  /** ****************************************************************************************
80  * Virtual destructor
81  *****************************************************************************************/
82  virtual ~XTernalizer() {}
83 
84  /** ********************************************************************************************
85  * Parses values found in string \p src and adds them to \p variable.
86  * @param variable The destination variable.
87  * @param src The source string
88  **********************************************************************************************/
89  ALIB_API
90  virtual
91  void LoadFromString( Variable& variable, const String& src );
92 
93  /** ********************************************************************************************
94  * Internalizes a variable value that is provided in externalized format.
95  * Specifically first the value is trimmed, then non-escaped quotes are removed, including
96  * any whitespaces found between quoted strings. For example, the following sequence:
97  *
98  * "one" - two" - "three"- four"
99  *
100  * would be read as:
101  *
102  * one- two - three- four
103  *
104  * Furthermore, escaped characters are converted into their original value.
105  *
106  * @param src The source string
107  * @param dest The destination string
108  **********************************************************************************************/
109  ALIB_API
110  virtual
111  void InternalizeValue( Substring& src, AString& dest );
112 
113  /** ********************************************************************************************
114  * Converts the given \p src string to an external representation.
115  *
116  * @param src The source string
117  * @param dest The destination string
118  * @param delim If this character is found in the string, the value is quoted
119  **********************************************************************************************/
120  ALIB_API
121  virtual
122  void ExternalizeValue( Substring& src, AString& dest, char delim );
123 };

125 /** ************************************************************************************************
126  * Abstract class that represents a plug in for class
127  * \ref aworx::lib::config::Configuration "Configuration"
128  * to provide configuration data from specific configuration data source.
129  *
130  * See documentation of namespace #aworx::lib::config for more information on \b %ALib
131  * external configuration variables.
132  *
133  * The plug-in also may have the ability to write data. The default implementation for writing
134  * configuration data returns constant false, indicating that this plug-in never writes
135  * anything (e.g. command line parameter plug-in, environment variable plug-in).
136  *
137  * Category and Variable names are character case insensitive for the plug-ins predefined
138  * with \b %ALib.
139  * It is recommended to to ignore character case in custom specializations of this class
140  * as well.
141  **************************************************************************************************/
142 class ConfigurationPlugin
143 {
144  // #############################################################################################
145  // internal fields
146  // #############################################################################################
147  protected:
148  /** The default external string converter */
149  XTernalizer defaultStringConverter;
150 
151 
152  // #############################################################################################
153  // public fields
154  // #############################################################################################
155  public:
156  /** The external string converter.
157  * By default this points to field #defaultStringConverter. */
158  XTernalizer* StringConverter;
159 
160  // #############################################################################################
161  // protected constructor/destructor
162  // #############################################################################################
163  protected:
164  /** ****************************************************************************************
165  * Constructor which is protected, as this is an abstract class.
166  ******************************************************************************************/
167  ConfigurationPlugin() { StringConverter= &defaultStringConverter; }
168 
169  /** ****************************************************************************************
170  * Virtual Destructor.
171  ******************************************************************************************/
172  virtual ~ConfigurationPlugin() {}
173 
174  // #############################################################################################
175  // abstract/virtual interface
176  // #############################################################################################
177  public:
178 
179  /** ****************************************************************************************
180  * Abstract method that has to be overwritten by descendants.
181  * Searches and by default retrieves the value of a configuration variable.
182  * If \p searchOnly is \c true, then the variable value is not read.
183  *
184  * @param variable The variable to retrieve.
185  * @param searchOnly If \c true, the variable must not be set. Defaults to \c false.
186  * @return \c true if variable was found within this configuration source, \c false if not.
187  ******************************************************************************************/
188  ALIB_API
189  virtual bool Load( Variable& variable, bool searchOnly= false ) const =0;
190 
191  /** ****************************************************************************************
192  * Writes a variable to the configuration.
193  * This default implementation just returns \c false. If this method is not overwritten
194  * in descendants, those descendants are not designed to write data.
195  *
196  * @param variable The variable to store.
197  *
198  * @return \c true if the variable was written, \c false if not which typically indicates
199  * that this plug-in is not able to write values.
200  ******************************************************************************************/
201  virtual bool Store( Variable& variable )
202  {
203  (void) variable;
204  return false;
205  }
206 
207  /** ****************************************************************************************
208  * Convenience method that parses the values from the given string using field
209  * #StringConverter and then invokes \ref Store(Variable&) "Store".
210  *
211  * @param variable The variable to store.
212  * @param externalizedValue The value to parse into the variable before storing
213  *
214  * @return \c true if the variable was written, \c false if not which typically indicates
215  * that this plug-in is not able to write values.
216  ******************************************************************************************/
217  virtual bool Store( Variable& variable, const String& externalizedValue )
218  {
219  if( externalizedValue.IsNotNull() )
220  StringConverter->LoadFromString( variable, externalizedValue );
221  return Store( variable );
222  }
223 };
