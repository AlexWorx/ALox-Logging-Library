64 /**
65  * ## Introduction ##
66  * In software, enumerations often need more static/constant information along with each
67  * enumeration element. This TMP struct is used to associate meta data with the elements
68  * of non-scoped or C++ 11 scoped enum types.<br>
69  * The meta data is stored in a a \alib{lang,Singleton,singleton} of type
70  * \alib{lang,EnumMetaData} within a \c std::vector of user defined \c std::tuple objects.
71  * Besides the vector, a reference to the data sets is available through a hash map that uses
72  * the underlying integer value of the elements of the enumeration as its hash key.
73  *
74  * While the code of this part of \alib uses template meta programming (TMP), when using the library
75  * this is not noticed as the otherwise complicated parts are supported by preprocessor macros
76  * which are very simple to use. The next section introduces a simple sample to implement
77  * an "element name translation table". While this is a most simple form of meta data,
78  * it is so frequently needed, that programming languages like C# or Java provide such table as a
79  * built-in language feature!
80  *
81  * \~Comment #################################################################################### \~
82  *
83  * ## %Enum Element Name Output ##
84  * One obvious use case of enum meta data is the frequent requirement of writing enum elements
85  * in human readable form (e.g. log or configuration files).
86  * To achieve this, a character string needs to be stored
87  * for every element of the enumeration.<br>
88  * For this use case, macro \ref ALIB_LANG_ENUM_NAMED does almost all that is needed.
89  * Consider the following sample code:
90  *
91  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_NORMAL
92  * Macro \ref ALIB_LANG_ENUM_NAMED is used to specialize this TMP struct for scoped enum class
93  * \c Fruits.
94  * The macro has to be placed <b>in the header file in the default namespace</b>.
95  * Then, in a non-header code unit, the meta data has to be initialized.
96  * Using macro \b %ALIB_LANG_ENUM_NAMED, the tuples that represent the meta data are fixed to type:
97  *
98  * std::tuple<int, aworx::String>
99  * The first element of the meta data tuple always stores the enumeration element value
100  * using the integer type which the enum class uses.
101  * Hence, the definition of the data \alib{lang,EnumMetaData::Table,Table} uses corresponding
102  * initializers:
103  *
104  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_NORMAL_NAME_TABLE_DEFINITION
105  *
106  * That is all we need to add to our software to start using the meta data! The following code
107  * gives a simple sample:
108  *
109  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_NORMAL_SAMPLE
110  *
111  * This code produces the following output:
112  * \snippet "DOX_ALIB_LANG_ENUMS_NORMAL_OUTPUT.txt" OUTPUT
113  *
114  * While custom code might also use the meta data for its own purpose, this sample uses
115  * \b %ALib built-in mechanism that become available to all enum types that specialize
116  * this TMP struct and in addition TMP struct \alib{lang,T_EnumMetaDataDeclReadWrite}.
117  * The latter is also done with macro \b ALIB_LANG_ENUM_NAMED.
118  *
119  * The output of the name is implemented by making such enum types "applicable" to objects of type
120  * \ref aworx::lib::strings::AString "AString".
121  *
122  * Documentation and details to this topic is found with
123  * - \alib{strings,T_Apply}:
124  * For general information about converting user types to \b %AStrings.
125  * - \alib{strings::applyto,T_Apply<TEnum>}:
126  * For information and code sample of applying non-bitwise enum types to \b %AStrings.
127  * - \alib{strings::applyto,T_Apply<TEnumBitwise>}:
128  * For information and code sample of applying bitwise enum types to \b %AStrings.
129  *
130  * \note
131  * If TMP struct \alib{lang,T_EnumIsBitwise} is specialized for the enum type, a different
132  * specialization of \alib{strings,T_Apply} is chosen. This version has a few special demands
133  * on the meta data given. Those are well documented and sampled
134  * with \alib{strings::applyto,T_Apply<TEnumBitwise>}.
135  * <p>
136  *
137  * \note
138  * Only for completeness, it should be mentioned that in the sample above, in addition operator
139  * \alib{strings::thirdparty::std,operator<<(std::ostream&\,const TApplicable&)}
140  * comes into play, which allows any \alib{strings,T_Apply,applicable type} to be used with
141  * \c std::cout respectively \c std::ostream.
142  *
143  *
144  * \~Comment #################################################################################### \~
145  *
146  * \anchor anchor_T_EnumMetaDataDecl_read_enum
147  * ## Parsing %Enum Elements ##
148  * The opposite of writing the name of an enum element in human readable form is parsing string
149  * data (back) to enumeration values.
150  * For this, \alib provides built-in methods
151  * - \alib{strings,Substring::ConsumeEnum},
152  * - \alib{strings,Substring::ConsumeEnumBitwise} and
153  * - \alib{strings,Substring::ConsumeEnumOrBool}.
154  *
155  * Besides what was stored as meta data in the previous section, a third tuple element
156  * of type \c int is needed: This is used to optionally limit the necessary number of characters to
157  * match an element name. If the table value is \c 0 or negative, all characters need to match.
158  * Otherwise, abbreviations may be used, which is often helpful when writing configuration files
159  * manually.
160  *
161  * To define the specialization of this TMP struct for enum types that support reading values,
162  * instead of \ref ALIB_LANG_ENUM_NAMED, macro \ref ALIB_LANG_ENUM_PARSABLE can be used.
163  * \note
164  * Like macro \ref ALIB_LANG_ENUM_NAMED,this macro also specializes
165  * \alib{lang,T_EnumMetaDataDeclReadWrite} in one step.
166  *
167  * As a sample we take the actual source code of built-in \alib enum \alib{lang,Case}:
168  *
169  * \snippet "alib/lang/enums.hpp" DOX_ALIB_LANG_ENUMS_PARSABLE_1
170  *
171  * Because \alib reads all enum meta data from resource files, an additional macro,
172  * \ref ALIB_LANG_RESOURCED is used:
173  *
174  * \snippet "alib/lang/enummetadata.hpp" DOX_ALIB_LANG_ENUMS_PARSABLE_2
175  *
176  * This macro receives the \alib{lang,Library} singleton to use for loading the resources along
177  * with the resource identifier.
178  *
179  * The corresponding default resource data definition looks like this:
180  *
181  * \snippet "alib/lang/langlib.cpp" DOX_ALIB_LANG_ENUMS_PARSABLE_3
182  *
183  * Here, the minimum length to identify the elements when parsing is \c 1 for both
184  * elements. The reason is that they have different starting characters, so one character is enough
185  * to identify each.
186  *
187  * A quick sample for parsing enum class \c %Case:
188  *
189  * \snippet "DOX_ALIB_LANG.cpp" DOX_ALIB_LANG_ENUMS_PARSABLE_SAMPLE
190  *
191  * produces the following output:
192  * \snippet "DOX_ALIB_LANG_ENUMS_PARSABLE_OUTPUT.txt" OUTPUT
193  *
194  * \~Comment #################################################################################### \~
195  * ### Sort Order of Table Entries ###
196  * In more complicated cases, it might be needed change the sort order of the table: longer
197  * names which include shorter ones have to be earlier in the table. The default resource
198  * for meta data of \alib enum type \alib{lang,ContainerOp} looks like this:
199  *
200  * \snippet "alib/lang/langlib.cpp" DOX_ALIB_LANG_ENUMS_PARSABLE_4
201  *
202  * If in this definition, "Get" was placed before "GetCreate", the latter could never be identified,
203  * because even if "Get" had a minimum length of \c 3, it would always be positively detected
204  * on a string like "GetCr".
205  *
206  * \~Comment #################################################################################### \~
207  * \anchor anchor_T_EnumMetaDataDecl_multiple_entries
208  * ### Enable Multiple Keywords For Parsing ###
209  * To allow multiple optional keywords for parsing enum elements, such keywords simply are
210  * added as new entries to the name translation table.
211  * In other words, it is allowed to have more than one table entry representing the same enum
212  * element. This is again true for both, normal enum types and those that are defined
213  * \alib{lang,T_EnumIsBitwise,bitwise}.
214  *
215  * This feature is extensively used with built-in enum \alib{lang,Bool} as seen in the \alib source
216  * code:
217  *
218  * \snippet "alib/lang/langlib.cpp" DOX_ALIB_LANG_ENUMS_PARSABLE_MULTIPLE
219  *
220  * \note
221  * Only the first entry for a dedicated enum element that is found in the table is stored in
222  * hash map \alib{lang,EnumMetaData::Map}. Furthermore, with output methods
223  * \alib{strings::applyto,T_Apply<TEnum>} and
224  * \alib{strings::applyto,T_Apply<TEnumBitwise>}, it is always the first entry found in a table
225  * that is used for writing the human readable name.
226  *
227  *
228  * \~Comment #################################################################################### \~
229  * ## Custom Meta data ##
230  * The last documentation sections introduced simple but commonly used and therefore predefined
231  * meta data types.
232  * Now, a sample that uses of a fully customized meta data tuple is discussed.
233  * The code above used macro \ref ALIB_LANG_ENUM_NAMED, which evaluates to
234  *
235  * ALIB_LANG_ENUM_META_DATA( TEnum, aworx::lib::strings::String )
236  * ALIB_LANG_ENUM_READ_WRITE( TEnum, 1, 0 )
237  * respectively \ref ALIB_LANG_ENUM_PARSABLE, which evaluates to
238  *

298  * ## Using Class %Resources For Loading %Enum Meta Data ##
299  * As it was already shown in some samples above, \alib{lang,Resources,ALib Resource Data} might
300  * be used to "externalize" enum meta data.
301  *
302  * This is supported by specializing TMP struct \alib{lang,T_Resourced} in addition
303  * to specializing TMP struct \alib{lang,T_EnumMetaDataDecl}. Such specialization is easily
304  * done with macros:
305  * - \ref ALIB_LANG_RESOURCED and
306  * - \ref ALIB_LANG_RESOURCED
307  *
308  * For standard meta data types ("named" and "parsable") this is all that is needed to
309  * start loading the meta data from the resources. However, with custom tuple types,
310  * method \alib{lang,EnumMetaData::CheckLoad} has to specialized to perform the load.
311  * With macro \ref ALIB_LANG_ENUM_META_DATA_RESOURCE_LOADER such specialization is implemented
312  * in a way that it should satisfy the most common needs: Static helper method
313  * \alib{lang,ResourcedTupleLoader::LoadTable} is used to load the data.
314  * As documented, this method (optionally) allows to have all data stored in one resource string
315  * or to separate the data of each enum element in an own numbered resource string.
316  *
317  * Using the macros above, the following things have to be kept in mind:
318  * - The macros refer to class \alib{lang,Library} which internally holds and instances of
319  * \alib{lang,Resources} and probably shares this with other libraries, like \alox.
320  * - Externalization allows translation of strings to different languages, as well as to
321  * change identifiers that a library uses by default. This way, a user of \alib might
322  * translate and change all of the built-in identifier names, configuration variables,
323  * exception messages and so forth.
324  * - The meta data is not loaded automatically from the resources, instead this is done only
325  * with an explicit invocation of
326  * \alib{lang,EnumMetaData::CheckLoad,EnumMetaData<TEnum>::GetSingleton()->CheckLoad}.<br>
327  * Therefore, prior to accessing the data, this method has to be invoked "manually".
328  * - All write and parse methods built-in to \alib, invoke this method upfront.
329  *
330  *
331  * \~Comment #################################################################################### \~
332  * # Reference Documentation #
333  *
334  * @tparam TEnum The enumeration type this struct applies to.
335  */

350 /** ************************************************************************************************
351  * This template struct implements what we call \alib{lang,T_EnumMetaDataDecl,ALib Enum Meta Data}.
352  * While the TMP struct that needs to be specialized to enable enum meta data is
353  * \alib{lang,T_EnumMetaDataDecl}, this template class is the interface into it: As soon as
354  * \b %T_EnumMetaDataDecl is specialized for type \p TEnum,
355  * (preferably with one of the macros found in \ref GrpALibMacros "Macros Supporting ALib Enum Types"),
356  * this template class is ready to be used with the type.
357  *
358  * The class leverages \alib{lang,Singleton,ALib Templated Singletons} to provide the static
359  * meta data table, a \c std::vector of
360  * \alib{lang,T_EnumMetaDataDecl::TTuple,T_EnumMetaDataDecl<TEnum>::TTuple}.
361  * Besides this, a \c std::map that hashes the enumeration
362  * element values to the table entries is provided.<br>
363  *
364  * \note
365  * This class is not implemented to be a \alib{lang,Singleton,strict singleton}. Therefore,
366  * an application might create different meta data tables for the same \p TEnum type.
367  * However, each of these tables will have the same data types attached, as those are defined
368  * with specialization \alib{lang,T_EnumMetaDataDecl}, unless different compilation units
369  * use different specializations. In this case, the singletons must not be accessed using
370  * \alib{lang,Singleton::GetSingleton,GetSingleton} as those are true singletons across
371  * processes.
372  *
373  * Finally, the class implements various interface methods to easily access the meta data of
374  * enumeration elements.
375  *
376  * \see For a tutorial and source code sample see documentation of TMP struct
377  * \alib{lang,T_EnumMetaDataDecl}.
378  *
379  * @tparam TEnum The enumeration type.
380  * @tparam TEnableIf To be ignored. Used to select this struct only for enum types which have
381  * a specialization of \alib{lang,T_EnumMetaDataDecl} in place.
382  **************************************************************************************************/
383 template<typename TEnum,
384  typename TEnableIf = typename std::enable_if<T_EnumMetaDataDecl<TEnum>::value>::type>
385 
386 struct EnumMetaData : public lang::Singleton<EnumMetaData<TEnum>>
387 {
388 
389  /**
390  * The type of the meta data table.
391  */
392  using TTable= std::vector<typename T_EnumMetaDataDecl<TEnum>::TTuple>;
393 
394  /**
395  * The meta info data table.
396  */
397  TTable Table;
398 
399  /**
400  * The type of the hash #Map.
401  */
402  using TMap= std::map<typename std::underlying_type<TEnum>::type,
403  typename T_EnumMetaDataDecl<TEnum>::TTuple&>;
404 
405  /**
406  * A container maps the underlying value of elements of enum class \p TEnum to meta data
407  * records of type \alib{lang,T_EnumMetaDataDecl::TTuple}.
408  * \note After changes in \alib{lang,EnumMetaData::Table,Table}, the map has to be rebuilt
409  * using \alib{lang,EnumMetaData::CheckLoad,CheckLoad}.
410  */
411  TMap Map;
412 
413 
414  /**
415  * Clears field #Map and re-inserts all elements of #Table unless both containers are
416  * equally sized and \p force evaluates to \c false.
417  *
418  * If a table contains more than one entry for the same enum element, only the
419  * first is inserted into the map.
420  * (See \ref anchor_T_EnumMetaDataDecl_multiple_entries "Enable Multiple Keywords For Parsing"
421  * for more information on duplicate table rows.)
422  *
423  * This method has to be invoked after changes of #Table.
424  *
425  * @param force If \c true, the map is rebuilt, even if it has the same size as the table.
426  * Defaults to \c false.
427  */
428  void CheckMap( bool force= false )
429  {
430  if( force || (Map.size() != Table.size() ) )
431  {
432  Map.clear();
433  for( auto& e : Table )
434  Map.emplace( std::get<0>(e), e) ;
435  }
436  }
437 
438  /**
439  * This default implementation just invokes #CheckMap unless:
440  * - \alib{lang,T_Resourced} is specialized for \p TEnum and
441  * - the meta data tuple type is either the same as defined with macro \ref ALIB_LANG_ENUM_NAMED
442  * or macro \ref ALIB_LANG_ENUM_PARSABLE.
443  *
444  * If this applies, the data is loaded from the resources.
445  *
446  * Specialized versions might load other data types from resources or otherwise fill
447  * the #Table.
448  *
449  * If the data is provided "in code", this method does not need a specialization.
450  * Instead #CheckMap has to be invoked after the insertions if it is not done by the methods
451  * that use the data.
452  */
453  void CheckLoad()
454  {
455  ALIB_WARNINGS_ALLOW_TEMPLATE_META_PROGRAMMING
456  if( Table.size() == 0 )
457  {
458  if( T_Resourced<TEnum>::value
459  && ( std::is_same<typename T_EnumMetaDataDecl<TEnum>::TTuple,
460  std::tuple<typename std::underlying_type<TEnum>::type,String>>::value
461  || std::is_same<typename T_EnumMetaDataDecl<TEnum>::TTuple,
462  std::tuple<typename std::underlying_type<TEnum>::type,String,int>>::value )
463  )
464  detail::loadStandardEnumTable( typeid( typename T_EnumMetaDataDecl<TEnum>::TTuple ),
465  &Table,
466  T_Resourced<TEnum>::Lib(),
467  T_Resourced<TEnum>::Name() );
468  }
469 
470  CheckMap();
471  ALIB_WARNINGS_RESTORE
472  }
473 
474  /**
475  * Searches and returns the meta data tuple for enum element \p element.
476  * The elements of the returned tuple can be accessed using <c>std::get<idx>()</c>.
477  * To receive the "original" enum element value, use static method
478  * \ref Enum(const typename T_EnumMetaDataDecl<TEnum>::TTuple&) "Enum( TTuple& )".
479  *
480  * In case \alib{lang,T_EnumMetaDataDeclReadWrite} is specialized for type \p TEnum,
481  * the returned tuple may be used with static method
482  * \alib{lang,EnumReadWriteInfo::Name(const typename T_EnumMetaDataDecl<TEnum>::TTuple&),EnumReadWriteInfo::Name( TTuple& )}
483  *
484  * If no entry was found, \c nullptr is returned.
485  *
486  * In debug compilations, an \alib assertion is raised, in case the map is empty.
487  * @param element The enum element to search for.
488  * @return The tuple that is associated with \p element.
489  */
490  typename T_EnumMetaDataDecl<TEnum>::TTuple* Get ( TEnum element )
491  {
492  #if ALIB_MODULE_ALL //Report
493  ALIB_ASSERT_ERROR( Map.size() > 0,
494  "No meta data for enum {}({}) available or "
495  "T_EnumInfo<TEnum>::CheckMap() not invoked.",
496  typeid(TEnum),
497  static_cast<typename std::underlying_type<TEnum>::type>( element ) )
498  #else
499  ALIB_ASSERT_ERROR( Map.size() > 0,
500  "No meta data for enum available or "
501  "T_EnumInfo<TEnum>::CheckMap() not invoked. Value=",
502  static_cast<typename std::underlying_type<TEnum>::type>( element ) )
503  #endif
504  auto it= Map.find( EnumValue(element) );
505  return it != Map.end() ? &it->second : nullptr;
506  }

509  /**
510  * Returns the enum element of \p tuple.
511  *
512  * @param tuple An entry of the #Table.
513  * @return The enum element that is represented by \p tuple.
514  */
515  static inline

1273  * \snippet "DOX_ALIB_LANG_ENUMS_BITWISE_OUTPUT.txt" OUTPUT
