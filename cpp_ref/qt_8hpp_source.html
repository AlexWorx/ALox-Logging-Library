44 /**
45  * With the inclusion of header file <c>alib/compatibility/qt.hpp</c>
46  * some compatibility between <b>ALib Strings</b> and types of
47  * [QT Class Library](https://www.qt.io) is given.
48  *
49  *
50  * <b>1. Specializations of %T_String:</b><br>
51  * Specializations of \ref aworx::lib::strings::T_String is given for types
52  * - \b QByteArray and
53  * - \b QLatin1String
54  * This allows to implicitly constructs objects of class \b %String from the named types.
55  *
56  * <b>2. Applicable Types:</b><br>
57  * Support for applying (appending contents of objects of the types to objects of class \b %AString)
58  * is given. Applicable types are:
59  * - QChar
60  * - QLatin1String
61  * - QByteArray
62  * - QVector<uint>
63  * - QString
64  *
65  * <b>3. Conversion to QString:</b><br>
66  * Several overloaded methods #ToQString are defined to convert <b>ALib Strings</b> to
67  * instances of class \b QString.
68  *
69  * <b>General Information On ALib And QT %String Compatibility:</b><br>
70  * ALib string classes are designed to support 3rd-party string types in a non-intrusive
71  * fashion. Template meta programming in two areas makes this possible:
72  * - Implicit conversion of external strings to class
73  * \ref aworx::lib::strings::String "aworx::String", which allows to pass 3rd party strings to
74  * method accepting parameters of type <b>const %aworx::String&</b> without conversion.
75  * - \e Application of arbitrary types to objects of class
76  * \ref aworx::lib::strings::AString "aworx::AString". Here, the term \e "application" most
77  * means concatenating a string representation of an object to the \b AString object.
78  * (This is all documented in reference manual of namespace \ref aworx::lib::strings.)
79  *
80  * %String classes of the [QT library](https://www.qt.io) are
81  * supported trough header file <em>"alib/compatibility/qt.hpp"</em>.
82  *
83  * <b>Implicit Conversion:</b><br>
84  * Unfortunately, objects of class \b %QString can not be implicitly converted (embedded) in
85  * thin and lightweight class \ref aworx::lib::strings::String "aworx::String". The reason
86  * for this is, that ALib strings are single byte strings, while \b QString represent
87  * (two byte) unicode strings. Therefore, template struct
88  * \ref aworx::lib::strings::T_String "T_String" can not be implemented for class \b QString
89  * and objects of this class can not be simply passed to methods accepting parameters of type
90  * <b>const %aworx::String&</b>.
91  *
92  * The way out is the use of classes \b QByteArray and \b QLatin1String. For these QT classes,
93  * an implicit conversion exists and this way objects of these types are implicitly converted
94  * to \b %aworx::String types. Class \b %QString provides methods to convert to these classes.
95  * Passing a \b QString object to a method expecting a <b>const %aworx::String&</b> can be
96  * done as follows
97  *
98  * MyMethodExpectingALibString( myQString.toLocal8Bit() );
99  * MyMethodExpectingALibString( myQString.toUtf8() );
100  * MyMethodExpectingALibString( myQString.toLatin1() );
101  *
102  * \note
103  * The encoding of strings represented by ALib strings is user defined. Therefore, it
104  * is up to the user to decide which of the above conversion methods of \b QString
105  * is to be used. In most cases, method \b toLocal8Bit should be the right choice. Also,
106  * ALib strings uses the current locale to convert characters of type wchar_t (or arrays of them)
107  * when \e applying to class \b %AString as well.
108  *
109  * The obvious drawback of this approach is that a conversion has to take place and
110  * that this conversion implies a temporary allocation of heap memory for the data stored
111  * in the (stack-allocated) \b QByteArray object.
112  *
113  * \attention
114  * This is a non-existing namespace, exclusively defined for the
115  * [documentation parser](http://www.stack.nl/~dimitri/doxygen).
116  * All types and functions described in this namespace are residing in namespace
117  * #aworx::lib::strings (just as they have to in respect to C++ rules).
118  * This documentation is moved here to keep the original namespace documentation clean.
119  */

248  /** Specialization of template struct \ref T_Apply for type \c QString. */
249  template<> struct T_Apply<QString> : public std::true_type
250  {
251  /** ****************************************************************************************
252  * Appends \p src of type \b QString to \p target.
253  *
254  * @param target The AString to append \p src to.
255  * @param src The source string.
256  * @return The length of the given string \p src which was appended to \p target.
257  ******************************************************************************************/
258  static inline integer Apply( AString& target, const QString& src )
259  {
260  int len= src.length();
261  wchar_t localBuffer[1024];
262  wchar_t* wcBuffer= localBuffer;
263  if ( len > 1024 )
264  wcBuffer= new wchar_t[len];
265 
266  src.toWCharArray( wcBuffer );
267  target.Append( wcBuffer, len );
268 
269  if ( wcBuffer != localBuffer )
270  delete[] wcBuffer;
271 
272  return 0; // avoid null check (already done in Append())
273  }
274  };

279  /** ********************************************************************************************
280  * Copies or appends a region of a given \ref aworx::lib::strings::String "String" into a QString.
281  * The region is adjusted to the length of the %String.<br>
282  * See \ref aworx::lib::strings::thirdparty::qt for general information about using QT
283  * strings with ALib.
284  *
285  * @param as The \ref aworx::lib::strings::String "String" to convert into a QString.
286  * @param target A result QString to copy the specified region into.
287  * @param regionStart The start index of the region to be copied.
288  * @param regionLength The maximum length of the region to be copied.
289  * Defaults to CString::MaxLen.
290  * @param targetData If \c CurrentData::Keep, the target \c QString is not cleared before
291  * the result is written. Defaults to \c CurrentData::Clear.
292  *
293  * @return The (modified) QString& result that was provided (for concatenation of calls).
294  **********************************************************************************************/
295  inline
296  QString& ToQString( const String& as,
297  QString& target,
298  integer regionStart,
299  integer regionLength = CString::MaxLen,
300  lang::CurrentData targetData = lang::CurrentData::Clear )
301  {
302  if ( targetData == lang::CurrentData::Clear )
303  target.clear();
304 
305  // adjust range, if empty return empty string
306  if ( as.AdjustRegion( regionStart, regionLength ) )
307  return target;
308 
309  // copy our buffer into target
310  target.fromUtf8( as.Buffer() + regionStart, static_cast<int>( regionLength - regionStart ) );
311  return target;
312  }
313 
314  /** ********************************************************************************************
315  * Copies a region of a given \ref aworx::lib::strings::String "String" into a QString which
316  * is created.
317  * The region is adjusted to the length of the %String.<br>
318  * See \ref aworx::lib::strings::thirdparty::qt for general information about using QT
319  * strings with ALib.
320  *
321  * @param as The \ref aworx::lib::strings::String "String" to convert into a QString.
322  * @param regionStart The start index of the region in this to create the QString from.
323  * @param regionLength The maximum length of the region to be copied.
324  * Defaults to CString::MaxLen.
325  *
326  * @return A string that represents the specified sub region of this object.
327  **********************************************************************************************/

400 /**
401  * With the inclusion of header file <c>alib/compatibility/qt.hpp</c>, support for
402  * boxing string types of the [QT Class Library](https://www.qt.io) is provided.
403  * QT provides several string types, the most prominent of-course is \c QString. Unfortunately,
404  * this type uses a 2-byte character type. This is neither compatible with C++ fundamental
405  * type \c char nor with \c wchar_t, as the latter has platform dependent size, e.g. is 4 bytes
406  * wide on GNU/Linux using GNU C++ compiler. \b %QString is boxed to type \c char16_t[].
407  *
408  * Besides this, supported types are \b QByteArray and \b QLatin1String.
409  * Those types are boxed to fundamental type \c char[] (the same as for example
410  * \b %aworx::String and \b std::string are), and therefor can be used completely transparently
411  * with any function or method that accepts boxes and processes single byte string values.
412  *
413  * All types are unboxable as values.
414  *
415  * For more information see chapter \ref alib_namespace_boxing_strings "Boxing Strings" of the
416  * namespace documentation of module \ref aworx::lib::boxing "ALib Boxing".
417  *
418  * \note
419  * Namespace function \ref InitQTString "aworx::lib::boxing::InitQTString" is to be invoked on
420  * bootstrap to complete the support of QT-string-type boxing.
421  *
422  *
423  * \attention
424  * This is a non-existing namespace, exclusively defined for the
425  * [documentation parser](http://www.stack.nl/~dimitri/doxygen).
426  * This documentation is moved here to keep the original namespace documentation clean.
427  */
