694  void GrowBufferAtLeastBy( int minimumGrowth )
695  {
696  int capacity= Capacity();
697 
698  // big enough?
699  ALIB_ASSERT_WARNING ( capacity < length + minimumGrowth,
700  "Unnecessary invocation of Grow()" );
701

703  if (capacity == 0 )

710  int newCapacity= capacity + (capacity / 2);
