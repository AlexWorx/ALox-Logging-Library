23 /** ************************************************************************************************
24  * This is the central class of <b>%ALib %Boxing</b>. By using template meta programming, an object
25  * of this class can be created by passing just any C++ type to the constructor. The passed
26  * value will be "boxed" within the instance of this class.
27  *
28  * Then, the instances of this class support type checking, value extraction ("unboxing") and the
29  * invocation of "virtual methods". All features are customizable in detail per "boxable type".
30  *
31  * A thorough introduction to and documentation of all aspects of <b>%ALib %Boxing</b> is given
32  * with namespace documentation \ref aworx::lib::boxing.
33  **************************************************************************************************/

112  /** ****************************************************************************************
113  * Constructor using template meta programming to fetch any type of \b %ALib Strings,
114  * unknown pointer types and nullptr.
115  *
116  * @tparam TBoxable Any C++ type to be boxed.
117  * @param src The src of (template) type \c T.
118  *****************************************************************************************/
119  template <typename TBoxable,
120  typename TEnableIf= typename std::enable_if< !std::is_enum<TBoxable>::value >::type >
121  inline
122  Box(const TBoxable& src )
123  {
124  // untyped nullptr
125  if ( std::is_same<TBoxable, decltype(nullptr)>::value )
126  {
127  boxer= BoxerT<decltype(nullptr)>::GetSingleton();
128  data.Value= (boxvalue) 0;
129  data.Length= 0;
130  return;
131  }
132 
133 
134  using TPlain= typename std::remove_const <
135  typename std::remove_pointer <
136  typename std::remove_reference<
137  typename std::remove_const < TBoxable >::type>::type>::type>::type;
138 
139  //ALIB_TMP_SHOW_TYPE_IN_DEBUGGER(TPlain)
140 
141  // array types
142  if( std::is_array<TBoxable>::value )
143  {
144  using TAElem= typename std::remove_extent<TPlain>::type;
145  //ALIB_TMP_SHOW_TYPE_IN_DEBUGGER(TAElem)
146 
147  boxer= ArrayBoxerT<TAElem>::GetSingleton();
148  data.Value= (boxvalue) &src;
149  data.Length= T_SetArraySize<TAElem>( (integer) ( std::extent<TPlain>::value ) );
150  return;
151  }
152 
153  // boxing a box?
154  if( std::is_same<Box, TPlain>::value )
155  {
156  if( std::is_pointer<TBoxable>::value )
157  {
158  ALIB_ASSERT_ERROR( * (TPlain **) &src != nullptr, "Can't box type Box* of value nullptr." );
159  boxer= (* (Box **) &src) ->boxer;
160  data.Value= (* (Box **) &src) ->data.Value;
161  data.Length= (* (Box **) &src) ->data.Length;
162  }
163  else
164  {
165  ALIB_ERROR( "This can't happen" );
166  }
167  return;
168  }
169 
170  // if a pointer was given, try to find boxer for pointer type first
171  if( std::is_pointer<TBoxable>::value )
172  {
173  if( T_Boxing<TPlain*>::value )
174  {
175  boxer= T_Boxing<TPlain*>::IsArray()
176  ? (Boxer*) ArrayBoxerT<typename T_Boxing<TPlain*>::Type >::GetSingleton()
177  : (Boxer*) BoxerT<typename T_Boxing<TPlain*>::Type >::GetSingleton();
178 
179  if( * (TPlain **) &src )
180  {
181  if(T_Boxing<TPlain*>::HasCustomBoxing() )
182  T_Boxing<TPlain*>::Boxing( *this, * (TPlain **) &src );
183  else
184  {
185  if(std::is_pointer<typename T_Boxing<TPlain*>::Type>::value )
186  DefaultBoxing( *this, * ((TPlain **) &src) );
187  else
188  DefaultBoxing( *this, * ((TPlain *&) src) );
189  }
190  }
191  else
192  {
193  data.Value= 0;
194  data.Length= 0;
195  }
196  return;
197  }
198 
199  if( T_Boxing<TPlain>::value )
200  {
201  boxer= T_Boxing<TPlain>::IsArray()
202  ? (Boxer*) ArrayBoxerT<typename T_Boxing<TPlain>::Type >::GetSingleton()
203  : (Boxer*) BoxerT<typename T_Boxing<TPlain>::Type >::GetSingleton();
204 
205  if( * (TPlain **) &src )
206  {
207  if(T_Boxing<TPlain>::HasCustomBoxing() )
208  T_Boxing<TPlain>::Boxing( *this, * ((TPlain *&) src) );
209  else
210  {
211  if(std::is_pointer<typename T_Boxing<TPlain>::Type>::value )
212  DefaultBoxing( *this, * ((TPlain **) &src) );
213  else
214  DefaultBoxing( *this, * ((TPlain *&) src) );
215  }
216  }
217  else
218  {
219  data.Value= 0;
220  data.Length= 0;
221  }
222 
223  return;
224  }
225  }
226 
227  // if a non-pointer was given, try to find boxer for non-pointer type type first
228  else
229  {
230  if( T_Boxing<TPlain>::value )
231  {
232  boxer= T_Boxing<TPlain>::IsArray()
233  ? (Boxer*) ArrayBoxerT<typename T_Boxing<TPlain>::Type >::GetSingleton()
234  : (Boxer*) BoxerT<typename T_Boxing<TPlain>::Type >::GetSingleton();
235 
236 
237  if(T_Boxing<TPlain>::HasCustomBoxing() )
238  T_Boxing<TPlain>::Boxing( *this, (TPlain&) *(int*) &src );
239  else
240  {
241  if(std::is_pointer<typename T_Boxing<TPlain>::Type>::value )
242  DefaultBoxing( *this, (typename std::remove_pointer<typename T_Boxing<TPlain>::Type>::type * ) &src );
243  else
244  DefaultBoxing( *this, (typename T_Boxing<TPlain>::Type&) *(int*) &src );
245  }
246 
247  return;
248  }
249 
250  if( T_Boxing<TPlain*>::value )
251  {
252  boxer= T_Boxing<TPlain*>::IsArray()
253  ? (Boxer*) ArrayBoxerT<typename T_Boxing<TPlain*>::Type >::GetSingleton()
254  : (Boxer*) BoxerT<typename T_Boxing<TPlain*>::Type >::GetSingleton();
255  if(T_Boxing<TPlain*>::HasCustomBoxing() )
256  T_Boxing<TPlain*>::Boxing( *this, (TPlain * ) &src );
257  else
258  {
259  if(std::is_pointer<typename T_Boxing<TPlain*>::Type>::value )
260  DefaultBoxing( *this, (typename std::remove_pointer<typename T_Boxing<TPlain>::Type>::type * ) &src );
261  else
262  DefaultBoxing( *this, (typename T_Boxing<TPlain>::Type&) *(int*) &src );
263  }
264  return;
265  }
266  }
267 
268  // Nothing was found: create a generic boxer for the given type (as pointer)
269  boxer= BoxerT<TPlain*>::GetSingleton();
270  if( std::is_pointer<TBoxable>::value )
271  {
272  if( * (TPlain **) &src )
273  DefaultBoxing<TPlain*>( *this, * (TPlain **) &src );
274  else
275  {
276  data.Value= 0;
277  data.Length= 0;
278  }
279  }
280  else
281  DefaultBoxing<TPlain*>( *this, (TPlain * ) &src );
282  }

410  /** ****************************************************************************************
411  * Returns the contents of this box converted to type \p TBoxable.
412  * By default this is done by invoking template method
413  * \ref aworx::lib::boxing::DefaultUnboxing "DefaultUnboxing" and might be customized
414  * by specializing struct
415  * \ref aworx::lib::boxing::T_Boxing "T_Boxing".
416  *
417  * In debug compilations, the actual type of this object is asserted to equal the templated
418  * return type.
419  *
420  * For technical reasons, type \c void* can not be unboxed using this method.
421  * Use #UnboxRaw() instead.
422  *
423  * @tparam TBoxable The supported type
424  * @return The type ID (BoID) of this object
425  ******************************************************************************************/
426  template <typename TBoxable> inline
427  TBoxable Unbox() const
428  {
429  static_assert( ( T_Boxing<TBoxable>::value && T_Boxing<TBoxable>::IsUnboxable() )

431  || ( !T_Boxing<TBoxable>::value
432  && std::is_pointer<TBoxable>::value
433  && !T_Boxing<typename std::remove_pointer<TBoxable>::type >::value )
434  , "This type can not be unboxed.");
435 
436  ALIB_ASSERT_ERROR( IsType<TBoxable>(), "Box::Unbox(): Incompatible type" );
437

540  * about how to enable custom types to support arbitrary method interfaces.

557  /** ****************************************************************************************
558  * Invokes method \b %"Invoke" of attached interface type \p TInterface.
559  *
560  * If no interface is attached to this classes #boxer, then a default value of \p TReturn
561  * returned. Hence, type \p TReturn needs to be default constructable.
562  * In debug compilations, \ref ALIB_ERROR is used to handle non-existing interface errors.
563  *
564  * In the case that a default instance of type \p TInterface is not set (see
565  * \ref aworx::lib::boxing::Boxer::DefineDefaultInterface "Boxer::DefineDefaultInterface"),
566  * method #HasInterface should be used to determine the existence of a specialized interface
567  * for the boxed type.
568  *
569  * If return type of method \b %"Invoke" of interface class \p TInterface is other than
570  * \c void, then this return type needs to be explicitly specified as second template
571  * parameter.
572  *
573  * @tparam TInterface The interface type to receive.
574  * @tparam TReturn The return type of the invocation. Defaults to \c void.
575  * @tparam Args The types of the variadic arguments.
576  * @param args The variadic arguments to interface methods.
577  * @return Nothing in case that \p TReturn is void, otherwise the result of the invocation,
578  * respectively <c>TReturn()</c> if requested interface type was not found for this
579  * \b %Box.
580  ******************************************************************************************/
581  template <typename TInterface, typename TReturn= void, typename... Args>
582  TReturn Invoke(Args&&... args) const
583  {
584  TInterface* interface= GetInterface<TInterface>();
585  if( interface != nullptr )
586  return interface->Invoke( *this, std::forward<Args>(args)... );
587 
588 
589  // not found: In debug mode, check if this interface exists anywhere
590  #if ALIB_DEBUG
591  if ( BOXING.dbgKnownInterfaces.find( typeid(TInterface) ) == BOXING.dbgKnownInterfaces.end() )
592  {
593  ALIB_ERROR( "Unknown interface type: ", debug::TypeDemangler(typeid(TInterface)).Get(),
594  ". Note: Invoked on box of type: ", debug::TypeDemangler(boxer->type ).Get()
595  );
596  }
597  else
598  {
599  ALIB_ERROR( "Interface type: ", debug::TypeDemangler(typeid(TInterface)).Get(),
600  " not defined for box of type: ", debug::TypeDemangler(boxer->type ).Get()
601  );
602  }
603  #endif
604 
605  return TReturn();
606  }

608  /** ****************************************************************************************
609  * Searches and receives the singleton instance of type \p TInterface (which needs to be of
610  * a type derived from class \ref aworx::lib::boxing::Interface "Interface")
611  * defined for boxed types represented by this object.
612  *
613  * \note
614  * Instead of using this method, it is advised to use method #Invoke instead. The use
615  * of this method and subsequent "manual" invocation of the method named \b "Invoke" on
616  * of the returned interface instance is useful, when multiple invocations of the
617  * interface method are to be done on the same box, like in the following sample:
618  *
619  * MyInterface* mIF= aBox.GetInterface<MyInterface>();
620  * if( mIF != nullptr )
621  * for( int i= 0; i< 100; ++i )
622  * mIF->Invoke( aBox, i );
623  *
624  *
625  * @tparam TInterface The interface type to receive.
626  * @return The interface singleton defined. \c nullptr in case that no interface is
627  * available.
628  ******************************************************************************************/
629  template <typename TInterface>
630  TInterface* GetInterface() const
631  {
632  TInterface* interface= nullptr;
633 
634  // search in interfaces
635  auto entry= boxer->interfaces.find( typeid(TInterface) );
636  if ( entry != boxer->interfaces.end() )
637  interface= dynamic_cast<TInterface*>( entry->second );
638 
639  // check for default interface (all types)
