34 /** ************************************************************************************************
35  * This templated class is a specialization of class #TString which has a fixed length.
36  * Functions that accept such objects as a parameter can rely
37  * at compile time on the constant length of the string and optimize their performance
38  * accordingly. <br>
39  * For example, template method
40  * \ref aworx::lib::strings::AString::operator<<(const T& op) "AString::operator<<(const T&)"
41  * which internally makes use of template method
42  * \ref aworx::lib::strings::AString::Apply "AString::Apply" will append objects of this type
43  * without doing a copy loop (up to a certain amount of characters).
44  *
45  * The class is useful to assign string literals to objects, including their lengths and pass
46  * them around, hence the name.<br>
47  *
48  * It might also used to provide strings with variable contents. For this, the character
49  * array given in the constructor needs to be managed and manipulated outside of this class.
50  *
51  * \note
52  * This class \b %StringLiteral, when used in source code as well as in documentations, is
53  * often referred to using the synonym '<b>%SLiteral</b>'. For more information about
54  * the synonymous names of \b %ALib string classes, refer to
55  * \ref alib_namespace_strings_class_overview "String Classes Overview" and
56  * \ref CPP_AWORX_NS_SHORTCUTS "Type Shortcuts of Namespace aworx".
57  *
58  * @tparam TLength The length of the represented string.
59 ***************************************************************************************************/
60 template<size_t TLength>
61 class StringLiteral : public TString
62 {
63  public:
64  /** ********************************************************************************************
65  * The only available constructor taking a string literal of exactly the size we have plus one
66  * for the termination character.
67  * @param src The string literal to represent.
68  **********************************************************************************************/
69  constexpr StringLiteral( const char (&src)[TLength + 1] )
70  : TString( src, TLength )
71  {}

73  /** ********************************************************************************************
74  * Copies the contents of this string (excluding the terminating zero) to the given
75  * destination using static method #Copy(char*, const char*).
76  *
77  * @param dest The destination buffer.
78  **********************************************************************************************/
79  inline
80  void Copy( char* dest) { Copy( dest, buffer ); }

82  /** ********************************************************************************************
83  * Static method that copies TLength characters from the given the source array to the
84  * given destination array.
85  * If template parameter \p TLength equals or is greater than zero, memcpy() is used. If it is
86  * below, the right amount of single character copy operations (no loops, no ifs) is inserted
87  * into the code.
88  *
89  * @param src The source buffer to copy from.
90  * @param dest The destination buffer to copy to..
91  **********************************************************************************************/
92  static
93  inline
94  void Copy( char* dest, const char* src)
95  {
96  if (TLength >= 10 )
97  {
98  memcpy( dest, src, TLength );
99  return;
100  }
101 
102  if (TLength == 9 ) goto copy9;
103  if (TLength == 8 ) goto copy8;
104  if (TLength == 7 ) goto copy7;
105  if (TLength == 6 ) goto copy6;
106  if (TLength == 5 ) goto copy5;
107  if (TLength == 4 ) goto copy4;
108  if (TLength == 3 ) goto copy3;
109  if (TLength == 2 ) goto copy2;
110  if (TLength == 1 ) goto copy1;
111  return;

113  copy9: *dest++= *src++;
114  copy8: *dest++= *src++;
115  copy7: *dest++= *src++;
116  copy6: *dest++= *src++;
117  copy5: *dest++= *src++;
118  copy4: *dest++= *src++;
119  copy3: *dest++= *src++;
