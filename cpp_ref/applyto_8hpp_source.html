164  template<typename T> inline int ApplyTo( AString& target, const T src )
165  {
166  #if defined(_MSC_VER)
167  #pragma warning( push )
168  #pragma warning( disable : 4127 ) //conditional expression is constant
169  #endif
170  // prevent invoking us with unknown types
171  using TPlain= typename std::remove_const <
172  typename std::remove_pointer <
173  typename std::remove_reference<
174  typename std::remove_const < T >::type>::type>::type>::type;
175 
176  static_assert(
177  std::is_pod < TPlain > ::value
178  || std::is_base_of <AS, TPlain > ::value
179  || IsApplicable <const TPlain&> ::value
180  , "ALib: T is not a known type to append to AString. Implement ApplyTo<T>() to enable conversion.");
181 
182  // fetch AStrings
183  if ( std::is_base_of<AS, typename std::remove_reference<T>::type>::value )
184  {
185 
186  if( ((AS&)src).IsNull() )
187  return -1;
188  target.Append<false>( ((AS&)src).Buffer(), ((AS&)src).Length() );
189  return ((AS&)src).Length();
190  }
191 
192  ALIB_DEBUG_CODE( target.Apply("ApplyTo<Unknown Type>") );
193  ALIB_WARNING("Unknown type for ApplyTo()");
194  return 0;
