37  * @{
38  * @name Macros Supporting TMP Class T_Apply
39  * TMP class
40  * \ref aworx::lib::strings::T_Apply "T_Apply" may be specialized for user types to allow the
41  * <em>"application"</em> of custom object of type
42  * \ref aworx::lib::strings::AString "AString".
43  * The following macros simplify such specialization.
44  *
45  * @{
46  * \def ALIB_STRINGS_SPECIALIZE_T_APPLY
47  * Helper macro for specializing struct
48  * \ref aworx::lib::strings::T_Apply "T_Apply" for type \p TYPE.
49  * This macro has to be positioned outside of any namespace and the given type has to include
50  * its full namespace qualification.
51  *
52  * This macro is to be used in combination with macro #ALIB_STRINGS_SPECIALIZE_T_APPLY_DEF.<br>
53  * As an alternative to the two macros, #ALIB_STRINGS_SPECIALIZE_T_APPLY_INLINE might be used, which will
54  * specialize \c T_Apply and define method \b %Apply inline.
55  *
56  * \see \ref alib_namespace_strings_astring_applyto "Applying Objects to AStrings"
57  *
58  * @param TYPE The type to specialize struct \b %T_Apply for.
59  *
60  * \def ALIB_STRINGS_SPECIALIZE_T_APPLY_DEF
61  * Macro used in combination with \ref ALIB_STRINGS_SPECIALIZE_T_APPLY which specializes struct
62  * \ref aworx::lib::strings::T_Apply "T_Apply" for type \p TYPE and with this declares its
63  * member function \b %Apply.<br>
64  * This macro is used for the implementation of this member function.
65  *
66  * @param TYPE The type that struct \b %T_Apply was specialized for.
67  * @param IMPL The implementation code for method \b %Apply.
68  *
69  * \def ALIB_STRINGS_SPECIALIZE_T_APPLY_INLINE
70  * Helper macro for specializing struct
71  * \ref aworx::lib::strings::T_Apply "T_Apply" for a custom type for custom type \p TYPE.
72  * This macro has to be positioned outside of any namespace and the given type has to include
73  * its full namespace qualification.
74  *
75  * With the specialization of struct \b %T_Apply, method \b %Apply will be defined and
76  * implemented inline.
77  *
78  * Macros #ALIB_STRINGS_SPECIALIZE_T_APPLY and #ALIB_STRINGS_SPECIALIZE_T_APPLY_DEF
79  * provide a non-inline alternative to this macro.
80  *
81  * \see \ref alib_namespace_strings_astring_applyto "Applying Objects to AStrings"
82  *
83  * @param TYPE The type to specialize struct \b %T_Apply for.
84  * @param IMPL The implementation code for method \b %Apply.
85  *
86  * @}
87  * @} //ingroup GrpALibMacros
88  **************************************************************************************************/

123  /** Specialization of template struct \ref T_Apply for type \c char*. */
124  template<> struct T_Apply<char*> : public std::true_type
125  {
126  /** ****************************************************************************************
127  * Writes a cstring to the given AString.
128  * @param target The object that \b Apply was invoked on.
129  * @param src The character string to append to \p target.
130  * @return The number of characters appended to target.
131  ******************************************************************************************/
132  static inline integer Apply( AString& target, const char* src )
133  {
134  integer length= static_cast<integer>( strlen( src ) ); // can't be nullptr
135  target.Append<false>( src, length ); // use false, as length check is anyhow down later
136  return length;
137  }
138  };

140  /** Specialization of template struct \ref T_Apply for type \c wchar_t*. */
141  template<> struct T_Apply<wchar_t*> : public std::true_type
142  {
143  /** ****************************************************************************************
144  * This method converts the provided wide character string (zero-terminated array of
145  * wchar_t) and appends it to the \p target. The conversion is done using
146  * - <em>wcsnrtombs</em> on glibc platforms (e.g. Linux)
147  * - <em>WideCharToMultiByte(CP_UTF8,...)</em> on Windows platform.
148  * @param target The object that \b Apply was invoked on.
149  * @param src The wide character string to append to \p target.
150  * @return The number of characters appended to target.
151  ******************************************************************************************/
152  static inline integer Apply( AString& target, const wchar_t* src)
153  {
154  target.Append( src, src != nullptr ? static_cast<integer>( wcslen( src ) ) : 0 ) ;
155  return 0; // nullptr treatment already done
156  }
157  };
