36 /** ************************************************************************************************
37  * This class is a simple wrapper around std::mutex (provided solely to reach compatibility of ALIB
38  * across programming languages) and hence allows *mutual exclusive access* to resources, by
39  * protecting data from concurrent access by different threads.
40  *
41  * Unlike class ThreadLock, this class ThreadLockNR does not use an internal counter to allow
42  * recursive calls to Acquire() and Release(). Also, no owner is recorded, no ALIB
43  * \ref aworx::lib::lang::ReportWriter "ReportWriter"
44  * is invoked on misuse and no time limit warnings are supported.
45  * Consider using ThreadLock in most cases. The advantage of ThreadLockNR is that it operates
46  * several times faster than ThreadLock. So, for very time critical sections with frequent
47  * Lock.Release()use of ThreadLockNR can be taken into consideration.
48  *
49  * \attention Nested (recursive) acquisitions are not supported and lead to undefined behavior.
50  **************************************************************************************************/
51 class ThreadLockNR : public lang::Ownable

113  * @param file Caller information. Available only in debug compilations.

145  #if ALIB_FEAT_THREADS

152  /** ****************************************************************************************
153  * If parameter is \c true, the whole locking system is disabled. The only objective here is
154  * to gain execution speed, as thread synchronization causes relatively expensive system
155  * calls. Use this method only if you are 100% sure that your (otherwise) critical section
156  * are executed in a single threaded environment. And: "relative expensive" means:
157  * they are not really expensive. This is only for the rare case that your critical
158  * section is very, very frequently executed.
159  * @param safeness Denotes the safeness mode.
160  ******************************************************************************************/
161  void SetSafeness( lang::Safeness safeness )
162  {
163  #if ALIB_FEAT_THREADS
164  if( safeness == lang::Safeness::Unsafe && mutex != nullptr )
165  {
166  delete mutex;
167  mutex= nullptr;
168  return;
169  }
170 
171  if( safeness == lang::Safeness::Safe && mutex == nullptr )
172  {
173  mutex= new std::mutex();
174  return;
175  }
176  #else
177  mutex= safeness == lang::Safeness::Safe ? reinterpret_cast<void*>(1) : nullptr;
178  #endif
179  }
180

184  ******************************************************************************************/
