18 /** ************************************************************************************************
19  * This enum is used in \b %ALox to control the 'verbosity' or 'verboseness' of the log output.
20  * The values herein - apart from special value 'Off' - are sorted in the following order
21  * - Verbose (highest level)
22  * - Info
23  * - Warning
24  * - Error (lowest level).
25  *
26  * A value of this set is provided to \b %ALox in two different ways:
27  * First, all methods of class \ref aworx::lox::Lox "Lox" that execute a log operation
28  * assign a value of this enum to the <em>Log Statement</em>. Secondly, methods
29  * \ref aworx::lox::Lox::SetVerbosity "Lox::SetVerbosity", are defining the 'accepted' \e minimal
30  * \e Verbosity for a pair of <em><Logger/%Log Domain></em>.
31  *
32  * \b %ALox, when executing a statement, checks both values against each other. A <em>Log Statement</em>
33  * is executed, when the <em><Logger/%Log Domain></em> setting is set to the same or a 'higher level'.
34  * For example if a <em><Logger/%Log Domain></em> setting is \b %Warning, then <em>Log Statements</em>
35  * with associated \e %Verbosity \b %Warning and \b %Error are executed and those with \b %Info and
36  * \b %Verbose are suppressed.
37  *
38  * If special value \b %Off is used with \ref aworx::lox::Lox::SetVerbosity "Lox::SetVerbosity",
39  * all logging is switched Off for this pair of <em><Logger/%Log Domain></em>.
40  *
41  * Some of the <em>Log Statements</em> accept the parameter directly (e.g.
42  * \ref aworx::lox::Lox::Entry "Lox::Entry",
43  * \ref aworx::lox::Lox::Once "Lox::Once" and
44  * \ref aworx::lox::Lox::If "Lox::If"), others inherently use the right value as their method
45  * name suggests (e.g.
46  * \ref aworx::lox::Lox::Error "Lox::Error",
47  * \ref aworx::lox::Lox::Warning "Lox::Warning",
48  * \ref aworx::lox::Lox::Info "Lox::Info",
49  * \ref aworx::lox::Lox::Verbose "Lox::Verbose" and
50  * \ref aworx::lox::Lox::Assert "Lox::Assert"). The latter group of methods do not support
51  * parameter \b %Off.
52  *
53  * If special value \b %Off is used with those <em>Log Statements</em>, that allow to specify the \e %Verbosity
54  * as a parameter, the <em>Log Statement</em> is never executed This is useful if the parameter is
55  * determined at runtime, depending on the state of an application.
56  **************************************************************************************************/
57 enum class Verbosity
58 {
59  /**
60  * The 'highest' level of \e %Verbosity.
61  * Statements with this value associated are logged only if a <em>%Log Domain</em> is set to \b %Verbose as well.
62  */
63  Verbose,
64 
65  /**
66  * The standard \e Verbosity for normal log output statements.
67  * Logged if a <em>%Log Domain</em> is set to \b %Info or \b %Verbose.
68  */
69  Info,
70 
71  /**
72  * A \e Verbosity for warning messages, hence things that might lead to errors or are not
73  * welcome for other reasons, but maybe are not errors.<br>
74  * Logged if a <em>%Log Domain</em> is set to \b %Warning, \b %Info or \b %Verbose.
75  */
76  Warning,
77 
78  /**
79  * A \e Verbosity for error messages.
80  * It is suppressed only if a <em>%Log Domains'</em> setting is \b %Off.
81  */
82  Error,
83 
84  /**
85  * Statements with this value associated are never logged (useful if \e %Verbosity is
86  * evaluated at runtime). <em>%Log Domains</em> with this setting do not execute any
87  * <em>Log Statement</em>.
88  */
89  Off
90 };

93 /** ************************************************************************************************
94  * These are definitions which are used as a parameter to certain \b %ALox methods to denote
95  * the \e Scope of a setting. \e Scopes are dependent of the programming language
96  * and hence differ slightly from each other in the different versions of \b %ALox.
97  *
98  * This enumeration is an \alib{lang,T_EnumIsArithmetical,ALib arithmetical enum}. However,
99  * the addition of values is only allowed with the last element, \b Path. By adding integer
100  * values, the Nth parent directory of a source files' location are addressed. As an example,
101  * invocations like this are used to select the source directory two levels above the source
102  * code file for a prefix scope:
103  *
104  * lox->SetPrefix( "#> ", Scope::Path + 2 );
105  *
106  * \note
107  * \b %ALox for C++ implements scope mechanisms using scope information generated by the
108  * preprocessor.
109  * By default, debug logging supports such 'caller information', while release logging
110  * does not.<br>
111  * Therefore, in release-logging, the use of \e Scopes 'Path', 'Filename' and
112  * 'Method' will just default to an empty scope and therefore the all reflect the same,
113  * shared scope, which is not very helpful. Therefore, for standard release logging,
114  * the use of the scope mechanisms should be be avoided, unless scope information is
115  * explicitly enabled.<br>
116  * For more information on how to change the defaults, see \ref ALOX_DBG_LOG_CI and
117  * \ref ALOX_REL_LOG_CI in section \ref GrpALoxCodeSelectorSymbols.
118  * For more information on \e Scopes consult the [ALox user manual](../manual.html).
119  **************************************************************************************************/
120 enum class Scope
121 {
122  /** Denotes the global (singleton) scope */
123  Global,
124 
125  /**
126  * Denotes the actual thread as the scope. When used with <em>Scope Domains</em>,
127  * 'inner' scopes can be defined optionally by multiple definitions.
128  */
129  ThreadOuter,
130 
131  /** Denotes the actual source file as the scope*/
132  Filename,
133 
134  /** Denotes the actual method as the scope*/
135  Method,
136 
137  /** Denotes the actual thread as the scope. When used with <em>Scope Domains</em>,
138  * 'inner' scopes can be defined optionally by multiple definitions. */
139  ThreadInner,
140 
141  /**
142  * Denotes the actual source path as the scope. By adding positive integer values
143  * to this element (the enum type is an \alib{lang,T_EnumIsArithmetical,ALib arithmetical enum}),
144  * 'outer' \e Scopes of this scope level itself can be defined using parent directories
145  * of the path.
146  */
147  Path
148 };

151 /** ************************************************************************************************
152  * This class defines "escape sequences" that influence the formatting of log output.
153  * Specific implementations of class
154  * \ref aworx::lox::core::Logger "Logger"
155  * have to convert or interpret this classes definitions of escape sequences
156  * when processing log data. If no formatting of the output is supported by a specific Logger
157  * implementation, such logger should filter and discard escape sequences defined here.
158  *
159  * The sequences are similar to ANSI Escape sequences and logger classes that
160  * log to 'VT100' compatible terminals will simply convert them.
161  *
162  * The name of the class was intentionally chosen to be short, because the escape codes
163  * defined with this class will be concatenated to log strings like that:
164  *
165  * \snippet "ut_dox_manual.cpp" DOX_ALOX_ESC
166  *
167  * \note
168  * With the introduction of own, \b %ALox specific escape codes, software that uses ALox becomes
169  * independent from any underlying, platform-specific sequences. For example, \b %ALox is not relying
170  * on ANSI color codes, which are not supported by colorful Windows consoles. Instead, on each
171  * platform, dedicated Loggers will perform the translation of \b %ALox codes to platform-specific
172  * ones.
173 **************************************************************************************************/
