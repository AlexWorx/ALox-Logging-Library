71  /** ************************************************************************************
72  * Static method that adjusts a given region to fit into the range [0 .. referenceLength].
73  *
74  * @param referenceLength The length of the reference region to which the other parameters
75  * get adjusted to.
76  * @param regionStart The proposed region start which might get adjusted to fit to
77  * span 0 to referenceLength.
78  * @param regionLength The proposed region length which might get adjusted to fit to
79  * span 0 to referenceLength.
80  *
81  * @return Returns \c true, if the adjusted region is empty.
82  **************************************************************************************/
83  inline
84  static bool AdjustRegion( integer referenceLength, integer& regionStart, integer& regionLength )
85  {
86  // if start exceeds string, set to empty range at the end of the string and return true
87  if (regionStart >= referenceLength)
88  {
89  regionStart= referenceLength;
90  regionLength= 0;
91  return true; // indicate empty
92  }
93 
94  // if negative start, cut it from the length
95  if (regionStart < 0 )
96  {
97  regionLength+= regionStart; // this adds a negative value, hence we do not exceed a given MAXINT
98  regionStart= 0;
99  }
100 
101  // adjust length
102  integer maxLen= referenceLength - regionStart;
103  if ( regionLength > maxLen )
104  regionLength= maxLen;
105  if ( regionLength < 0 )
106  regionLength= 0;
107 
108  // return true if adjusted region is empty
109  return regionLength == 0;
110  }
