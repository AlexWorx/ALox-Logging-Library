72  /** ****************************************************************************************
73  * Template method that boxes and adds one value to this vector of boxes.
74  *
75  * If the given value is an instance of this class, then, instead of "boxing" the list,
76  * the values of that list are each added. The latter also works if the given value
77  * is a boxed instance of this class.
78  *
79  * @param src The value to be boxed and added.
80  *****************************************************************************************/
81  template <typename TBoxable>
82  inline void Add( TBoxable&& src )
83  {
84  ALIB_WARNINGS_ALLOW_TEMPLATE_META_PROGRAMMING
85 
86  // add another list of Boxes
87  using TPlain= typename std::remove_const <
88  typename std::remove_pointer <
89  typename std::remove_reference<
90  typename std::remove_const < TBoxable >::type>::type>::type>::type;
91 
92  //ALIB_TMP_SHOW_TYPE_IN_DEBUGGER(TPlain);
93 
94  if( std::is_same< Boxes, TPlain >::value )
95  {
96  Boxes* boxes= std::is_pointer<TBoxable>::value ? * (Boxes **) &src
97  : (Boxes * ) &src;
98  reserve( size() + boxes->size() );
99  for( Boxes::iterator it= boxes->begin() ; it != boxes->end() ; ++it )
100  push_back( *it );
101  return;
102  }
103  if( std::is_same< Box, TPlain >::value )
104  {
105  Box* box= std::is_pointer<TBoxable>::value ? * (Box **) &src
106  : (Box * ) &src;
107  if( box->IsType<Boxes*>() )
108  {
109  Boxes* boxes= box->Unbox<Boxes*>();
110  reserve( size() + boxes->size() );
111  for( Boxes::iterator it= boxes->begin() ; it != boxes->end() ; ++it )
112  push_back( *it );
113  return;
114  }
115  }
116 
117  // add a single object
118  emplace_back( std::forward<TBoxable>(src) );
119 
120  ALIB_WARNINGS_RESTORE
121  }
