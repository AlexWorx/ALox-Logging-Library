26 /** ************************************************************************************************
27  * A simple class template specializing
28  * \ref aworx::lib::strings::AString "AString". All it does, is providing an internal character
29  * array of \e 'templated' as the default buffer to \b %AString. If an instance of this class
30  * is allocated on the stack, allocation and deallocation is performed in \e O(1).
31  * If an instance of this class is allocated on the heap (using new), then still some performance
32  * improvements apply, because besides the class object itself, no second memory allocation for
33  * the string buffer is performed.
34  *
35  * If during the use of an instance of this class a larger buffer is needed, such buffer will get
36  * allocated from the heap. This means, the use of this class is safe and no restrictions apply.
37  * Of-course, for performance critical code sections, the predefined size should be chosen large
38  * enough to allow the internal buffer to survive the use.<br>
39  *
40  * In debug compilations of ALib, parent class #AString optionally raises an one-time warning
41  * (using
42  * \ref aworx::lib::lang::Report::DoReport "Report::DoReport") if an external buffer is replaced by a
43  * new (heap) allocation. (From an \b %AString perspective, this classes' internal buffer is an
44  * external one). For more information, see
45  * \ref aworx::lib::strings::AString::SetBuffer "AString::SetBuffer".<p>
46  *
47  * \note For commonly used sizes, some convenient typedefs exists in namespace aworx.
48  * (See \ref CPP_AWORX_NS_SHORTCUTS "Type Shortcuts of Namespace aworx".)
49  * Type definitions are #String16, #String32, #String64 ... #String1K, #String2K, #String4K.
50  * <p>
51  *
52  * \note The move constructor of this class will copy the buffer of the movable \b %AString or
53  * \b %PreallocatedString object rather, instead of moving it. This is to keep the statically allocated
54  * buffer in place. Only if the contents does fit into our TCapacity, the buffer is '\e stolen'
55  * (and an one-time warning is issued in debug compilations).<br>
56  * Like parent class \b %AString, this class does not provide move assignment. See
57  * \ref aworx::lib::strings::AString "AString" for the rational of this design decision.
58  *
59  * @tparam TCapacity The capacity of the buffer allocated inside the class with member #preAllocBuf.
60  **************************************************************************************************/

106  ******************************************************************************************/
107  PreallocatedString( PreallocatedString&& move) noexcept
108  : AString( preAllocBuf, TCapacity )
109  {

111  if ( move.Length() < TCapacity || !move.HasInternalBuffer() )
112  {

211  /** ****************************************************************************************
212  * The <em>default C++ assign operator</em> (<code>ClassT & operator=(const ClassT &)</code>).<br>
213  * Copies the contents of the given object \p copy.
214  * Other assignment operators are inherited (with <em>using</em> statement) from
215  * parent \b %AString.
216  *
217  * @param copy The object to copy the contents from.
218  * @return \c *this to allow concatenated calls.
219  ******************************************************************************************/
220  PreallocatedString<TCapacity>& operator= (const PreallocatedString& copy)
221  {
222  if ( copy.IsNull())
223  {
224  SetNull();
225  return *this;
226  }
227  Clear();
228  return static_cast<PreallocatedString<TCapacity>&>( Append<false>( copy.Buffer(), copy.Length() ) );
229  }
