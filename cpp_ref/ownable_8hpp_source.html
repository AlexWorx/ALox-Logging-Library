78 /** ************************************************************************************************
79  * This abstract class represents objects that can be owned by some other instances.
80  * 'To own' means an abstract concept of getting 'acquired' and getting 'released' later on.
81  * The obvious and well known example are objects that implement a 'mutex' to lock data against
82  * concurrent threads.<br>
83  * The pattern of having ownership on ownable objects is usable in other areas as well.
84  * The C++ language provides the concept of "stack unwinding", which ensures that all objects are
85  * destructed, even when exceptions are thrown (that is why C++ does not have a 'finally'
86  * keyword).<br>
87  * See class \ref Owner and macro \ref OWN for information on how to 'own' an Ownable automatically
88  * for exactly one block of code.
89  *
90  * In debug compilations, the class supports information about the source location where ownership
91  * was gained.
92  **************************************************************************************************/
93 class Ownable
94 {
95  protected:
96  #if ALIB_DEBUG
97  /** Debug information on acquirement location */
98  strings::TString acquirementSourcefile =nullptr;
99 
100  /** Debug information on acquirement location */
101  int acquirementLineNumber;
102 
103  /** Debug information on acquirement location */
104  strings::TString acquirementMethodName =nullptr;
105  #endif
106 
107  public:
108  /** ****************************************************************************************
109  * Virtual destructor
110  *****************************************************************************************/
111  virtual ~Ownable() {}
112 
113  /** ****************************************************************************************
114  * Virtual abstract method. Functionality due to implementation in descendants.
115  * @param file Caller information. Available only in debug compilations.
116  * @param line Caller information. Available only in debug compilations.
117  * @param func Caller information. Available only in debug compilations.
118  *****************************************************************************************/
119  #if ALIB_DEBUG
120  virtual void Acquire(const TString& file, int line, const TString& func)
121  {
122  acquirementSourcefile= file;
