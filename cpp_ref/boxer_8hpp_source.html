36 /**
37  * This class is used internally with <b>ALib %Boxing</b> to provide runtime type information and
38  * virtual method invocations to boxed values.
39  * A singleton instance of a derived type of this class is attached to each
40  * \ref aworx::lib::boxing::Box "Box". Such derived type is generated using templated
41  * derived class
42  * \ref aworx::lib::boxing::BoxerT "BoxerT<TBoxed>", respectively, in case of array types, class
43  * \ref aworx::lib::boxing::ArrayBoxerT "ArrayBoxerT<TBoxed>".
44  */

216  /**
217  * Adds the given interface object to the list of default interfaces. Default interfaces
218  * are kept in a static collection and are used with boxable types that have no
219  * type-specific interface set.
220  *
221  * Adding a default interface is optional. As a sample, ALib provides interface
222  * \ref aworx::lib::strings::boxing::IApply "IApply" to write a string
223  * representation of an object. Although, being the base to type specific interfaces,
224  * this class itself provides a default implementation for \e 'unknown' types,
225  * writing the runtime type name and memory address of the object.
226  * That 'base' version of the interface is registered with this method.
227  *
228  * Default interfaces are not detectable with method
229  * \ref aworx::lib::boxing::Box::HasInterface "Box::HasInterface", but of-course they are
230  * used with
231  * \ref aworx::lib::boxing::Box::Invoke "Box::Invoke" and
232  * \ref aworx::lib::boxing::Box::GetInterface "Box::GetInterface" in the moment no
233  * type-specific interface is available.
234  *
235  * \note
236  * This method should not be called directly. It is advised to use namespace function
237  * \ref aworx::lib::boxing::DefineDefaultInterface "DefineDefaultInterface"
238  * instead.
239  *
240  * @param interface The singleton instance of the default interface class
241  */
242  static inline void DefineDefaultInterface( Interface* interface )
243  {
244  Boxer::defaultInterfaces[interface->typeInfo]= interface;
245  #if ALIB_DEBUG
246  Boxer::dbgKnownInterfaces[interface->typeInfo]= interface;
247  #endif
248  }
249 
250  /**
251  * Adds the given interface object to the list of interfaces of this boxer. This way,
252  * method
253  * \ref aworx::lib::boxing::Box::HasInterface "Box::HasInterface" will return true for
254  * the provided interface base type and methods
255  * \ref aworx::lib::boxing::Box::Invoke "Box::Invoke" and
256  * \ref aworx::lib::boxing::Box::GetInterface "Box::GetInterface" will use, respectively
257  * return the given singleton instance.
258  *
259  * \note
260  * This method should not be called directly. It is advised to use namespace function
261  * \ref aworx::lib::boxing::DefineInterface "DefineInterface", respectively,
262  * to define interfaces.
263  *
264  * @param interface The singleton instance of the default interface class
265  */
266  inline void DefineInterface( Interface* interface)
267  {
268  interfaces[interface->typeInfo]= interface;
269  #if ALIB_DEBUG
270  Boxer::dbgKnownInterfaces[interface->typeInfo]= interface;
271  Boxer::dbgKnownInterfaceImpl[typeid(*interface)]= interface;
272  #endif
273  }
274 
275  /** ****************************************************************************************
276  * Returns \c true if this boxer represents boxed array types. In this case, method
277  * \ref aworx::lib::boxing::Box::Length "Box::Length"
278  * will return the length of the array and
279  * \ref aworx::lib::boxing::Box::UnboxElement "Box::UnboxElement" may be used to
280  * access elements of the array.
281  *
282  * @return \c true if this boxer is representing an array of custom types.
283  ******************************************************************************************/
284  inline bool IsArray() const { return type != elemType; }
