35 /**
36  * This class is used internally with <b>%ALib %Boxing</b> to provide runtime type information and
37  * virtual method invocations to boxed values.
38  * A singleton instance of a derived type of this class is attached to each
39  * \ref aworx::lib::boxing::Box "Box". Such derived type is generated using templated
40  * derived class
41  * \ref aworx::lib::boxing::BoxerT "BoxerT<TBoxed>", respectively, in case of array types, class
42  * \ref aworx::lib::boxing::ArrayBoxerT "ArrayBoxerT<TBoxed>".
43  */
44 class Boxer
45 {
46  friend class Boxing; ///< A friend class.
47  friend class Box; ///< A friend class.

50  // Protected fields
51  // #############################################################################################
52  protected:
53  /** Information about the encapsulated type. In case of arrays, this represents the
54  * boxer type, while the element type is provided in #elemType. */
55  const std::type_info& type;

58  const std::type_info& elemType;

76  protected:
77  /**

79  * which has to represent the type of a derived class.

81  * Array types need to use the alternative constructor.

118  // #############################################################################################
119  // Interface
120  // #############################################################################################
121  public:
122  /**
123  * Adds the given interface object to the list of default interfaces. Default interfaces
124  * are kept in a static collection and are used with boxable types that have no
125  * type-specific interface set.
126  *
127  * Adding a default interface is optional. As a sample, \b %ALib provides interface
128  * \ref aworx::lib::strings::boxing::IApply "IApply" to write a string
129  * representation of an object. Although, being the base to type specific interfaces,
130  * this class itself provides a default implementation for \e 'unknown' types,
131  * writing the runtime type name and memory address of the object.
132  * That 'base' version of the interface is registered with this method.
133  *
134  * Default interfaces are not detectable with method
135  * \ref aworx::lib::boxing::Box::HasInterface "Box::HasInterface", but of-course they are
136  * used with
137  * \ref aworx::lib::boxing::Box::Invoke "Box::Invoke" and
138  * \ref aworx::lib::boxing::Box::GetInterface "Box::GetInterface" in the moment no
139  * type-specific interface is available.
140  *
141  * \note
142  * This method should not be called directly. It is advised to use namespace function
143  * \ref aworx::lib::boxing::DefineDefaultInterface "DefineDefaultInterface"
144  * instead.
145  *
146  * @param interface The singleton instance of the default interface class
147  */
148  static inline void DefineDefaultInterface( Interface* interface )
149  {
150  Boxer::defaultInterfaces[interface->typeInfo]= interface;
151  ALIB_DBG( BOXING.dbgKnownInterfaces[interface->typeInfo]= interface; )
152  }
153 
154  /**
155  * Adds the given interface object to the list of interfaces of this boxer. This way,
156  * method
157  * \ref aworx::lib::boxing::Box::HasInterface "Box::HasInterface" will return true for
158  * the provided interface base type and methods
159  * \ref aworx::lib::boxing::Box::Invoke "Box::Invoke" and
160  * \ref aworx::lib::boxing::Box::GetInterface "Box::GetInterface" will use, respectively
161  * return the given singleton instance.
162  *
163  * \note
164  * This method should not be called directly. It is advised to use namespace function
165  * \ref aworx::lib::boxing::DefineInterface "DefineInterface", respectively,
166  * to define interfaces.
167  *
168  * @param interface The singleton instance of the default interface class
169  */
170  inline void DefineInterface( Interface* interface)
171  {
172  ALIB_DBG(
173  if( interfaces.find( interface->typeInfo ) != interfaces.end() )
174  {
175  ALIB_MESSAGE( "ALib Boxing: Replacing interface ",
176  debug::TypeDemangler(interface->typeInfo).Get(),
177  " for type ",
178  debug::TypeDemangler(type).Get() )
179  }
180  )
181 
182  interfaces.insert(std::make_pair(std::reference_wrapper<const std::type_info>(interface->typeInfo), interface) );
183 
184  ALIB_DBG(
185  BOXING.dbgKnownInterfaces [interface->typeInfo]= interface;
186  BOXING.dbgKnownInterfaceImpl[typeid(*interface) ]= interface;
187  )
188  }
189 
190  /** ****************************************************************************************
191  * Returns \c true if this boxer represents boxed array types. In this case, method
192  * \ref aworx::lib::boxing::Box::Length "Box::Length"
193  * will return the length of the array and
194  * \ref aworx::lib::boxing::Box::UnboxElement "Box::UnboxElement" may be used to
195  * access elements of the array.
196  *
197  * @return \c true if this boxer is representing an array of custom types.
198  ******************************************************************************************/
199  inline bool IsArray() const { return type != elemType; }
200 };
