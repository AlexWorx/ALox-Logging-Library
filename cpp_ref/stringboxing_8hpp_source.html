229 /**
230  * This interface class exposes interface method \c Invoke which writes the content of the
231  * box to the given \b %AString object.
232  * This base interface class, implements a default behaviour for all boxable types which are not
233  * equipped with a specific implementation of this class and hence, a singleton hereof is
234  * attached to the <b>%%ALib %Boxing</b> using
235  * \ref aworx::lib::boxing::Boxer::DefineDefaultInterface "Boxer::DefineDefaultInterface"
236  *
237  * Variant
238  * \ref aworx::lib::strings::boxing::IApply_TApplicable "IApply_TApplicable"
239  * can be attached without changes to any Boxer of custom types that implement the TMP interface
240  * introduced by \ref aworx::lib::strings::AString::Apply "AString::Apply".
241  */
242 class IApply : public Interface
243 {
244  public:
245  /**
246  * Declares a static singleton used for the default interface registration and returns that.
247  * \note We do not inherit template class Singleton here, because each of our descendants
248  * needs to do this as well, which would lead to some confusion.
249  * @return A singleton instance of myself.
250  */
251  static IApply* GetDefaultSingleton()
252  {
253  static IApply me;
254  return &me;
255  }
256 
257  /** Constructor providing our runtime type to the parent. */
258  IApply() : Interface( typeid(IApply) ) {}
259 
260  /**
261  * The interface method. This default implementation writes out the raw value of the box
262  * in hexadecimal format. For standard types, such raw value reflect the memory address of
263  * the object boxed.<br>
264  * In debug-compilations, in addition the type name of the boxed value is written.
265  * @param box The box that the interface was invoked on.
266  * @param target The target \b %AString.
267  */
268  ALIB_API
269  virtual void Invoke( const Box& box, AString& target );
270 };

277  * information.
278  *
