28 /** ************************************************************************************************
29  * Simple class that allocates a series of bigger memory blocks and provides sequential
30  * allocation of sub-ranges of those. Furthermore, the allocations can be reset and the
31  * allocated blocks reused.
32  *
33  * The size of the blocks allocated is defined with constructor parameter \p pSize. If an invocation
34  * to #GetMemory is requesting a bigger allocation, then a new block with that given size is
35  * created.
36  *
37  * \note
38  * This class is new with ALib. It is not considered finished, optimized and stable in design.
39  **************************************************************************************************/
40 class MemoryBlocks
41 {
42  // #############################################################################################
43  // inner types
44  // #############################################################################################
45  protected:
46  /** ****************************************************************************************
47  * Internal type defining an allocated memory chunk.
48  ******************************************************************************************/
49  struct Block
50  {
51  char* mem; ///< The allocated memory
52  integer size; ///< The size of the block
53  integer fill; ///< The current fill
54 
55  /** ************************************************************************************
56  * Constructor allocating the given \p pSize of memory.
57  * This memory is not deallocated with a destructor, therefore
58  * objects of this type need to be deleted manually.
59  *
60  * @param pSize The size of memory chunk to allocate
61  **************************************************************************************/
62  inline Block( integer pSize )
63  {
64  mem= new char[static_cast<size_t>(pSize)];
65  size= pSize;
66  fill= 0;
67  }
68 
69  /** ************************************************************************************
70  * Returns \c nullptr if the requested size is not available in the block.
71  * Otherwise, the pointer to the now allocated memory inside the block is returned.
72  *
73  * @param requestedSize The size to allocate.
74  * @return \c nullptr on failure, otherwise pointer to the requested memory.
75  **************************************************************************************/
76  inline char* Allocate( integer requestedSize )
77  {
78  if( size-fill < requestedSize )
79  return nullptr;
80 
81  char* result= mem + fill;
82  fill+= requestedSize;
83  return result;
84  }
85  };
