31 /** ************************************************************************************************
32  * Simple class that allocates a series of bigger memory blocks and provides sequential
33  * allocation of sub-ranges of those. Furthermore, the allocations can be reset and the
34  * allocated blocks reused.
35  *
36  * The size of the blocks allocated is defined with constructor parameter \p pSize. If an invocation
37  * to #GetMemory is requesting a bigger allocation, then a new block with that given size is
38  * created.
39  *
40  * \note
41  * This class is new with \b %ALib. It is not considered finished, optimized and stable in design.
42  **************************************************************************************************/
43 class MemoryBlocks
44 {
45  // #############################################################################################
46  // inner types
47  // #############################################################################################
48  protected:
49  /** ****************************************************************************************
50  * Internal type defining an allocated memory chunk.
51  ******************************************************************************************/
52  struct Block
53  {
54  char* mem; ///< The allocated memory
55  integer size; ///< The size of the block
56  integer fill; ///< The current fill
57 
58  /** ************************************************************************************
59  * Constructor allocating the given \p pSize of memory.
60  * This memory is not deallocated with a destructor, therefore
61  * objects of this type need to be deleted manually.
62  *
63  * @param pSize The size of memory chunk to allocate
64  **************************************************************************************/
65  inline Block( integer pSize )
66  : size(pSize)
67  , fill(0)
68  {
69  mem= new char[static_cast<size_t>(pSize)];
70  }
71 
72  /** ************************************************************************************
73  * Returns \c nullptr if the requested size is not available in the block.
74  * Otherwise, the pointer to the now allocated memory inside the block is returned.
75  *
76  * @param requestedSize The size to allocate.
77  * @return \c nullptr on failure, otherwise pointer to the requested memory.
78  **************************************************************************************/
79  inline char* Allocate( integer requestedSize )
80  {
81  if( size-fill < requestedSize )
82  return nullptr;
83 
84  char* result= mem + fill;
85  fill+= requestedSize;
86  return result;
87  }
88  };
