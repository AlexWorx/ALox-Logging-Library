44 /** ************************************************************************************************
45  * This class implements a map suitable for keys that are hierarchically organized.
46  * While the class works with any type of keys, it is efficient only if keys are rather
47  * long and similar to each other.
48  * Each portion of any key inserted is stored only once.
49  *
50  * \note
51  * This class is not considered as complete and final in the current version of %ALib.
52  * The functionality currently is restricted to the needs of %ALox logging library.
53  **************************************************************************************************/

146  /** ****************************************************************************************
147  * Gets a node. If not existent and parameter \p create is \c true, the node is created.
148  * @param key The key to the stored value.
149  * @param create Flag if a non-existent entry should be created.
150  * @param separators A list of characters recognized as separators.
151  * @return Returns the ourselves or a child node representing the key string.
152  ******************************************************************************************/
153  PathMap* get ( Substring& key, bool create, const TString& separators )
154  {
155  // find the index of the match length and consume these characters from the key
156  integer idx= 0;
157  integer pLen= Path.Length();
158  if ( pLen > 0 )
159  {
160  integer cmpLen= pLen < key.Length() ? pLen : key.Length();
161  const char* kBuf= key.Buffer();
162  const char* pBuf= Path.Buffer();
163 
164  while( idx < cmpLen && *(kBuf + idx) == *(pBuf +idx ) )
165  idx++;
166 
167  key.ConsumeChars<false>( idx );
168  }
169 
170  // all of 'our' path characters matched
171  if( idx == pLen)
172  {
173  // identical to the searched string?
174  if( key.IsEmpty() )
175  return this;
176 
177  // return matching child
178  for ( auto child : Children )
179  {
180  if( key.CharAtStart<false>() == child->Path.CharAtStart() )
181  {
182  PathMap* search= child->get( key, create, separators );
183  if ( search != nullptr )
184  return search;
185  }
186  }
187 
188  // no child found
189  if(create)
190  {
191  PathMap* newChild= nullptr;
192  newChild= new PathMap( this );
193  newChild->Path._( key );
194  Children.emplace_back( newChild );
195  return newChild;
196  }
197  }
198 
199  // nothing matched
200  else if ( idx == 0 )
201  return nullptr;
202 
203 
204  // just a part of us matched
205  else if ( create )
206  {
207  // create new child receiving our old path (rest), our value and children
208  PathMap* child1= new PathMap( this );
209  child1->Path._( Path, idx );
210 
211  for( auto child : Children )
212  child->Parent= child1;
213 
214  child1->Children= Children;
215  child1->Value= Value;
216  Children.clear();
217  Children.emplace_back( child1 );
218 
219  // cut my path and clear my value
220  Path.SetLength<false>( idx );
221  Value= 0;
222 
223  // create second child if remaining path is not empty
224  if( key.IsNotEmpty() )
225  {
226  PathMap* child2= new PathMap( this );
227  child2->Path._( key );
228 
229  Children.emplace_back( child2 );
230  return child2;
231  }
232 
233  return this;
234  }
235 
236  // return if this is the root or a real node
237  if( Parent == nullptr
238  || separators.IndexOf( Path.CharAt<false>( idx - 1 ) ) >= 0
239  )
240  return this;
241 
242  return Parent;
243  }

283  /** Comparison operator.
284  * @param other The object to compare ourselves to.
285  * @return \c true if this and given iterator are equal, \c false otherwise. */

287  {
288  if ( nodesAndChildren.size() == 0 || other.nodesAndChildren.size() == 0)
289  return nodesAndChildren.size() == other.nodesAndChildren.size();

294  }

311  /** Increment operator.
312  * @return A reference to ourselves. */
313  Iterator& operator++()
314  {
315  while(nodesAndChildren.size() > 0 )
316  {
317  int child= ++nodesAndChildren.top().childNo;
318 
319  if ( child >= static_cast<int>( nodesAndChildren.top().node->Children.size() ) )
320  {

323  }
324  else if ( child >= 0 )
325  {
326  nodesAndChildren.push( NodeAndChild( nodesAndChildren.top().node->Children[static_cast<size_t>(child)], -1 ) );
327  }
328 
329  if ( nodesAndChildren.top().node->Value != 0 )

331  }
