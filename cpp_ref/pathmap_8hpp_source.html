155  PathMap* get ( Substring& key, bool create, const TString& separators )
156  {
157  // find the index of the match length and consume these characters from the key
158  int idx= 0;
159  int pLen= Path.Length();
160  if ( pLen > 0 )
161  {
162  int cmpLen= pLen < key.Length() ? pLen : key.Length();
163  const char* kBuf= key.Buffer();
164  const char* pBuf= Path.Buffer();
165 
166  while( idx < cmpLen && *(kBuf + idx) == *(pBuf +idx ) )
167  idx++;
168 
169  key.Consume<false>( idx );
170  }
171 
172  // all of 'our' path characters matched
173  if( idx == pLen)
174  {
175  // identical to the searched string?
176  if( key.IsEmpty() )
177  return this;
178 
179  // return matching child
180  for ( auto child : Childs )
181  {
182  if( key.CharAtStart<false>() == child->Path.CharAtStart() )
183  {
184  PathMap* search= child->get( key, create, separators );
185  if ( search != nullptr )
186  return search;
187  }
188  }
189 
190  // no child found
191  if(create)
192  {
193  PathMap* newChild= nullptr;
194  newChild= new PathMap( this );
195  newChild->Path._( key );
196  Childs.emplace_back( newChild );
197  return newChild;
198  }
199  }
200 
201  // nothing matched
202  else if ( idx == 0 )
203  return nullptr;
204 
205 
206  // just a part of us matched
207  else if ( create )
208  {
209  // create new child receiving our old path (rest), our value and childs
210  PathMap* child1= new PathMap( this );
211  child1->Path._( Path, idx );
212 
213  for( auto child : Childs )
214  child->Parent= child1;
215 
216  child1->Childs= Childs;
217  child1->Value= Value;
218  Childs.clear();
219  Childs.emplace_back( child1 );
220 
221  // cut my path and clear my value
222  Path.SetLength<false>( idx );
223  Value= 0;
224 
225  // create second child if remaining path is not empty
226  if( key.IsNotEmpty() )
227  {
228  PathMap* child2= new PathMap( this );
229  child2->Path._( key );
230 
231  Childs.emplace_back( child2 );
232  return child2;
233  }
234 
235  return this;
236  }
237 
238  // return us, or if this is not a real node, our parent
239  if( Parent == nullptr || idx == 0 || separators.IndexOf( Path.CharAt<false>( idx - 1 ) ) >= 0 )
240  return this;
241  else
242  return Parent;
243  }
